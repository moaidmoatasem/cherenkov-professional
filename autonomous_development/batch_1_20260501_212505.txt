import requests
from typing import List

class CVEDatabaseConnector:
    """This class provides methods to fetch vulnerabilities from the CVE database."""
    
    def __init__(self):
        self.base_url = "https://services.nvd.nist.gov/api/v1/cves"
    
    def get_vulnerabilities(self, query_parameters: dict) -> dict:
        """
        Fetches cve data based on provided parameters.

        :param query_parameters: A dictionary containing the query parameters.
                                  Example: {'keyword': 'sql'}
        :return: The response payload from the NVD API.
        """
        full_url = f"{self.base_url}/content/{query_parameters.get('keyword')}"
        headers = {"accept": "application/json"}
        
        try:
            response = requests.get(full_url, params=query_parameters, headers=headers)
            response.raise_for_status()  # Raise an HTTPError if the request failed
            return response.json()
        except requests.RequestException as e:
            raise Exception(f"Request to NVD API failed: {e}")
    
    def search_cves(self, query_keyword: str) -> List[int]:
        """
        Queries and returns a list of CVE IDs for a given keyword.

        :param query_keyword: The keyword parameter used in the NVD API.
        :return: A list of CVE IDs for the specified topic or keyword.
        """
        payload = {'keyword': query_keyword}
        response_data = self.get_vulnerabilities(payload)
        cve_ids = [cve['cve']['id'] for cve in response_data]
        
        return cve_ids

def main():
    """Example usage of querying CVE database."""
    connector = CVEDatabaseConnector()
    
    # Example: Search for vulnerabilities related to SQL
    sql_vulnerabilities = connector.search_cves(query_keyword='sql')
    
    print("SQL Vulnerability CVE IDs:")
    print(sql_vulnerabilities)
