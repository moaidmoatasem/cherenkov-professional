#!/usr/bin/env python3

"""
Purpose:
Implement and showcase smart retry logic that handles partial failures, retries on specific error codes,
and limits the total retries for robustness.
"""

import requests
from urllib3.exceptions import ProtocolError, ConnectTimeoutError, PoolTimeout

class SmartRetryLogic:
    def __init__(self, max_retries=3):
        """Initialize the retry manager with a maximum number of retries."""
        self.max_retries = max_retries
        self.sleep_intervals = [1, 2, 4]

    def make_request(self, url, method='GET', params=None, headers=None):
        """
        Make an HTTP request, retry on specific errors, and limit the total retries.

        Args:
            url (str): The URL to send the request to.
            method (str): The type of HTTP request (e.g., 'GET', 'POST').
            params (dict or None): Parameters for the request.
            headers (dict or None): Headers for the request.

        Returns:
            requests.Response: Response object from the server.
        """
        session = requests.Session()
        retries = 0
        while True:
            try:
                # Define any custom settings here as needed, e.g. timeout, proxies, redirect settings etc.
                response = session.request(method=method, url=url, params=params, headers=headers)
                response.raise_for_status()  # Thrown if an HTTP error occurred
                return response
            except (ProtocolError, ConnectTimeoutError, PoolTimeout):
                retries += 1
                self._log_error(f'Non-fatal error encountered on attempt {retries}: {response}')
                if retries <= self.max_retries:
                    delay = self.sleep_intervals[retries - 1]
                    self._log_timeout_with_warning(delay)
                    import time; time.sleep(delay)  # Sleep is in seconds.
                else:
                    raise Exception('Max retries exceeded')
            except BaseException as e:  # Catch-all to handle any unknown errors
                self._log_error(f'Unexpected error on attempt {retries}: {str(e)}')
                raise

    def _log_error(self, message):
        """Log an error message for the user."""
        # Implementing logging mechanism for error messages
        print(message)

    def _log_timeout_with_warning(self, delay):
        """Log a timeout message with a warning."""
        self._log_error(f'Retry will happen after {delay} seconds')


def main():
    api_url = 'http://example.com/api'
    response = SmartRetryLogic().make_request(api_url)
    assert response is not None and response.ok, "Expected valid response but got an error."
    print('Request successful.')


if __name__ == "__main__":
    smart_retry_logic_instance = SmartRetryLogic(max_retries=3)  # Use this instance in your application
    main()
