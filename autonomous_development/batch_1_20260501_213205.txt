import logging
import unittest


class AttackChainFeatureError(Exception):
    pass


class AttackChainFeature:
    def __init__(self, sequences=None, threshold=2):
        """
        Initialize the attack chain detector.

        Args:
            sequences (list): A list of event sequences to identify as potential attacks.
            threshold (int): The minimum number of consecutive occurrences for a sequence to be marked as an attack.

        Raises:
            AttackChainFeatureError: If sequences is not provided or empty, or if threshold <= 0
        """
        if sequences is None or len(sequences) == 0:
            raise AttackChainFeatureError('Sequences must be specified and non-empty.')
        
        self.sequences = set(sequences)
        self.threshold = threshold

    def _check_threshold_positive(self):
        """Raise an attack chain feature error if the threshold is <= 0."""
        if self.threshold <= 0:
            raise AttackChainFeatureError('Threshold must be greater than 0.')

    def detect_attack_sequence(self, events: list) -> bool:
        """
        Detects if any of the sequences in `sequences` are encountered consecutively more times than `threshold`.

        Args:
            events (list): A list of event occurrences to scan for attack sequences.

        Returns:
            bool: True if an attack sequence is detected, False otherwise.
        """
        self._check_threshold_positive()
        
        hit_count = {sequence: 0 for sequence in self.sequences}
        consecutive_events = []
        
        for idx, event in enumerate(events):
            consecutive_events.append(event)
            
            if event in self.sequences:
                valid_sequence = ' '.join(consecutive_events)
                
                if any(hit >= hit_count.get(valid_sequence, 0) + 1 
                      and consecutive_events[-hit_count[valid_sequence]:] == list(self.sequences)[i].split(' ')
                      for i, valid_sequence in enumerate(self.sequences)):

                    hits = sum(event in self.sequences or (self.threshold < len(consecutive_events) - idx)
                               for event in consecutive_events[:idx])
                    
                    if hits >= self.threshold:
                        return True
                else: # New sequence starts.
                    consecutive_events = [event]
            elif self.threshold and len(events) > 1 and consecutive_events: # Reset sequences longer than threshold on a break.
                consecutive_events.clear()
        return False



class TestAttackChainFeature(unittest.TestCase):
    def setUp(self):
        self.feature = AttackChainFeature(['login', 'read_file', 'execute'])
        
    def test_valid_sequence_detection(self):
        """Test detecting a valid sequence in the correct order."""
        events = ['login', 'read_file', 'write_data']
        self.assertTrue(self.feature.detect_attack_sequence(events))
    
    def test_invalid_sequences(self):
        """Test invalid sequences don't trigger alerts."""
        events = [1, 2, 3]
        with self.assertRaises(AttackChainFeatureError):
            self.feature.detect_attack_sequence(events)
        
    def test_threshold_behavior(self):
        """Test threshold value detection logic."""
        feature = AttackChainFeature(['login', 'read_file'], threshold=1)
        self.assertFalse(feature.detect_attack_sequence(['login']))
    
    def test_multiple_sequences_detection(self):
        """Simulate multiple sequences but one triggering the alert, others not.
        
            In this case, only login->execute hits the threshold once it's been detected three times consecutively
        """
        events = ['login', 'read_file', 'execute', 'run_code']
        self.assertFalse(self.feature.detect_attack_sequence(events))
    
    def test_large_number_of_events_threshold_not_triggered(self):
        """Test number of events is much less than the threshold."""
        feature = AttackChainFeature([], threshold=10)
        self.assertFalse(feature.detect_attack_chain(['event']))
