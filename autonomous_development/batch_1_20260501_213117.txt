import json
from collections import defaultdict
from datetime import datetime
from flask import Flask, request, jsonify

app = Flask(__name__)

class VulnerabilityDatabase:
    def __init__(self):
        self.db_path = 'vulnerable_systems.json'
        self._load_db()
    
    def _load_db(self):
        try: 
            with open(self.db_path, 'r') as db_file:
                self.vulnerabilities = json.load(db_file)
        except FileNotFoundError:
            self.vulnerabilities = defaultdict(set)

    def store_vulnerability(self, system_ip, vulnerability_id):
        now = datetime.now().strftime("%d-%m-%Y %H:%M:%S")
        entry = {'time': now, 'vuln_id': vulnerability_id}
        self.vulnerabilities[str(system_ip)].add(entry)
        
        with open(self.db_path, "w") as db_file:
            json.dump(dict(sorted(self.vulnerabilities)), db_file, indent=4)

    def get_vulnerability_history(self, system_ip):
        return sorted(self.vulnerabilities.get(str(system_ip), []), key=lambda x: x['time'], reverse=True)
