import requests

class SlackConnector:
    """A class to connect with Slack and perform actions like sending messages."""

    def __init__(self, slack_token):
        """
        Initialize the SlackConnector with a provided token.
        
        :param slack_token: Token for authenticating with Slack API.
        """
        self.slack_api_url = "https://slack.com/api"
        self.token = slack_token

    def send_message(self, channel_name, message_content):
        """
        Send a message to the specified channel in Slack.

        :param channel_name: Name of the Slack channel to post to.
        :param message_content: The actual content of the message.
        :return: None
        """
        url = f"{self.slack_api_url}/chat.postMessage"
        data = {
            "token": self.token,
            "channel": self._get_slack_channel_id(channel_name),
            "text": message_content
        }
        response = requests.post(url, data=data)
        
        if response.status_code != 200:
            raise Exception(f"Failed to send message: {response.text}")

    def _get_slack_channel_id(self, channel_name):
        """
        Fetches the ID of a Slack channel by name.

        :param channel_name: Name of the Slack channel.
        :return: Id of the given Slack channel or None if not found.
        """
        url = f"{self.slack_api_url}/channels.list"
        data = {"token": self.token}
        
        response = requests.post(url, data=data)
        channels_list = response.json().get("channels", [])
        
        for channel in channels_list:
            if channel['name'] == channel_name and channel['is_member']:
                return channel["id"]
        
        raise Exception(f"Channel not found: {channel_name}")

def example_usage():
    """Example usage of SlackConnector."""
    token = "your_slack_api_token"
    connector = SlackConnector(token)
    
    # Assuming you want to post a message to the 'general' channel
    connector.send_message("general", "Hello, this is a test message!")
