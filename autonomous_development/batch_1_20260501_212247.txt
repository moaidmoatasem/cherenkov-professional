#!/usr/bin/env python3

import sys
from PyPDF2 import PdfWriter, PdfReader
from datetime import datetime


class ReportGenerator:
    """
    A class responsible for creating PDF reports with error handling.
    """

    def generate_report(self, data):
        """
        Method to generate a report and save it as a PDF file.

        :param data: List of dictionaries. Each dictionary represents a line in the report
        """

        # Check if input is provided
        if not isinstance(data, list) or not all(isinstance(item, dict) for item in data):
            raise ValueError("Data must be a list of dictionaries")

        # Create a new PDF file to write content into it.
        pdf = PdfWriter()
        
        with open('report.pdf', 'wb') as output_pdf:
            for row in data:
                if not all(key in row for key in ['name', 'status']):
                    raise ValueError("Each dictionary must contain 'name' and 'status'")
                
                # Write text to the PDF file
                pdf_obj = PdfReader(output_pdf).pages[0]
                content = f"Name: {row['name']}\nStatus: {row['status']}"
                pdf.add_page(pdf_obj)
                pdf.pages[-1].text = content

                pdf.write(output_pdf)

        # Print success message after report is written
        print(f'Report saved as "report.pdf".')
        
    def parse_and_create_report(self, source_file_path):
        """
        Method to read a text file and create reports.

        :param source_file_path: Path to the input file. Each line consists of an object in format: <name>,<status> separated by comma.
        """

        if not source_file_path.endswith('.txt'):
            raise ValueError("Source path should point to .txt file")
        
        with open(source_file_path, 'r') as file:
            data = [d.split(',') for d in file.readlines()]
            return self.generate_report(data)

    def test(self):
        """
        Example usage and test of the `generate_report` method.
        """
        try:
            # Test data
            report_data = [
                {"name": "Alice", "status": "Approved"},

            ]
            
            # Generate a simple report from the data
            self.parse_and_create_report("test_input.txt")
            
            
            print(f'Tests passed, no errors during execution.')
        except Exception as e:
            print(e)

def main():
    """
    Main function to drive the program's flow.
    
    The `ParseAndCreateReport` object takes a source file path and uses it to generate reports. The test method is also called for verification of functionality.
    """
    generator = ReportGenerator()

if __name__ == "__main__":
    main()
