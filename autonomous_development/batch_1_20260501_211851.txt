import subprocess

def command_injection_scanner(code_snippet):
    """
    This function is designed to scan for potential command injection vulnerabilities in code snippets.
    
    Parameters:
        - code_snippet (str): The code snippet that contains the potentially vulnerable call to a shell command.
        
    Returns:
        - str: A warning message indicating whether there are any suspicious patterns. 
               If no issues are found, it returns an empty string.
               
    Example Usage:
        result = command_injection_scanner('subprocess.call(["{}"])'.format(input_command))
        if 'Potential Command Injection' in result:
            print("Security Alert: Command injection detected!")
    """
    
    # Define a dictionary of suspicious patterns for command injections
    suspicious_patterns = [
        ['shell_command'],  # Base case where a raw shell command is used without sanitization
        ['os.popen', r"'.*'"],  # os.popen() function with dynamic input
        ['subprocess.Popen.*', '.*winexe.*']]  # Windows-specific: Popen with executable containing "winexe"
    ]
    
    for pattern in suspicious_patterns:
        for item_to_find in pattern:
            if isinstance(item_to_find, list):  # It might be a regex or string pattern within the list
                pattern_to_search = '|'.join(map(re.escape, item_to_find))
                if re.search(pattern_to_search, code_snippet):
                    return "Potential Command Injection: Suspicious code found"
            elif isinstance(item_to_find, str):  # Otherwise it's just looking for a literal string match
                if item_to_find in code_snippet:
                    return "Potential Command Injection: Suspicious code found"      
    
    # No suspicious behavior detected
    return ""

def test_command_injection_scanner():
    """
    Function to test the command_injection_scanner function with various scenarios.
    """
    print("Testing command injection scanner...")
    assert all([command_injection_scanner("'{}'".format(command)) == ""
                for command in [
                    "subprocess.call('echo Hello World')",
                    "%wscript%\shtcmt.exe"
                ]]), "Test 1 failed."
    
    assert command_injection_scanner("'system 'net users' > test.txt'") != "", \
           "Test 2 failed: No warning for suspicious pattern detected."
    
    assert all([command_injection_scanner(code) == ""
                for code in [
                    r"subprocess.call(['echo', 'Hello World'])",
                    "%wscript%\shtcmt.exe"
                ]] + 
               [("'{}'".format(input_command) for input_command in ["net users", "calc.exe"]])
                ), "Test 3 failed: False positive found where no injection is present."
    
    print("All tests passed.")

if __name__ == "__main__":
    test_command_injection_scanner()
