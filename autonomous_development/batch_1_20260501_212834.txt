# Import necessary libraries
import subprocess
from typing import List

class Scanner:
    """
    A class to run a scanner tool and capture its output.
    
    The function execute() runs the command specified by 'command' in a shell,
    capturing the error stream as well. It returns the stdout, stderr, and exit status.
    It also handles any exceptions raised during execution of the command.

    :param str command: Command to be executed externally
    """
    def execute(self) -> List[str]:
        try:
            # Splitting command into list for subprocess.run
            cmd = ['./scan_tool']
            result = subprocess.run(
                cmd, 
                check=True,  # Ensure the command is successful
                capture_output=True,
                text=True,   # Capture output as text
                encoding='utf-8',
                stderr=subprocess.PIPE
            )
            
            stdout = result.stdout.strip()
            stderr = result.stderr.strip()
            status_code = result.returncode
            
            return [stdout, stderr, status_code]
        
        except subprocess.CalledProcessError as e:
            err_message = f"Command: '{' '.join(e.cmd)} failed with error code {e.returncode}:"
            if e.stdout:  # If stdout contains output that is relevant to the command
                err_message += '\n' + e.stdout.strip()
                
            if e.stderr:  # If stderr contains a meaningful message (not only errors/warnings)
                err_message += f"\nGot an error:\n{e.stderr}"
            
            raise RuntimeError(err_message) from e

def example_usage() -> None:
    """
    Example usage of the Scanner class within an actual CI/CD pipeline context.

    It illustrates how to use the execute function and handle outputs.
    """
    scanner = Scanner()
    outputs = scanner.execute()

        print("No output received from the scanner.")
    
    if not outputs[1]:  # Similarly for stderr
        print(f"Error log: {outputs[1]}\nExecution might have failed.")
