def perform_function_with_smart_retry(func_to_call):
    """
    This function performs a specified function (func_to_call) and handles errors by implementing a smart retry mechanism.
    The retry mechanism will attempt to execute func_to_call up to a maximum number of retries defined in the environment. 
    After each successful execution, or if reaching the maximum retries limit, it will return the result.

    :param func_to_call: Function to be retried
    :return: Result from func_to_call

    Example Usage:
    result = perform_function_with_smart_retry(my_safe_function)
    # 'result' contains the response from my_safe_function or an error message on failure.
    """

    import os
    
    retry_count = int(os.environ.get('SMART_RETRY_MAX_RETRIES', 5))
    
    for attempt in range(retry_count + 1):
        try:
            return func_to_call()
        
        except Exception as e:
            if attempt < retry_count:
                # Log the error and wait before trying again
                print(f"Attempt {attempt} failed, retrying with a sleep of {os.environ.get('SMART_RETRY_SLEEP', 2)} seconds...")
                import time
                
                # Wait for another attempt to execute
                time.sleep(int(os.environ.get('SMART_RETRY_SLEEP', 2)))
            else:
                raise Exception(f"The '{func_to_call.__name__}' function has been retried the maximum number of times and failed, giving up.")
