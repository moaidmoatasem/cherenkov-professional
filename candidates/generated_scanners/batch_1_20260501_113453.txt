```python
import xml.etree.ElementTree as ET

class XxeScanner:
    """
    A class for scanning XML documents to detect and report potential XXE (XML External Entity Injection) vulnerabilities.
    
    Attributes:
        input_data: The XML data to be scanned.
    """

    def __init__(self, input_data):
        self.input_data = input_data
        self.scan_results = []

    def parse_xml(self):
        """
        Parses the provided XML data using xml.etree.ElementTree. 
        In case of an error during parsing, raises an exception.
        """
        try:
            # ElementTree can fail quietly and not raise under most errors, like encoding issues
            ET.fromstring(self.input_data)
        except ET.ParseError as e:
            self.scan_results.append(f"XMLParseError: {e}")
            return False

    def scan_for_xxe(self):
        """
        Scans the XML document for potential XXE vulnerabilities. 
        Modifies self.scan_results with relevant findings.
        """
        if not self.parse_xml():
            return
        # Here would follow complex logic for detecting XXE, this mock uses simple heuristic that looks for a string in an external entity reference
        self.findings = []
        root = ET.fromstring(self.input_data)
        for elem in root.iter():
            if "http:://" in str(elem.attrib.get("href", "")):
                self.findings.append(f"Potential XXE found: {elem.attrib['href']}. It appears to reference a URL!")
        
    def report_results(self):
        """
        Prints out the list of potential XXE vulnerabilities found.
        """
        for result in sorted(self.scan_results, key=lambda x: x if isinstance(x, Exception) else (x + "\n")):
            print(result.strip())

def main():
    """
    Example usage of XxeScanner.

    ```
    <xml_data>

           <name>XxeAttack1</name>

    ```
    scanner.scan_for_xxe()
    scanner.report_results()
```
