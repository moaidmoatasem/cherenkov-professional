#!/usr/bin/env python3

import os
from typing import Union

class ExecutiveSummaryGenerator:
    """
    Generates executive summaries based on provided documents.
    
    Attributes:
        document_input (str): Path to the document file.
    """
    
    def __init__(self, document_input: str):
        self.document_input = document_input
        
    @staticmethod
    def validate_document(document_path: str) -> Union[bool, None]:
        if not os.path.exists(document_path):
            print(f"Error: {document_path} does not exist.")
            return False
        return True

    def generate_summary(self) -> None:
        """
        Generates an executive summary from the provided document.
        
        Raises:
            RuntimeError: If validation or generation fails.
        """

        if not self.validate_document(self.document_input):
            raise RuntimeError("Cannot proceed without a valid document")

        try:
            with open(self.document_input, 'r', encoding='utf-8') as file:
                data = file.read()
                
                # Placeholder content for example purposes
                summary_content = (
                    f"Summary for Document: {self.document_input}\n"
                    f"{data[:100]}... (Content cropped due to length constraints)"
                )

                with open(os.path.splitext(self.document_input)[0]+'_executive_summary.txt', 'w') as output_file:
                    output_file.write(summary_content)
        except Exception as err:
            raise RuntimeError(f"Error generating summary: {err}") from None

# Example Usage
def main():
    document_path = input("Please enter the path to the document: ").strip()
    generator = ExecutiveSummaryGenerator(document_input=document_path)
    
    try:
        print("Generating executive summary...")
        generator.generate_summary()
        print("Executive summary has been generated!")
    except Exception as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    main()

# Example Tests
