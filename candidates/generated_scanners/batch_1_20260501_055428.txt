import json

class FeatureScanner:
    def check_features(self):
        """Check whether the system has necessary features installed."""
        if self._features is None:
            raise ValueError("Features not initialized. "
                              "Please initialize the feature scanner first.")
        return self._features


class CreateExecutiveSummaryGenerator:

    EXECUTIVE_SUMMARY_TEMPLATE = {
        'title': '',
        'summary': '',
        'details': []
    }

    def __init__(self, features: FeatureScanner):
        self.features = features

    def generate_summary(self, title: str) -> dict:
        """
        Generates an executive summary with default settings.

        :param title: Title of the summary
        :return: Generated executive summary as a dictionary
        """
        try:
            if not self.features.check_features():
                raise ValueError("Scanning features failed. System may be corrupt.")
            
            summary = self.EXECUTIVE_SUMMARY_TEMPLATE.copy();
            summary['title'] = title
            summary['summary'] = f"Here is an executive summary for {title}."
            summary['details'].append({"name": "Feature 1", "status": "Up"})
            summary['details'].append({"name": "Feature 2", "status": "Down"})
            
        except Exception as e:
            raise ValueError(f"Génération de résumé d'exécutive : {str(e)}")
        
        return summary

    def generate_summary_with_features(self, title: str, features: list) -> dict:
        """
        Generates an executive summary with specific feature details.

        :param title: Title of the summary
        :param features: List of features and their status ('Up' / 'Down')
        :return: Generated executive summary as a dictionary
        """

        try:
            if not self.features.check_features():
                raise ValueError("Scanning features failed. System may be corrupt.")
            
            summary = self.EXECUTIVE_SUMMARY_TEMPLATE.copy();
            summary['title'] = title

            for index, feature in enumerate(features):
                summary['details'].append({"name": f"Feature {index}", "status": feature})

            summary['summary'] = f"This is an executive summary for title {title}. It lists specific features and their status."
            
        except Exception as e:
            raise ValueError(f"Génération de résumé d'exécutive : {str(e)}")
        
        return summary


if __name__ == "__main__":
    import sys

    if len(sys.argv) < 2:
        raise ValueError("Error: Please provide a title for the executive summary.")
    
    feature_scanner = FeatureScanner()
    executor = CreateExecutiveSummaryGenerator(feature_scanner)

    # Generate summary with default settings
    custom_summary = executor.generate_summary(title=sys.argv[1])
    print(json.dumps(custom_summary, indent=4))
    
    if len(sys.argv) >= 3:
        feature_status_list = []
        try: 
            for i in range(2, len(sys.argv)):
                feature_status_list.append(sys.argv[i])
            summary_feature_details = executor.generate_summary_with_features(title=sys.argv[1], features=feature_status_list)
            print(json.dumps(summary_feature_details, indent=4))
        
        except Exception as e:
            sys.exit(f"Génération de résumé d'exécutive : {str(e)}\n")
