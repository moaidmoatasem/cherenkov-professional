import os

class PathTraversalScanner:
    """
    PathTraversalScanner class performs file path traversal checks to identify potential security vulnerabilities.
    
    Methods:
        run(path: str) -> None:
            Executes a full check on the given directory for any likely security path traversal flaws.
            
            Args:
                path (str): The root directory path to scan. E.g. '/var/www/html'.
                
            Returns:
                None
    """
    
    def __init__(self):
        self.FOUND_ERROR = False
    
    def run(self, path: str) -> bool:
        """Run the scanner on a given path, checks for path traversal flaws."""
        try:
            if not os.path.exists(path) or not os.path.isdir(path):
                raise ValueError("The provided path does not exist as a directory.")
            
            self._check_path_traversal_issues(os.listdir(path))
        except Exception as e:
            print(f"Error during scanning: {e}")
        
        return self.FOUND_ERROR
    
    def _check_path_traversal_issues(self, current_dir):
        """Recursive helper function to check for potential path traversal."""
        if not isinstance(current_dir, list) or len(current_dir) == 0:
            raise ValueError("Could not gather items from provided directory.")
        
        for item in current_dir:
            relative_item = os.path.join('/', *current_dir)
            full_path = os.path.join(relative_item, item)
            
            self._check_for_directories_and_files(full_path)
        
    def _check_for_directories_and_files(self, path_to_check: str) -> None:
        """Helper function for checking directories and files in a given path."""
        try:
            if not os.path.exists(path_to_check):
                raise FileNotFoundError(f"Path does not exist: {path_to_check}")
            
            item = os.stat(path_to_check)
            mode = oct(item.st_mode)[-3:]
            
            # Check for potentially harmful file types
            if 'w' in mode or 'x' in mode:
                self.FOUND_ERROR = True
                print(f"Path Traversal flaw found at: {path_to_check}")
        
        except Exception as e:
            pass  # We do not want to halt the scan on any errors
        
    def reset(self) -> None:
        """Reset the scanner for repeated use."""
        self.FOUND_ERROR = False

if __name__ == "__main__":
