import datetime
from PyPDF2 import PdfWriter

class PDFReportGenerator:
    def __init__(self):
        self.report = None
    
    @staticmethod
    def generate_report(data):
        """
        Generates a report in PDF format.
        
        Parameters:
            data (dict): Data to include in the report. Should be well-formatted, suitable for inclusion in a document.
            
        Returns:
            str: The filename of the generated PDF report.
        """
        today = datetime.date.today()
        filename = f"report_{today.year}{today.month}{today.day}.pdf"
        with open(filename, "wb") as output_file:
            pdf_writer = PdfWriter()
            for key, value in data.items():
                page = PdfWriter.add_blank_page(pdf_writer)
                
                # Example of adding text to the page
                # PyPDF2 does not support complex PDF structures like tables or images natively.
                # Hence we only add basic placeholder: "Data: {key}"
                pdf_writer.add_text(page, f"Data: {key}")
                pdf_writer.add_text(page, str(value))
            pdf_writer.write(output_file)
        return filename
    
    @staticmethod
    def handle_errors():
        """
        Handles errors in the report generation process.
        
        Raises:
            ValueError: If the input data is not a dictionary or doesn't contain 'data'.
        """
        if not isinstance(data := dict(), dict):
            raise ValueError("Input must be a dictionary.")
            
        # Example of handling missing key error
        if "data" not in data:
            raise KeyError("Data subdictionary is mandatory.")
