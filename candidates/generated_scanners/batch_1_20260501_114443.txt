# Import necessary libraries
import os
import json

class VulnerabilitySeverityScorer:
    """
    This class is responsible for scoring vulnerabilities based on their CVSS V3 scores.
    
    It contains methods to initialize the scrapper, score vulnerability severity,
    and validate input data as required by the security tool.
    """

    def __init__(self):
        self.cve_db_path = os.path.join(os.path.dirname(__file__), "cve_database.json")
        
    @staticmethod
    def _load_cve_data():
        with open(VulnerabilitySeverityScorer.cve_db_path, 'r') as file:
            return json.load(file)
    
    def score_vulnerabilities(self, cves):
        """
        Scores vulnerabilities based on their CVSS V3 scores.

        Args:
            cves (list): A list of dictionaries containing vulnerability information.

        Returns:
            dict: A dictionary with a severity score for each CVE.
        """
        score = {}
        
        loaded_cve_data = self._load_cve_data()
        for cve in cves:
            # Check if the CVE is present in our database
            if str(cve.get('CVE', 'NoRecord')) not in loaded_cve_data:
                return {"score": "N/A", "message": f"No record found for {cve['ID']}"}
            
            cvss_v3_score = self._get_cvss_v3_from_db(loaded_cve_data, str(cve.get('CVE', 'NoRecord')))
            cve['cvssV3Score'] = cvss_v3_score

            # Assign a severity based on the CVSS V3 score
            if 0.5 <= float(cvss_v3_score) < 4:
                score[cve['ID']] = "Low"
            elif 4 <= float(cvss_v3_score) < 7:
                score[cve['ID']] = "Medium"
            elif 7 <= float(cvss_v3_score) < 9.5:
                score[cve['ID']] = "High"
            else:
                score[cve['ID']] = "Critical"
        
        return {"score": score, "message": f"Scores have been calculated for all CVEs."}
    
    def _get_cvss_v3_from_db(self, loaded_cve_data, cve_id):
        """
        Retrieves the CVSS V3 score from loaded CVE data.

        Args:
            loaded_cve_data (dict): The database of vulnerability information.
            cve_id (str): The ID of the CVE to fetch the CVSS V3 score for.

        Returns:
            str: The CVSS V3 score corresponding to the CVE ID.
        """
        if 'scores' not in loaded_cve_data.get(cve_id, {}):
            raise ValueError(f"No scores found for {cve_id}")
        
        return loaded_cve_data[cve_id]['scores']['cvssv3'].get('score', "N/A")
    
    def _validate_input(self, cves):
        """
        Validates input: Check that cves is a list of dictionaries with 'ID' and 'CVSS V3 score data'.

        Args:
            cves (list): The provided CVEs.

        Returns:
            bool: True if the data is valid, False otherwise.
        """
        for cve in cves:
            if not isinstance(cve, dict) or 'ID' not in cve or 'CVSSV3Score' not in cve:
                return False
        return True
    
    def __call__(self, cves):
        """
        Allows the VulnerabilitySeverityScorer instance to be called like a function.
        
        Args:
            cves (list of dicts): List of dictionaries representing CVEs.

        Returns:
            dict: A dictionary containing the severity score for each CVE and an error message if necessary.
        """
        
        if not self._validate_input(cves):
            return {"score": None, "message": "Invalid input data provided."}
        
        try:
            severity_score = self.score_vulnerabilities(cves)
            return severity_score
        except Exception as e:
            return {"score": f"{type(e).__name__}: {str(e)}", "message": str(e)}

if __name__ == "__main__":
