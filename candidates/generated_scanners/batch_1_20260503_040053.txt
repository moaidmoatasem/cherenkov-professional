import sys

from typing import List, Dict, Any
import pdfkit

@dataclass
 class Reporter:
    """
    This is a simple PDF report generator. It takes data and generates a PDF document.
    The data can be structured in any way (e.g., as a list of dictionaries) and should be provided to the __call__ method.

    Attributes:
        path_to_css (str): Path to CSS file containing styles for better formatting.
    """
    path_to_css: str

    def __post_init__(self):
        if not pdfkit.is_patched():
            raise EnvironmentError("wkhtmltopdf must be properly installed and patched.")

    def generate_pdf(self, data_dict_list: List[Dict[str, Any]]):
        """
        Generates a PDF report using the provided data.

        Args:
            data_dict_list (List[Dict[str, Any]]): A list of dictionaries where each dictionary represents an entry with keys for different sections.
        """
        css = f'.{self.path_to_css}'
        
        options = {
            'page-size': 'Letter',
            'margin-top': '0.75cm',
            'margin-right': '1cm',
            'margin-bottom': '0.75cm',
            'margin-left': '1cm',
            'encoding': "UTF-8",
            'no-outline': None,
            'footer-font-size': '8pt',
            'css': css
        }

        config = pdfkit.configuration(wkhtmltopdf='C:\\Program Files\\wkhtmltopdf\\bin\\wkhtmltopdf.exe')
        
        for section_data in data_dict_list:
            data = {
                'page_title': f'Report - {section_data.get("title", "")}',
                'content_html': self._build_section(section_data)
            }
            pdfkit.from_string(pdfkit.HTML(self.generate_pdf_header()), data['page_title'], options=options, configuration=config)

    def generate_pdf_header(self):
        """
        Generates the header of the PDF document.
        Returns:
            str: HTML content for the header section.
        """ 
        return '<h1>Security Report</h1><p>Date: ' + pdfkit.from_url('https://www.google.com', options={'footer-font-size': '8pt'}) + '</p><hr>'

    def _build_section(self, data_dict):
        """
        Builds the HTML content for a single section based on provided data.

        Args:
            data_dict (Dict[str, Any]): A dictionary containing keys and values for this section.
        
        Returns:
            str: The complete HTML snippet for this section.
        """ 
        return '<h2>' + data_dict.get('title', '') + '</h2>\n' + 
                '<p>Summary: ' + data_dict.get('summary') + '\n</p>\n' +  
                '<table><tbody>\n' +
                self._build_table_data() +
                '</tbody></table>\n'

    def _build_table_data(self, header_elements: List[str] = None):
        """
        Generates the table content for a given set of columns.

        Args:
            header_elements (List[str], optional): A list containing headers strings. Defaults to None.
        
        Returns:
            str: The HTML snippet for the table section.
        """ 
        if not header_elements:
            return '\n'.join(['<tr><td></td>' for _ in range(1, 4)])
            
        elements = ['<' + e + '/' '>' for e in header_elements]
        elements.append('</tr>\n')
        
        content_str = '<table style="border: solid">\n' 
        for row in self._load_data_to_table():
            line_content = []
            for element, el_content in zip(elements, row):
                inner_element = ''
                if 'colspan' in element:
                    col_span = int(float(element.split(' ')[-1]))
                    inner_element += f'<td rowspan={col_span} '
                else: 
                    inner_element += '<td '
                
                header_str = str(el_content)
                colspan_str = f' colspan="{len(header_elements) - 1}"'

                if 'class' in element:
                    class_str = f'class="{header_str.split(";")[0]}"'                   
                else: 
                   class_str = ''

                inner_element += f" {inner_element}{' '.join([colspan_str, class_str])}"
                
                line_content.append(header_str + el_content)
            row_content = '<tr>'.join(line_content) + '</tr>\n'
            content_str += row_content
        return ''.join((content_str, '</table>'))
