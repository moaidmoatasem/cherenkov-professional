import requests

# Constants for CVE API endpoints
CVE_API_URL = "https://services.nvd.nist.gov/api/v1/cves/feeds/json"
NVD_API_KEY = "<YOUR_API_KEY>"  # Replace with your actual NVD API key if required

def get_cve_data_from_api():
    """
    Fetches JSON data from the CVE API.

    Returns:
        dict: The response from the CVE API.
    """
    
    headers = {
        "Accept": "application/json",
        "X-Nvd-Apikey": NVD_API_KEY,
    }

    try:
        response = requests.get(CVE_API_URL, headers=headers)
    except requests.exceptions.RequestException as e:
        raise Exception(f"Failed to fetch CVE data: {e}")

    if not response.ok:
        error_message = f"API call failed with status code {response.status_code}"
        if 400 <= response.status_code < 600:
            raise Exception(error_message)
        
        details = response.json().get("error", None)
        if isinstance(details, list):
            for detail in details:
                print(f"{error_message}: {detail['message']}")
        else:
            raise Exception(detail)

    try:
        return response.json()
    except ValueError as e:
        raise Exception(f"Invalid JSON received from the API: {e}")

def get_cves_with_severity(data):
    """
    Parses the CVE data to extract cve entries with severity.

    Args:
        data (dict): The raw JSON data retrieved from the API.

    Returns:
        list[dict]: A list of dictionaries containing CVE information.
    """

    try:
        filtered CVEs = [cve for cve in data if 'severity' in cve and isinstance(cve['severity'], dict)]
        return sorted(filtered CVEs, key=lambda x: x['cvssV3']['baseScore'], reverse=True)
    except (KeyError, TypeError) as e:
        raise Exception(f"Data parsing error: {e}")

def main():
    """
    Entry point to fetch and display information via the API for all CVEs with severity.
    """

    try:
        cve_data = get_cve_data_from_api()
        filteredCVEs = get_cves_with_severity(cve_data)
        
        print("CVE Entries sorted by Severity:")
        for i, cve in enumerate(filteredCVEs):
            # Example way to use the data - further functions or logging here
            print(f"CVE ID: {cve['cve']['id']} | Severity: {cve['severity'].get('name', '.')}, CVSS:V3 Score={cve['cvssV3']['baseScore']:.2f}")
            
    except Exception as e:
        print(f"An error occurred during execution: {e}")
