import requests
from urllib.parse import urljoin

class SlackDiscordIntegrationManager:
    def __init__(self, slack_webhook_url, discord_api_url):
        self.slack_webhook_url = urljoin(slack_webhook_url)
        self.discord_api_url = urljoin(self.build_path_from_root("/webhooks"), discord_api_url)
    
    @staticmethod
    def build_path_from_root(path):
        # This method assumes you have the base URL in a variable named root_url. 
        # You should replace it with your actual server root URL.
        """Builds the absolute path from the provided relative path."""
        return path.replace(root_url, '')

    def send_message_to_slack(self, message_data):
        headers = {'content-type': 'application/json'}
        try:
            response = requests.post(self.slack_webhook_url, json=message_data, headers=headers)
            if response.status_code not in [200, 202]:
                raise Exception(f"Error sending data to Slack: {response.text}")
            print("Message sent to Slack successfully.")
        except requests.RequestException as e:
            raise SystemError(f'Failed to send message to Slack. Error: {e}')
    
    def post_to_discord(self, message_data):
        try:
            response = requests.post(self.discord_api_url.format(message_data['message']), json=message_data)
            response.raise_for_status()
            print("Message posted to Discord successfully.")
        except requests.RequestException as e:
            raise SystemError(f'Failed to send message to Discord. Error: {e}')
