"""
VulnerabilityCorrelationEngine - Engine to correlate vulnerabilities across multiple systems and databases.
"""

from typing import List, Dict, Any
import requests
from abc import ABCMeta, abstractmethod


class VulnerabilityScanner(metaclass=AMeta):
    @abstractmethod
    def scan(self, identifier: str) -> None:
        """
        Scans the vulnerability for the given system or database identifier.

        :param identifier: A string uniquely identifying a system or database to be scanned.
        """
        raise NotImplementedError


class VulnerabilityDatabase:
    def __init__(self, engine: VulnerabilityScanner):
        self.engine = engine
        self.vulnerabilities: Dict[str, List] = {"system1": [], "system2": []}

    @property
    def all_vulnerabilities(self) -> List[Any]:
        return list(chain(*self.vulnerabilities.values()))

    def register_new_vulnerability(self, vulnerability_details: str, system_id: str):
        self.vulnerabilities[system_id].append(vulnerability_details)

class VulnerabilityCorrelationEngine:
    """Creates a correlation engine to correlate vulnerabilities across connected systems."""

    __slots__ = ["_scanners", "_db_connector"]

    def __init__(self, scanners: List[VulnerabilityScanner]):
        assert len(scanners) > 0, "At least one scanner must be defined."
        
        self._scanners = scanners if isinstance(scanners[0], VulnerabilityScanner) else None
        self._db_connector = VulnerabilityDatabase(engine=self._scanners[-1] if self._scanners else None)
    
    def _run_scanner(self, identifier: str):
        try:
            scanner = next(scanner for scanner in self._scanners if scanner.engine == self._db_connector.engine)
            return scanner.scan(identifier=identifier)
        except StopIteration:
            raise AttributeError(f"Scanner not found or incompatible with database engine. No vulnerabilities reported.")
   
    def correlate_and_register(self, identifiers: List[str]):
        """
        Correlates and registers new vulnerabilities from the identified systems/databases.

        :param identifiers: A list of system or database identifiers to scan.
        """
        
        for identifier in identifiers:
            try:
                self._run_scanner(identifier)
        except requests.exceptions.RequestException as e:
            print(f"Request error occurred: {e}")

    def report_vulnerabilities(self) -> List[Any]:
        '''
        Retrieves all registered vulnerabilities from the database.

        :return: A list of vulnerabilities (strings) reported by connected scanners.
        '''
        
        return self._db_connector.all_vulnerabilities
