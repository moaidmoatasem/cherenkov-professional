#!/usr/bin/env python3

# Import necessary libraries and modules
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import classification_report, confusion_matrix
import mlflow
import mlflow.sklearn

class MLVulnerabilityPredictor:
    """
    A class for implementing an ML-based vulnerability prediction model.
    
    Attributes:
        data (pd.DataFrame): The dataset used to train and test the model.
        feature_columns (list): List of column names representing features.
        target_column (str): Column name representing vulnerabilities.
        random_state (int): Seed for DataFrame's constructor.
        mlflow_tracking_uri (str): URI where MLFlow tracking server is hosted.
    """
    
    def __init__(self, data: pd.DataFrame, feature_columns=None,
                 target_column="vulnerability", random_state=42, mlflow_tracking_uri=""):
        self.data = data
        if feature_columns:
            self.feature_columns = feature_columns
        else:
            self.feature_columns = [col for col in data.columns if col != target_column]
        self.target_column = target_column
        self.random_state = random_state
        
        # Enable MLFlow tracking
        mlflow.set_tracking_uri(mlflow_tracking_uri)
        
    def run(self):
        """Run the training and testing of the vulnerability prediction model."""
        features, labels = self._prepare_data()
        estimator = RandomForestClassifier(n_estimators=100, random_state=self.random_state)
        X_train, X_test, y_train, y_test = train_test_split(features, labels, test_size=0.33, random_state=self.random_state)
        
        # Train the model
        with mlflow.start_run():
            estimator.fit(X_train, y_train)

            # Evaluate the model
            predictions = estimator.predict(X_test)
        
            print("Model Evaluation Results:")
            print(classification_report(y_test, predictions))
            print(confusion_matrix(y_test, predictions))

        # Save model in MLFlow artifact registry for easier access and reusability.
        mlflow.sklearn.log_model(estimator, "model")
        
        return estimator
    
    def _prepare_data(self):
        """Prepare feature matrix (X) and target response vector y for the training process."""
        if isinstance(self.data, pd.DataFrame):
            features_df = self.data[self.feature_columns]
            labels = self.data[self.target_column].values
        else:
            raise ValueError("Invalid data format: Please pass a Pandas DataFrame.")
        
        return features_df, labels

if __name__ == "__main__":
    # Example usage of the VulnerabilityPredictor class
    
    # Load your dataset. This can be any .csv file.
    data_path = "path_to_your_dataset.csv"
    df_data = pd.read_csv(data_path)
    
    target_column = ['vulnerability']
    feature_columns = ['feature1', 'feature2']  # Replace this with actual columns names
    
    predictor = MLVulnerabilityPredictor(df_data, target_column=target_column, feature_columns=feature_columns)
    model = predictor.run()
