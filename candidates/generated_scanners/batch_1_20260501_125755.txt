```python
from flask import Flask, jsonify, request
import logging

app = Flask(__name__)
logging.basicConfig(level=logging.INFO)

# Mocked Scanner Functionality
def simple_scanner(target):
    # Simulate scanning behavior here.
    # This could involve actual network scanning or directory traversal logic
    if target == "example.com":
        raise ValueError("Example domain is not allowed.")
    
    return f"Scanning: {target}"

# Database-like structure for storing scanner results (mocked in-memory)
scan_results = {}

@app.route('/scans', methods=['POST'])
def add_scan():
    try:
        # Get the target from JSON body
        json_data = request.get_json()
        if not json_data or 'target' not in json_data:
            raise KeyError("Target is required.")
        
        target = json_data['target']
        
        scan_result = simple_scanner(target)
        logging.info(f"Scanner added for {json_data}")
        
        # Store the result
        scan_results[target] = scan_result
        
        return jsonify({"message": f"Scan of {target} completed successfully."}), 201
    
    except KeyError as e:
        return jsonify({"error": str(e)}), 400
    except Exception as e:
        logging.error(f"Unexpected error: {str(e)}")
        return jsonify({"error": "Internal Server Error"}), 500

@app.route('/scans/<target>', methods=['GET'])
def get_scan(target):
    if target not in scan_results:
        return jsonify({"error": f"No scan record for {target}"}), 404
    return jsonify(scan_results[target])

@app.after_request
def log_response_time(response):
    logging.info(f"Request: {request.method} to /{request.path}, Status Code: {response.status_code}")
    return response

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000, debug=True)
```
