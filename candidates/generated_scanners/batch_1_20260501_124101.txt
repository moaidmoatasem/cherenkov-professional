# import necessary libraries
import os
from PyPDF2 import PdfWriter, PdfReader  # Ensure that PyPDF2 is installed as pip install PyPDF2

class PDFReportGenerator:
    def __init__(self):
        self.report = []
    
    def add_data(self, data: str) -> None:
        """
        Adds a string of text to the current report
        :param data: The content you want to add to the report
        :type data: str
        """
        if not isinstance(data, str):
            raise TypeError("Data must be a string")
        self.report.append(f"Page: {self._page_counter} -> {data}")
        self._print_status()
        
    def save_as_pdf(self, output_path: str) -> None:
        """
        Saves the current report as a PDF file specified by `output_path`
        :param output_path: Path to which the PDF should be saved (with .pdf extension)
        :type output_path: str
        """
        if not os.path.isfile(output_path):
            raise FileNotFoundError(f"Output path {output_path} is invalid or does not point to a file")
        
        writer = PdfWriter()
        for item in self.report:
            writer.add_page(PdfReader(item).pages[0])
            
        with open(output_path, "wb") as f:
            writer.write(f)
    
    def _page_counter(self) -> int:
        """ Internal function that returns the number of pages."""
        return len(self.report)
    
    def _print_status(self):
        print("Printing Status", self.report)


def example_usage():
    # Create a PDFReportGenerator object
    report_generator = PDFReportGenerator()
    
    # Add text to the report
    report_generator.add_data("This is some initial data.")
    
    # Call a method that would read more content and add it.
    # Here for demonstration, we assume we called another function or code block:
    report_generator.add_data('Subsequent additions...')
    
    # Save the PDF
    output_path = 'output/report.pdf'
    report_generator.save_as_pdf(output_path)
    
    # Print saved location for confirmation
    print(f"PDF file has been created at: {os.path.abspath(output_path)}")


if __name__ == "__main__":
    example_usage()
