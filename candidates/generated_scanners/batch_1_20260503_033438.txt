#!/usr/bin/env python3

"""
XXE (XML External Entity) Scanner
===========================================

This script is designed to scan XML documents for potential XXE vulnerabilities.
It aims to identify potentially dangerous external entities in an XML document,
ensuring the scanner's output is sanitized and safe.

SYNOPSIS
---------
The XXE Scanner can be invoked with a single argument - 
XML file path. It checks if the internal entity "system" or "external"
have no defined URI, which could lead to XXE vulnerabilities.
If vulnerabilities are found, it sanitizes the input by removing all entities,
ignoring any potential issues with well-formedness compliance.

USAGE
-----
$ python xxescanner.py <xml_file_path>
"""

import re

def load_xml(xml_file):
    """
    Load an XML file and return a string of its content.
    
    :param xml_file: Path to the XML file to be loaded
    """
    with open(xml_file, 'r', encoding='utf-8') as f:
        return f.read()

def find_internal_entities(xml_content):
    """
    Search for internal entities in an XML document.

    This function looks specifically for 'system' and 'external'
    entity references. If these are found without a URI definition,
    it indicates potential XXE vulnerability.
    
    :param xml_content: The content of the XML file as a string
    :return: A list of tuples, where each tuple contains an internal entity name (tag) and its value
    """
    pattern = re.compile(r'<!\[CDATA\[(.*?)?\[CDATA\]')
    matches = pattern.findall(xml_content)
    
    if not matches:
        return []
    
    return [(el.group(1),  el.group(1))
            for el in pattern.finditer(xml_content) 
            if el.group(0).startswith('<!--') or (el.group(0)[2:5] == ']]>')]
 

def validate_xml(sanitized):
    """
    Validates the sanitized result to ensure it conforms to well-formed XML structure.

    This function ensures the input is a clean, non-well-formed string, suitable for further analysis.
    
    :param sanitized: The cleaned output of an entity scanning (e.g. the list returned by find_internal_entities).
                       Should contain only safe content without entity references.
    """
    # Basic check to ensure it contains no valid XML characters
    return not bool(sanitized)

def clean_xml_input(xml_content):
    """ 
    Remove all entities found in an XML document.

    This function sanitizes the input by stripping out known dangerous internal entities, 
    leaving only plain text content.
    
    :param xml_content: The XML content to sanitize as a string
    """
    for tag, value in find_internal_entities(xml_content):
        xml_content = xml_content.replace(f'<!{tag} [CDATA[{value}]]', f'<{tag}>{re.escape(value)}</{tag}>')
        
    return re.sub('<\!--.*?-->', '', xml_content.strip(), flags=re.S)


def xxescan(xml_file_path):    
    """
    Main entry point to run the XXE Scanner.
    
    This method parses the XML file provided as an argument,
    identifies internal entities that could lead to XXE vulnerabilities,
    and cleans up the content by removing these internal entities.
    The sanitized result is then returned for display or analysis.

    :param xml_file_path: Path to the XML document to be scanned
    """
    error_handling(error='XML file not found', does_exist=True, path=xml_file_path)
    
    if validate_xml(clean_xml_input(load_xml(xml_file_path))):
        print("Input is well-formed XML.")
    else:
        print("Input may contain unescaped HTML entities. Content might need further validation.")

    # Output the cleaned-up result which now contains only non-entity elements
    return clean_xml_input(load_xml(xml_file_path).strip())


def error_handling(error, does_exist=True, path=None):
    """
    Handle errors such as file not found or XML parsing issues.

    :param error: Error reason to be printed.
    :param does_exist: Bool stating whether the path did exist if True
    :param path: Path of the resource that failed validation (XML File)
    """
    
    if does_exist:
        raise ValueError(f'Could not open {path or "file"}: "{error}"')
    else:
        print(error)


if __name__ == "__main__":
