import requests
from flask import Flask, request

app = Flask(__name__)

def create_webhook_notification(url):
    """
    Creates and sends a webhook notification to the specified URL upon receiving POST requests with valid JSON data.

    Parameters:
        url (str): The endpoint of the webhook. This should be in an HTTPS or HTTP scheme.
    
    Returns:
        None. Sends notifications via webhooks.

    Raises:
        ValueError: If the provided URL is not a valid HTTPS or HTTP URL.
        
    Example usage:
    
        create_webhook_notification("https://example.com/webhook")
    """
    if 'http://' in url or 'https://' in url:
        try:
            # Listen for POST requests containing JSON data
            @app.route("/", methods=['POST'])
            def webhook():
                data = request.get_json()
                response_code, message = handle_notification(data)
                return "", response_code  # No content is expected
                
            app.run(host='0.0.0.0', port=5005)  # Run the Flask server on localhost and all IPs port 5005
        except TypeError:
            raise ValueError("URL must be a valid HTTPS or HTTP scheme URL.")
    else:
        raise ValueError("URL must be a valid HTTPS or HTTP scheme URL.")

def handle_notification(data):
    """
    Handles webhook notification by processing JSON data. 

    Parameters:
        data (dict): The incoming POST data as dict

    Returns:
         tuple: A response code and message. Response codes correspond to different statuses such as success, error, etc.
    
    Example Usage:

        # In a testing or production environment where you want to log errors for debugging
        handle_notification({
            "user_id": 1234,
            "timestamp": "2023-10-15T19:49:18+00:00"
        })

    Raises:
        Exception: If the processing fails due to invalid data or other unexpected conditions.
    
    """
    try:
        # Example of handling and validating incoming JSON data
        if all(key in data for key in ('user_id', 'timestamp')):
            response_code = 200
            message = "Notification processed successfully"
        else:
            response_code = 400
            message = "Required fields missing from the POST data."
    except Exception as e:
        # Log exception details before responding with an error code, if further processing is not needed in production
        print(f"Error processing notification: {e}")
        response_code = 502
        message = "Could not process webhook notification"
    
    return response_code, message

if __name__ == "__main__":
    create_webhook_notification("https://example.com/webhook")
