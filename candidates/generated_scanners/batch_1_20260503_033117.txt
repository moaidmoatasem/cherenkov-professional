#!/usr/bin/env python3

import os
import unittest


class PathTraversalScanner:
    def __init__(self):
        self.path_traversal_entries = ['/..']

    @staticmethod
    def is_path_valid(path):
        """
        Validates a path to ensure it does not exceed the allowed depth,
        contains only absolute paths, and excludes any known bad entries.

        :param path: Path string to validate.
        :return: True if valid, False otherwise.
        """
        return bool(PathTraversalScanner._is_absolute(path) and not PathTraversalScanner._is_path_traversal(path))

    @staticmethod
    def _is_absolute(path):
        """Checks if the given path is absolute."""
        try:
            return os.path.isabs(path)
        except TypeError:
            return False

    @staticmethod
    def _is_path_traversal(path):
        """Scans the path for path traversal attacks by checking for '..', '/' in the full path resolution process."""
        components = PathTraversalScanner._path_components(path) if isinstance(path, str) else [path]
        depth = 0
        for part in components:
            normalized_part = os.path.normcase(part)
            is_dotdot = normalized_part.startswith(os.path.sep + '..' * depth)
            depth += (is_dotdot or normalized_part == os.path.sep)
            if depth > len(PathTraversalScanner.path_traversal_entries):
                return True
        return False

    @staticmethod
    def _path_components(path):
        """Breaks path into components and removes any '.' or 'curdir' equivalents."""
        return [os.path.normcase(p) for p in os.path.normpath(path).split(os.path.sep)]

if __name__ == '__main__':
    unittest.main()

# Example usage of PathTraversalScanner
def main():
    scanner = PathTraversalScanner()
    examples = ['/etc/passwd', '/..//etc/README.txt', 'file.txt']
    for example in examples:
        print(f'Checking: {example}... Valid: {scanner.is_path_valid(example)}')

if __name__ == '__main__':
    main()
