from flask import Flask, request, jsonify
from werkzeug.security import generate_password_hash, check_password_hash

app = Flask(__name__)

# Mock Data: Users and Scanners in JSON format for example purposes.
SCANNERS_DATABASE = {
    "123456": {"username": "scanneruser", "password_hash": generate_password_hash("hunter2")}
}

@app.route("/api/v1/scanners/<scanner_id>", methods=["GET"])
def get_scanner_data(scanner_id):
    """
    Get details of a specific scanner.
    
    :param scanner_id: The unique identifier for the scanner.
    :return: JSON object with scanner data. If not found, it returns an error message.
    """
    if scanner_id in SCANNERS_DATABASE:
        return jsonify(SCANNERS_DATABASE[scanner_id]), 200
    else:
        return {"error": "Scanner not found"}, 404

@app.route("/api/v1/scanners", methods=["POST"])
def create_scanner():
    """
    Create a new scanner.
    
    :return: JSON object with the newly created scanner's details. On error, it returns an error message.
    """
    data = request.json
    if not data or 'username' not in data or 'password' not in data:
        return {"error": "Bad request - missing required parameters"}, 400
    
    hashed_password = generate_password_hash(data['password'])
    
    new_scanner_id = max(int(list(SCANNERS_DATABASE.keys())[-1]), 1) + 1
    SCANNERS_DATABASE[str(new_scanner_id)] = {
        "username": data["username"],
        "password_hash": hashed_password,
    }
    
    return jsonify({"id": new_scanner_id}), 201

@app.route("/api/v1/scanners/<scanner_id>", methods=["PUT"])
def put_scanner(scanner_id):
    """
    Update the details of a specific scanner.
    
    :param scanner_id: The unique identifier for the scanner to be updated.
    :return: Updated JSON object with the changed or existing scanner's details. If not found, it returns an error message.
    """
    if scanner_id in SCANNERS_DATABASE:
        data = request.json
        # Note: This is a minimal implementation and should not validate all fields for security reasons.
        if 'username' in data and 'password' in data:
            hashed_password = generate_password_hash(data['password'])
            SCANNERS_DATABASE[scanner_id]['password_hash'] = hashed_password
            
            return jsonify(SCANNERS_DATABASE[scanner_id]), 200
    
    return {"error": "Scanner not found"}, 404

@app.route("/api/v1/scanners/<scnr_uid>", methods=["DELETE"])
def delete_scanner(scnr_uid):
    """
    Delete a specific scanner.
    
    :param scanner_id: The unique identifier for the scanner to be deleted.
    :return: Confirmation message if successful, error otherwise.
    """
    if scnrd_uid in SCANNERS_DATABASE:
        del SCANNERS_DATABASE[scnr_uid]
        return {"message": "Scanner deleted successfully"}, 200
    else:
        return {"error": "Scanner not found"}, 404

if __name__ == "__main__":
    app.run(debug=False)  # Note: Never use debug=True in a production environment.
