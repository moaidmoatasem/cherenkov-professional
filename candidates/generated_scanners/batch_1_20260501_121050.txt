#!/usr/bin/env python3

import xml.parsers.expat

class XxeScanner:
    """A class to scan for XML External Entities in an input."""
    
    def __init__(self):
        self.parser = xml.parsers.expat.ParserCreate()
    
    def parse_string(self, data: str) -> list:
        """
        Scans the given string for external entity references and returns a list of possible attacks.
        
        :param data: The XML string to be parsed
        :return: A list containing any detected external entities. If an unexpected token is encountered,
                 None will be returned as part of this list.
        """
        found_entities = []
        self.parser.StartElementHandler(self.element_start)
        self.parser.EndElementHandler(self.element_end)
        self.parser.CharacterDataHandler(self.data)
        try:
            self.parser.Parse(data, True)
            return found_entities
        except xml.parsers.expat.ExpatError as e:
            print(f"Error parsing: {e}")
            raise
        finally:
            del data

    def element_start(self, name, attributes):
        """
        Start of an XML element handler.
        
        :param name: The name of the start tag
        :param attributes: A dictionary representing named arguments passed to this function. It is used to track 
                            parameter parsing information and to determine which character data handling methods are 
                            called within processing instructions or comments.
        """
        if '!' in name:
            entities = self.parse_entity(attributes)
            found_entities.append((name, entities)) if entities else None

    def element_end(self, elt):
        """Handler for the ending of an XML element."""
        pass

    def data(self, content: str) -> None:
        """
        Handler called to process character data in a start-tag or internal parameter.
        
        :param content: A single token containing one piece of character data
        """
        self.parser.CharacterDataHandler(content)
    
    def parse_entity(self, attributes):
        """Helper method for parsing entity content."""
        if 'xmlns' in attributes:
            return []
        
        has_exp = False
        entities = []
        xml_data = b""
        for value in self.parse_value(attributes):
            entities.append(xml_data.decode())
            xml_data = (value + "\n").encode()
            
            def get_entity():
                try:
                    self.parser.ExplicitEntityValueHandler(value)
                    return True, ""
                except xml.parsers.expat.ExpatError as e:  # This is not an actual ExpException handler
                    has_exp = True
                except Exception as e:
                    raise e
                
            if not (self.parser.StartElementHandler(get_entity) and get_entity()):
                break
        
        self.parser.EntityValueHandler(" ")
        return entities
    
    def parse_value(self, dic):
        """Generator method for value parsing."""
        if isinstance(dic.get(0), str):
            values = [dic.pop(0)]
        else:
            values = dic.pop(0)
        while True:
            next = None
            for v in (values[-1], *values[1:], values[0]):
                yield v
                del values[1:-1]
                next = v
            if not self.parser.StartElementHandler(lambda: get_entity(next)):
                break
