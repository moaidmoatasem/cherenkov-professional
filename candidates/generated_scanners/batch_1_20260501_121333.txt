import os
from urllib.parse import urljoin

class FileUploadVulnerabilityScanner:
    def __init__(self):
        self.valid_mime_types = {'image/jpeg', 'image/png', 'application/pdf'}
        
    def check_file_upload(self, base_url, file_paths):
        """
        Scans a given set of file paths for possible file upload vulnerabilities.

        Parameters:
        - base_url (str): The URL base to construct the full request URLs.
        - file_paths (List[str]): A list of file paths relative to the current working directory.

        Returns:
        - vulnerabilities_found: A list of potential file upload vulnerabilities, with details about each found vulnerability. 
          If no vulnerabilities are found, returns an empty list
        """
        vulnerabilities_found = []
        
        for path in file_paths:
            full_url = urljoin(base_url, os.path.relpath(path, start=os.getcwd()))
            response = self.send_request(full_url)
            
            content_types = set()
            try:
                mime_type = response.headers['Content-Type']
                if ';' in mime_type:  # Check for valid MIME types even when there's a parameter like "boundary"
                    ctype = mime_type.split(';')[0]
                    if ctype not in self.valid_mime_types:
                        vulnerabilities_found.append({
                            'file_path': os.path.abspath(path),
                            'mime_detected': mime_type,
                            'base_url': base_url
                        })
                content_types.add(ctype)
            except KeyError:
                vulnerabilities_found.append({
                    'file_path': os.path.abspath(path),
                    'content_type': None,
                    'base_url': base_url  # This may not be accurate due to server limitations.
                })
                    
        return vulnerabilities_found
