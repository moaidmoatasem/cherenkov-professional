import subprocess

class CommandInjectionScanner:
    """CommandInjectionScanner class to scan and validate commands for security vulnerabilities.
    
    Attributes:
        _allowed_commands (list): List of allowed command values.
    """

    def __init__(self):
        self._allowed_commands = ['ls', 'pwd', 'echo']  # Example of allowed commands

    @staticmethod
    def is_command_safe(command: str) -> bool:
        """Check if a given command is safe to execute.

        Args:
            command (str): The command to check.
            
        Returns:
            bool: True if the command is in the list of allowed commands, False otherwise.
        """
        return any(map(lambda cmd: cmd == command, CommandInjectionScanner._allowed_commands))
    
    def scan_command(self, input_string: str) -> tuple[str, int]:
        """Scan a given string for possible command injection and validate it.

        This method attempts to extract commands from the provided input. If safe commands are found,
        return (True, 0). If no commands are found or allowed commands should raise error.
        Otherwise, indicate that the operation is not safe by returning (False, exception raised).

        Args:
            input_string (str): The string to scan and verify.

        Returns:
            tuple[bool, int]: A tuple where:
                - bool indicates if the operation is secure,
                - int contains an error code: 0 if successful, a non-zero value indicating issues.

        Raises:
            ValueError: If the command cannot be safely validated or is insecure.
            
        """
        for cmd in self._allowed_commands:
            if cmd.lower() in input_string.lower():
                return (True, 0)
               
        try:
            subprocess.Popen(input_string).wait()
            return (False, f"Operation aborted! Command: '{input_string}'").encode('utf-8')
        except Exception as e:
            raise ValueError(f"Insecure command detected. Possible injection attempt: {e}")

def main():
    """Example usage of the CommandInjectionScanner."""
    scanner = CommandInjectionScanner()
    
    # Example 1: Safe operation
    result, err_code = scanner.scan_command("echo Hello")
    if not err_code:
        print("Operation succeeded as expected.")
  
    # Example 2: Unsafe operation (should trigger ValueError)
    result, err_code = scanner.scan_command("rm -rf /") 
    if not err_code:
        print("An unsafe injection attempt was made but no exception was raised.")
  
if __name__ == '__main__':
    main()
