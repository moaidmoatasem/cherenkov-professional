#!/usr/bin/env python3

import json
import requests


def slack_discord_integration(webhook_url):
    """
    Function to send messages to Slack/Discord using the provided webhook URL.
    
    Args:
    - webhook_url (str): The Webhook URL of the destination channel/Slack/Discord.

    Returns:
    None. Sends a message after validation and error handling is in place.
    
    Example Usage:
    >>> slack_discord_integration("https://hooks.slack.com/services/T0001AAA/B0002BBB/XXXXXXXXXXXX")
    No Error: Message sent successfully

    Raises:
    - ValueError: If the provided webhook_url is invalid or null.
    """
    # Validation of input
    if not webhook_url or not isinstance(webhook_url, str):
        raise ValueError("Webhook URL must be a valid non-null string.")

    message = "Test message from Security Development Tool"

    try:
        response = requests.post(
            url=webhook_url,
            json={"text": message},
            timeout=10
        )
        
        if response.status_code == 200:
            print("No Error: Message sent successfully")
        else:
            raise ValueError(f"Error: Received HTTP status {response.status_code} when sending the slack/Discord message. Body content: {message}")
    except requests.exceptions.RequestException as e:
        # Handle network errors
        raise ValueError(f"An unexpected error occurred while attempting to send a Slack/Discord message: {e}")
