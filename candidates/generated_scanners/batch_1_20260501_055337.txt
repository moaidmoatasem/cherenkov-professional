def generate_executive_summary(overview_data):
    """
    Generates an executive summary based on the provided overview data.
    
    Parameters:
    overview_data (dict): A dictionary containing details about various aspects of a system, 
                           including key performance indicators, security vulnerabilities, 
                           compliance metrics, and other relevant information.

    Returns:
    str: The generated executive summary as a formatted string.

    Raises:
    ValueError: If `overview_data` is not a dictionary.
    KeyError: If required keys are missing in the `overview_data`.
    
    Example Usage:
    >>> overview = {
    ...     "key_performance_indicators": ["Savings from Energy Reduction", 50],
    ...     "security_vulnerabilities": [{"cve_number": "CVE-2023-4321", "severity_level": "High"},
    ...                                   {"cve_number": "CVE-2023-6789"}], 
    ...     "compliance_metrics": ["Regulatory Compliance"],  
    ...     "other_details": ["Custom Security Framework Installed"]
    ... }
    >>> summary = generate_executive_summary(overview)
    >>> print(summary)
    
    """
    # Check if input is a dictionary
    if not isinstance(overview_data, dict):
        raise ValueError("Overview data must be provided as a dictionary.")
        
    required_attributes = ["key_performance_indicators", "security_vulnerabilities",
                            "compliance_metrics", "other_details"]
    
    missing_keys = [key for key in required_attributes if key not in overview_data]
    if missing_keys:
        raise KeyError(f"The following keys are missing: {', '.join(missing_keys)}")
        
    summary = ""
    # Generate the executive summary
    summary += ("**Executive Summary**\n\n")
    
    if overview_data.get("key_performance_indicators"):
        summary += "Key Performance Indicators:\n"
        for key, value in overview_data["key_performance_indicators"]:
            summary += f"- {key} : {value}%\n"

    if overview_data.get("security_vulnerabilities"):
        summary += "\n**Security Vulnerabilities:**\n"
        for vulnerability in overview_data["security_vulnerabilities"]:
            severity = vulnerability.pop("severity_level")
            summary += "- CVE: {}, Severity: {}\n".format(vulnerability['cve_number'], severity)
  
    if overview_data.get("compliance_metrics"):
        summary += "\n**Compliance Metrics:**"
        
    if overview_data.get("other_details"):
        summary += "\n**Other Details:**\n"

        for category, details in overview_data["other_details"]:
            summary += "- {} : {}\n".format(category, ", ".join(details))
    
    # Append a conclusion statement
    if "compliance_metrics" in overview_data:
        summary += ("\nThese findings indicate that the system is well-positioned to meet its key compliance objectives. "
                     "However, continued monitoring and proactive measures are necessary.")
  
    return summary
