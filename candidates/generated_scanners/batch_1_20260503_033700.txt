import xml.etree.ElementTree as ET

class XXEScanner:
    """
    This scanner is designed to detect XML External Entities (XXE) vulnerabilities in XML documents. 
    It parses an XML file and raises an exception if it detects the presence of potentially harmful entity references.
    
    The XML document should be provided via the 'xml_file' parameter of the scan method, or read from stdin with -f option.

    Example usage:
    >>> scanner = XXEScanner()
    >>> scanner.scan('path/to/unsafe/xml/file.xml')
    
    Raises Exception when XXE vulnerability is found.
    """

    def __init__(self):
        self._error_messages = [
            "Could not resolve external entities in XML document. Potential XXE vulnerability detected.",
            "Detected potentially harmful entity reference. Please sanitize the input to prevent XXE injection."
        ]

    def scan(self, xml_file=None, xml_data=b""):
        """
        Scans an XML file or provided data for XXE vulnerabilities.
        
        Parameters:
        - xml_file: str. Path to the XML file (default is None).
        - xml_data: bytes. Binary data containing XML content (default is b"").
        
        Returns:
        True if no XXE vulnerability found, False otherwise.
        
        Raises:
        Exception with error message on XXE vulnerability detection.
        """

        # Define a callback function to handle entity expansion
        def callback(expanding_entity, original_attr, expanded_attr):
            """
            Callback triggered by ElementTree when handling external entities. 
            If any entity is found, an exception gets raised and appropriate error message displayed.

            Parameters:
            - expanding_entity: str. Attribute name containing the external DTD.
            - original_attr: str. The original content read from XML file before expansion.
            - expanded_attr: str. Content after processing the external attribute.

            Raises Exception if XXE is detected to prevent injection attack.
            """
            raise Exception(f"{self._error_messages[0]}")

        # Define filter function to remove unwanted entities
        def filter_callback(expanding_entity, original_attr):
            """
            Retains only allowed XML entities and removes potentially harmful ones. 
            This callback will be used for entity expansion filtering.

            Parameters:
            - expanding_entity: str. The attribute name containing raw DTD.
            - original_attr: str. Content from the input XML before processing.

            Returns:
            True to continue expansion, False to stop further process or raise an exception.
            
            In this case we return False for all expansions other than 'system' entities to prevent injection attack.
            """
            if expanding_entity.startswith("&") and " ") not in original_attr:
                # Check entity against our filter
                if expanding_entity.lower() != "-//XML nullEntity---->" and expanding_entity.lower()[
                    2:-1] not in ["&", "", "<", "!"]:
                    return False
                else:    
                    return True

        try:
            xml = None
            with open(xml_file, 'r') as f:
                xml_data = f.read()
        except OSError:
            raise ValueError("Invalid XML file path provided. Please provide a valid PATH.")

        if xml_data is not None and xml_data:
            # Parse the XML data directly
            xml_content = ET.fromstring(xml_data)
            xml_tree = ET.ElementTree(xml_content)
        else:
            # Read from stdin with -f option
            xml = ET.fromstring(xml_data)

        # Enable external entity expansion if no filter is provided
        ET.register_entity_handler(ET.XML_PROLOG_Parser, callback)

        # Register our custom filtering mechanism for entities to detect XXE attack.
        ET.register_expansion_callback(filter_callback)

        try:
            ET.tostring(xml_tree)  # This should raise an exception on XXE
        except ET.ParseError as e:

            if 'Could not resolve external entities in XML document' in str(parsed_xml):
                print(f"{self._error_messages[1]}")
            else:

                raise
