```python
import requests


def post_message_to_slack(channel_id, message):
    """
    Posts a message to a specific channel in Slack.
    
    Arguments:
        channel_id (str): The unique ID of the Slack channel where the message is posted.
        message (str): The message content to be sent.
    
    Returns:
        dict: Response dictionary from Slack API containing status and data if successful.
              Otherwise, returns None with an error message.
    """
    webhook_url = "https://hooks.slack.com/services/T0XXXXXXXXX/BXXXXXXX/BXXXXXXX"
    headers = {"Content-Type": "application/json"}
    payload = {
        "channel": channel_id,
        'text': message
    }
    
    try:
        response = requests.post(webhook_url, headers=headers, json=payload)
        response.raise_for_status()  # Raise HTTPError if the status is not successful.
        return {"status": True, "data": response.json()}
    except (requests.RequestException, ValueError) as e:
        return {"status": False, "error_message": f"Failed to post message: {str(e)}"}
    

def test_post_message_to_slack():
    """Test function for post_message_to_slack with dummy data."""
    dummy_channel_id = "C1234567890"
    dummy_message = "Hello from Python!"
    
    result = post_message_to_slack(dummy_channel_id, dummy_message)
    
    assert result['status'] is True
    assert 'data' in result
    
    print(f"Message posted successfully to Slack channel {dummy_channel_id}: {result['data']['message']}")
    

if __name__ == "__main__":
    # Example usage
    channel_id = "C1234567890"
    message = "This is a test message from Python bot."
    
    result = post_message_to_slack(channel_id, message)
    print(f"Result: {result}")
    
    # Running tests for reliability
    test_post_message_to_slack()
```

This code assumes the Slack API configuration (`webhook_url`) and channel IDs are correct for your environment. Please replace placeholders like `T0XXX`, `BXXXXXXX` with actual values if you have them.
