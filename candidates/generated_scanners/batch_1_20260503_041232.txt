from typing import List

class ErrorReporter:
    """
    Reports errors encountered during feature/scanner execution.
    
    Attributes:
        error_counter (int): The count of reported errors.
    """

    def __init__(self):
        self.error_counter = 0

    def report(self, message: str) -> None:
        """
        Prints an error message and increments the error counter.

        Args:
            message (str): The error message to print.
        """
        self.error_counter += 1
        print(f"Error {self.error_counter}: {message}")

class HTMLDashboardReportGenerator:
    def __init__(self, title: str, features_list: List[FeatureScanner]):
        """
        Initializes an instance of an HTML dashboard report generator.

        Args:
            title (str): The title of the dashboard report.
            features_list (List[FeatureScanner]): A list of features to be reported on.
        """
        self.title = title
        self.features_list = sorted(features_list, key=lambda scanner: (scanner.name))
    
    def generate(self) -> str:
        """
        Generates an HTML dashboard report summarizing the status and outcomes of each feature.

        Returns:
            str: The generated HTML content of the dashboard.
        """
        template = """<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>{}</title>
</head>
<body>
<h1>Dashboard Report</h1>
<ul>
{}
</ul>
</body>
</html>"""
        
        errors_section = "\n".join(
            f"  - <li><strong>{feature.name}</strong>: {', '.join(error.message for error in feature.error_reporter.errors)}</li>"
            for feature, error in zip(self.features_list, self.features_list[1].error_reporter.errors)
        )
        
        return template.format(self.title, "".join(
            f"  - <li>{feature.name}</li>"
            for feature in self.features_list
        ) or "All features scanned successfully!" + (f"<br><p>None of the scanned features reported any issues.</p>" if not errors_section else "") +
        "\n".join(
            f"<details open><summary>{error.feature_name}'s status</summary>"
            f"<ul>{errors_section}</ul></details>"
            for error in self.features_list[-1].error_reporter.errors
        ) + (f"<p>Overall, {len(self.features_list[0].error_reporter.errors)} errors were reported.</p>" if self.features_list else ""))


class FeatureWrapper(FeatureScanner):
    def __init__(self, feature: str):
        FeatureScanner.__init__(self, feature)
    
    @classmethod
    def from_file(cls, filename: str) -> "FeatureWrapper":
        """
        A classmethod to facilitate instantiation of FeatureWrapper with features data defined in a file.

        Args:
            filename (str): The path to the CSV or TXT file containing features and their statuses.
        
        Raises:
            FileNotFoundError: If the specified file does not exist.
        
        Returns:
            FeatureWrapper: An instance of FeatureWrapper initialized from provided feature data.
        """
        with open(filename, 'r') as f:
            if cls.features_data := [line.split(',') for line in f.read().strip().split('\n')]:
                return cls._ensure_valid(cls.features_data[0])(*cls.features_data[1:])
        
        raise FileNotFoundError(f"{filename} not found.")
    
    def _ensure_valid(self, valid_features: str) -> None:
        """
        Ensures input is formatted as expected for feature data in the wrapper's features_list.

        Args:
            valid_features (str): Features to expect a list of strings.
        
        Raises:
            ValueError: If the data could not be formed into expected format indicating success or failure of features' existence.
                        
        Returns:
            None
        """
        if not all(isinstance(feature, str) for feature in valid_features):
            raise ValueError("Feature scanning failed at start.")
    
    @classmethod
    def validate(cls, feature_validation_list: List[str]) -> "FeatureWrapper":
        """
        Classmethod to ensure the data of FeatureScanner instances within FeatureWrapper are validated properly.

        Args:
            feature_validation_list (List[str]): A list where each element is a comma-separated string pair representing a feature.
            
        Returns:
            FeatureWrapper: An initialized instance ready for scanning
                               provided features and their validating statuses.
        """
        
        self.features_data = [feature_validation_list[i].strip().split(",")[:2] for i in range(len(feature_validation_list)) if feature_validation_list[i]]
        for i, (name, status) in enumerate(self.features_data):
            FeatureScanner.__init__(self, name)
            if not (status in ["true", "false", None]):
                raise ValueError(f"FeatureValidation error: {feature_validation_list[i]} is invalid.")
        
        return self
