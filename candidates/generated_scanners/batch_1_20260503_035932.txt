import os
from datetime import datetime
import pdfplumber  # Install this library if you haven't already (pip install pdfplumber)
from typing import List

class ReportGenerator:
    """
    A simple class to generate PDF reports containing a summary of system health checks or audits.
    
    Attributes:
        report_title: Title for the generated PDF report
        output_directory: Directory where the generated PDF will be saved.
    """

    def __init__(self, report_title: str, output_directory: str):
        self.report_title = report_title
        self.output_directory = os.path.abspath(output_directory)
        if not os.path.exists(self.output_directory):
            os.makedirs(self.output_directory)

    @staticmethod
    def generate_pdf_report(results: list) -> None:
        """
        Generates a PDF report using pdfplumber. The content of the report is based on the results parameter,
        which should be a list of items (like dictionaries or other iterable collections). Each item in the
        results lists will become a 'page' with its own title.

        Parameters:
            results: A list containing elements, each of which can be iterated over to extract details like status codes.
                For example it could look like: [{'title': 'Server Status', 'status_code': 200}, {'title': 'Database Connection', 'status_code': 500}]
        """
        file_name = datetime.now().strftime(f"{ReportGenerator.report_title}_{datetime.utcnow().isoformat()}.pdf")
        with pdfplumber.open(os.path.join(ReportGenerator.output_directory, file_name)) as pdf:
            for index, result in enumerate(results):
                page = pdf.pages[index]
                # Example formatting, you might want to customize how the report looks.
                page.text += f"Page {index+1}\n"
                if isinstance(result, list) or isinstance(result, tuple):
                    for result_item in result:
                        if 'status_code' in result_item and 'title' in result_item:
                            page.add_textbox(
                                bottom_left_corner=[0, -3 * pdf.page_height / 2],
                                text=result_item['title'],
                                height=pdf.page_height / 4,
                                width=pdf.page_width
                            )
                        if 'status_code' in result_item and isinstance(result_item['status_code'], int):
                            page.add_textbox(
                                bottom_left_corner=[0, -3 * pdf.page_height / 2 + pdf.page_height / 5],
                                text=f"Status Code: {result_item['status_code']}",
                                height=pdf.page_height / 4,
                                width=pdf.page_width
                            )
                else:
                    if 'status_code' in result and isinstance(result['status_code'], int):
                        page.add_textbox(
                            bottom_left_corner=[0, -3 * pdf.page_height / 2],
                            text=f"Status Code: {result['status_code']}",
                            height=pdf.page_height / 4,
                            width=pdf.page_width
                        )
                    if 'title' in result and isinstance(result['title'], str):
                        page.add_textbox(
                            bottom_left_corner=[0, -3 * pdf.page_height / 2 + pdf.page_height / 5],
                            text=result['title'],
                            height=pdf.page_height / 4,
                            width=pdf.page_width
                        )

    def generate_report(self, results: List[dict]) -> None:
        """
        Generates the final PDF and save it in self.output_directory.
        
        Parameters:
            results: A list of dictionaries representing system checks or audit results.
        """

        self.generate_pdf_report(results)
        report_path = os.path.join(ReportGenerator.output_directory, f"{self.report_title}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.pdf")
        return report_path
