```python
# .github/workflows/security-scanner.yml

name: Security Scanner

on:
  workflow_dispatch:

jobs:
  security-checks:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v2

      - name: Install Dependencies
        run: |
          sudo apt-get update -qq && sudo apt-get install -y python3
          pip3 install --user pylint pep8

      - name: Run Static Analysis with Pylint and Pep8
        env:
          PYTHONPATH: "$PYTHONPATH:$GITHUB_WORKSPACE"
        run: |
          pylint --rcfile=pylintrc $PYTHONPATH/.github/workflows/security-scanner.py
          python3 $PYTHONPATH/.github/workflows/security-scanner.py
          
      - uses: actions/cache@v2
        with:
          path: ~/.cache/pip
          key: ${{ runner.os }}-pip-cache-${{ hashFiles('**/requirements.txt') }}
          restore-keys: |
            ${{ runner.os }}-pip-cache-

      - name: Create Security Report
        env:
          PYTHONPATH: "$PYTHONPATH:$GITHUB_WORKSPACE"
        run: |
          python3 .github/workflows/security-scanner.py --report
```
