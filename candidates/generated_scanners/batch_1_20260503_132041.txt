#!/usr/bin/env python3

import json
from urllib.request import Request, urlopen

class VulnerabilityCorrelationEngine:
    def __init__(self):
        self.vulnerabilities = []

    def add_vulnerabilities(self, vulnerabilities_list):
        """
        Add a list of vulnerabilities to the engine for correlation.
        
        :param vulnerabilities_list: List of dictionaries, each representing a vulnerability
                                      with keys 'name', 'severity', and other relevant attributes.
        """
        if not isinstance(vulnerabilities_list, list):
            raise ValueError("Input should be a list")
        
        self.vulnerabilities.extend(vulnerabilities_list)
    
    def correlation_engine(self):
        """
        Perform correlation on the current set of vulnerabilities.

        Returns:
            A dictionary where keys are 'common_vuln', value is another dictionary with the vulnerabilities
            and their counts.
        """
        # Convert list to json for easier processing
        self._convert_to_json()
        common_vulnern_name = {}
        
        for vuln in self.vulnerabilities:
            if vuln["name"] in common_vulnern_name:
                common_vulnern_name[vuln["name"]] += 1
            else:
                common_vulnern_name[vuln["name"]] = 1
        
        # Filter out unique vulnerabilities, keeping counts for others
        filtered_vulns = {k: v for k, v in common_vulnern_name.items() if v > 1}
        
        return {
            "common_vuln": filtered_vulns,
        }
    
    def _convert_to_json(self):
        """
        Convert the vulnerabilities list to a JSON string ready for external processing.
        This is useful as the original form of this data might not be ideal for correlation.
        """
        if self.vulnerabilities:
            self._json_data = json.dumps(self.vulnerabilities)
    
    def _validate_and_convert(self, new_vuln: dict) -> list:
        """
        Validate a new vulnerability and ensure its format before appending to the engine.

        :param new_vuln: A dictionary representing a single vulnerability.
        
        Returns:
            A list of dictionaries with error messages if any are found during validation process,
            otherwise an empty list indicating no issues were encountered.
        """
        errors = []
        
        # Example validation rules, adjust based on actual data requirements
        required_keys = ['name', 'severity']
        for key in required_keys:
            if key not in new_vuln:
                errors.append({"error": f'Key "{key}" is missing'})
            elif not isinstance(new_vuln[key], str):
                errors.append({"error": f'Value of key "{key}" must be a string'})

        # Assume severity can only be defined in specific values for now
        permissible_severity = ["Low", "Medium", "High", "Critical"]
        if new_vuln["severity"] not in permissible_severity:
            errors.append({"error": f'Value of key "{key}" must be one of {permissible_severity}'})

        return errors

    def add_single_vulnerability(self, new_vuln: dict):
        """
        Ensure a single vulnerability is correctly formatted before adding to the engine.

        :param new_vuln: A dictionary representing a single vulnerability.
        
        Raises:
            ValueError: If a required key or incorrect format is detected and not validated/allowed.
        """
        errors = self._validate_and_convert(new_vuln)
        if errors:
            raise ValueError("New vuln contains invalid data!")
        
        self.vulnerabilities.append(new_vuln)
