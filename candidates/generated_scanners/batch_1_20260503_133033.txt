import os

class PathTraversalScanner:
    """
    A PathTraverseScanner class that checks if input paths are safe from path traversal attacks.
    
    Usage example:
    >>> scanner = PathTraversalScanner()
    >>> valid_path = "/valid/path/123"
    >>> invalid_path = "./../../../../etc/passwd"
    >>> is_safe_valid = scanner.is_path_safe(valid_path)
    >>> print(f"Path {valid_path} : {'Safe' if is_safe_valid else 'Not Safe'}")
    Path /valid/path/123 : Safe
    
    Attributes:
        safe_paths: A set of predefined valid paths. This should not be managed by the user.

    """
    
    def __init__(self):
        self.safe_paths = {
            "/",
            "/usr/local/sbin/",
            "/var/www/html/",
            # Add any other known safe paths here
        }
    
    def is_path_safe(self, path: str) -> bool:
        """
        Checks if a given path is safe from path traversal attacks.
        
        :param path: The directorypath to verify.
        :type path: str
        :return: True if the path can be considered as safe based on checks performed. False otherwise.

        """
        # Normalize path case for comparisons
        normalized_path = os.path.normcase(path)
        
        for safe_path in self.safe_paths:
            # This check may need to accommodate more conditions depending on target application (like user paths, file extensions, etc.)
            if not os.access(safe_path, os.R_OK | os.W_OK):
                continue
            if safe_path.startswith(normalized_path) and normalized_path.endswith(os.path.sep):
                return True
        return False

def example_usage():
    scanner = PathTraversalScanner()
    
    # Example valid directories that should return true - these are some of the known safe paths. Additional checks can be added here.
    is_safe_valid1 = scanner.is_path_safe("/valid/path/123")
    assert is_safe_valid1, "Safe Directory 1 not identified as Safe!"
    
    is_safe_valid2 = scanner.is_path_safe("/usr/local/sbin/")
    print(f"Path /usr/local/sbin : {'Valid' if is_safe_valid2 else 'Invalid'}")  # Valid
    
    invalid_path1 = "./../../../../etc/passwd"
    is_safe_invalid1 = scanner.is_path_safe(invalid_path1)
    
    assert not is_safe_invalid1, "Unsafe directory not identified as Unsafe!"
    
    print(f"Path {invalid_path1} : {'Safe' if is_safe_invalid1 else 'Not Safe'}")  # Not Safe
    
    invalid_path2 = "/var/www/html/.htpasswd"
