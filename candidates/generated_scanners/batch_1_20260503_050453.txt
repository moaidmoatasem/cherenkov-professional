import os
from slack_sdk import WebClient
from discord_webhook import DiscordWebhook, DiscordException

def add_slack_integration(slack_token):
    """
    Create a Slack client using provided token to send messages.
    
    :param slack_token: str - The OAuth access token for authenticating with the Slack API.
    :return: WebClient - A new instance of Slack's REST API wrapper client.
    """
    return WebClient(token=slack_token)

def add_discord_integration(webhook_url):
    """
    Initialize a Discord webhook to send messages.
    
    :param webhook_url: str - The URL where a webhook was set up.
    :return: DiscordWebhook - A new instance of the discord_webhook module.
    """
    try:
        return DiscordWebhook(webhook_url)
    except Exception as e:
        raise RuntimeError(f"Error initializing Discord webhook with provided URL: {e}") from e

def send_message_to_slack(client, message_text):
    """
    Sends a text-based message to Slack using the client object.
    
    :param client: WebClient - A Slack API wrapper client instance.
    :param message_text: str - The content of the message to be sent.
    """
    try:
        response = client.chat_postMessage(
            channel="#general",  # Replace with your desired channel name
            text=message_text,
        )
        response_data = response.json()
        assert response_data["ok"], "Message not delivered successfully"
    except Exception as e:
        raise RuntimeError(f"Failed to send message via Slack: {e}")

def send_message_to_discord(webhook, message_text):
    """
    Sends a text-based message directly to the designated Discord webhook.
    
    :param webhook: discord_webhook.DiscordWebhook - The Discord Webhook object.
    :param message_text: str - The content of the message to be sent.
    """
    try:
        response = webhook.send(message_text)
        assert response.status_code == 204, "Message not delivered successfully"
    except DiscordException as e:
        raise RuntimeError(f"Failed to send message via Discord: {e}")

def main():
    slack_token = os.getenv("SLACK_ACCESS_TOKEN")
    discord_webhook_url = os.getenv("DISCORD_WEBHOOK_URL")

    if slack_token and discord_webhook_url:
        if slack_token and not discord_webhook_url or not slack_token and discord_webhook_url:
            raise ValueError("Either both Slack token AND Discord webhook URL must be set, or neither.")
        
        slack_client = add_slack_integration(slack_token)
        discord_webhook = add_discord_integration(discord_webhook_url)

        slack_message = "This is a test message sent via Slack."
        discord_message = "This is a test message sent via Discord."

        send_message_to_slack(slack_client, slack_message)
        send_message_to_discord(discord_webhook, discord_message)

if __name__ == "__main__":
    main()
