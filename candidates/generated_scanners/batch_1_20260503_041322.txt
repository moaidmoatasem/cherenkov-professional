import jinja2
from fpdf import FPDF
import logging

logging.basicConfig(level=logging.INFO)

class ReportGenerator:
    """
    A class to generate PDF reports.
    
    Methods:
        - create_report(data): Generates a PDF report with the given data.
    """

    @staticmethod
    def create_pdf(report_data):
        """ 
        Generate a PDF using Jinja2 for template rendering and FPDF to handle PDF output.

        Parameters:
            report_data: dictionary containing data for the PDF report.
        """
        # Initialize logger
        logger = logging.getLogger(ReportGenerator.__name__)

        try:
            # Load template from file
            with open('template.html', 'r') as f:
                template = f.read()
            
            # Create environment and select loader
            env = jinja2.Environment(loader=jinja2.FileSystemLoader('.'))
            
            # Replace Jinja placeholders in template with report_data content
            template = env.from_string(report_data)

            # Render the template into html.
            rendered_template = template.render()

            # Initialize PDF instance
            pdf = FPDF()
            pdf.add_page()

            # Get font from fpdf package (Helvetica)
            pdf.set_font("Arial", size=12)

            # Insert content in single line to prevent table splitting on new page
            lines_of_content = []
            for i, line in enumerate(rendered_template.splitlines()):
                if len(line) < 65:  # Adjust this threshold as needed
                    continue
                elif i % 8 == 0:  # Use the 'table_row' layout when reaching rows end
                    lines_of_content += ['<br><hr>&nbsp;<br>']
                else:
                    lines_of_content.append('<div>')
                    lines_of_content.append(line)
                    lines_of_content += ['</div>']

            # Append each line into the pdf using layout defined in 'template.html'
            for content in lines_of_content:
                logger.info(f"Adding content to PDF: {content}")
                pdf.multi_cell(0, 5, content)

            # Save and write file
            pdf.output('report.pdf')

            logger.info("PDF report generated successfully.")
        except Exception as e:
            logger.error(f"An error occurred while generating the PDF: {e}", exc_info=True)
            raise

if __name__ == "__main__":
    # Example usage of ReportGenerator.create_pdf()
    report_data = {
        "Name": "<NAME>",
        "Position": "Security Developer",
        "Experience": 12,
        "Achievements": ["Secured multiple systems", "Developed security tools"],
    }
    
    try:
        ReportGenerator.create_pdf(report_data)
    except Exception as e:
        print(f"An error occurred: {e}")
