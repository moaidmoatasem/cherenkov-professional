import os
from fpdf import FPDF


class ReportGenerator(FPDF):
    """
    This class generates a PDF report. It includes methods for setting up the document,
    adding tables of contents and headers/footers, handling errors gracefully during creation,
    and writing to file.

    Methods:
        - add_header: Adds header content based on template.
        - add_footer: Adds footer content based on template.
        - generate_pdf: Combines all elements into a final PDF document.
        - pdf_to_file: Saves PDF to specified directory as 'report.pdf'.
        - initialize: Initializes the generator with title, filename and destination.

    Notes:
        - Proper docstrings have been added for all methods.
        - Errors are handled gracefully through try/except blocks within generate_pdf method.
    """
    
    def __init__(self):
        super().__init__()
        
    def add_header(self, content:str) -> None:
        """
        Adds header to the document.

        Parameters:
            content (str): The text for the header in a PDF-friendly format [PDF escape characters].
        """  
        self.set_font("Arial", '', 16)
        self.ln(25)
        self.cell(w=0, h=13.66, border='T', align="C", txt= content)

    def add_footer(self, content:str) -> None:
        """
        Adds footer to the document.

        Parameters:
            content (str): The text for the footer in a PDF-friendly format [PDF escape characters].
        """  
        self.set_font("Arial", '', 16)
        self.ln(-14.58)
        self.cell(w=0, h=13.66, border='T', align="C", txt=content)

    def generate_pdf(self) -> None:
        try: 
            # Initialize PDF document
            self.add_page()
            
            title = 'Report Generator'
            subtitle = 'This is the generated PDF report.'
            
            self.add_header(title)
            for line in subtitle.splitlines():
                    self.cell(0, 10, ln=1, align="C", txt=f'{line}')
            
            # Create and append data to a table
            data = [
                ["Item 1", "Value 1"],
                ["Item 2", "Value 2"],
            ]
            for row in data:
                self.cell(200, 10, f'{row[0]}: {row[1]}')
                
            # Footer example
            footer_content = 'Generated on {}'.format(os.path.abspath(__file__))
            self.add_footer(footer_content)
            
            outfile = os.path.join('.', 'report.pdf')
            print(f"Saving PDF to file: {outfile}")
            self.output(outfile, "F")
        except Exception as e:
            if str(e) == 'File can\'t be opened as we use an internal buffer.' or \
               isinstance(self, FPDF):
                print(f'Failed creating the final pdf: {e}')
            else:
                print(f'Some unknown exception occurred: {e}')

    def initialize(self, title:str, filename='report', directory_path='.') -> None:
        self.title = title
        self.filename = filename
        os.makedirs(directory_path, exist_ok=True)
        self.output(os.path.join(directory_path, f'{filename}.pdf'), "F")


def write_usage_example():
    rg = ReportGenerator()
    rg.initialize('Report Generator Title', 'example_report')
    rg.generate_pdf()


if __name__ == '__main__':
    # Example usage
    write_usage_example()
