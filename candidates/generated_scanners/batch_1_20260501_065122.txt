```python
#!/usr/bin/env python3

from argparse import ArgumentParser
import os
import sqlite3
import logging

# Setup logger
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger(__name__)

def create_database():
    """Create the SQLite database and tables for storing scan results."""
    conn = sqlite3.connect('scan_db.db')
    c = conn.cursor()
    c.execute('''
        CREATE TABLE IF NOT EXISTS scans (
            id INTEGER PRIMARY KEY,
            timestamp TEXT,
            product_version TEXT,
            security_score REAL
        )
    ''')
    conn.commit()
    logger.info("Database created/fetched.")
    return conn

def add_scan_result(conn, result):
    """Add a scan result to the database."""
    try:
        with conn.execute('''
            INSERT INTO scans (timestamp, product_version, security_score)
            VALUES (?, ?, ?)
        ''', 
        (
            result['timestamp'],
            result['product_version'],
            float(result['security_score'])
        )
    ):
            logger.info("Scan result added to the database.")
    except sqlite3.DatabaseError as e:
        conn.rollback()
        logger.error('Could not add scan result: %s. Rollback is performed.', e)
        
def get_scan_results(conn):
    """Get a list of all scan results from the SQLite database."""
    cursor = conn.execute("SELECT * FROM scans")
    return cursor.fetchall()

def run_feature_scanner():
    # This function could be anything, for this example we'll simulate a feature scanner
    products = ['ProductA', 'ProductB', 'ProductC'] if os.getenv('ENVIRONMENT', 'prod') == "test" else []
    
    results = {}
    for product in products:
        logger.info(f'Beginning scan for {product}')
        # Simulated scanning operation which returns a dictionary with result info
        try:
            result = {'timestamp': '2023-10-05T14:10', 'product_version': product, 'security_score': 85.7}
        except Exception as e:
            logger.warning(f'Scan failed for {product}: {e}')
            continue
        
        add_scan_result(create_database(), result)

    return results

def pytest_main():
    """Integration of Feature Scanner and Tests for CI/CD."""
    
    # Setup argument parser
    parser = ArgumentParser(description='Run feature scanner.')
    parser.add_argument('-p', '--prod', action='store_true', help='Perform a production scan (default)')
    parser.add_argument('--test', action='store_true', help='Do not perform any scans')
    args = parser.parse_args()
    
    # Start environment based on flags
    if not(args.prod and not args.test):
        logger.info("Running in test environment.")
    else:
        logger.warning("No production scan is run because the environment flag is missing or '--prod' is not specified.")

    results = run_feature_scanner() 
    
    return results

if __name__ == "__main__":
    pytest_main()

if __name__ == "__TEST__":
    # Example usage/test data
    def test_scan_database():
        conn = sqlite3.connect('scan_db.db')
        assert conn.execute("SELECT name FROM sqlite_master WHERE type='table' AND name='scans';").fetchone() is not None, "The scans table does NOT exist."
        results = get_scan_results(conn)
        logger.info("Test Scan Results: %s", results)
        
    if __name__ == "__TEST__":
        test_scan_database()
```
