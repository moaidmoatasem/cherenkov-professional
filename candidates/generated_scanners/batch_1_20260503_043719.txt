import requests
from datetime import datetime

# Constants
CVE_DB_URL = "https://api.example.com/cve"
CREDITS_API_URL = "https://api.example.com/credits"  # Not part of CVE Database, for reference only

def get_cves():
    """
    Fetches the latest CVEs from the CVE database API.
    Returns:
        list: A list of dictionaries containing details about each CVE.
    Raises:
        requests.RequestException: If there is a network issue or request problem.
    """
    try:
        response = requests.get(CVE_DB_URL)
        response.raise_for_status()
        return response.json()
    except requests.ConnectionError as e:
        print(f"Connection Error: {e}")
        raise
    except requests.HTTPError as e:
        if e.response.status_code == 401:  # Unauthorized or invalid credentials.
            print("Unauthorized access to the CVE database. Please check your API key.")
            return []
        if e.response.status_code == 429:  # Rate limit reached.
            print("Rate Limit Reached. Please wait and try again later.")
            raise
        print(f"HTTP Error {e.response.status_code}: Server error.")
    except requests.RequestException as e:
        print(f"General request exception occurred: {str(e)}")
        return []

def get_cve_details(cve_id):
    """
    Fetches the details of a specific CVE from the CVE database API.
    Args:
        cve_id (str): The CVE Identifier.
    Returns:
        dict: A dictionary containing details about the specified CVE or None if not found.
    Raises:
        requests.RequestException: If there is a network issue or request problem.
    """
    try:
        response = requests.get(f"{CVE_DB_URL}/{cve_id}")
        response.raise_for_status()
        return response.json()
    except (requests.ConnectionError, requests.HTTPError) as e:
        print(f"Error fetching CVE details for ID {cve_id}: {str(e)}")
        return None

def validate_credits():
    """
    Validates credits to ensure continued access to the CVE database.
    This function is optional and may not exist in an ideal production environment, but included here for completeness.
    Raises:
        ValueError: If there are not enough credits.
    Returns:
        bool: True if validated successfully, False otherwise.
    """
    try:
        response = requests.get(CREDITS_API_URL)
        if response.status_code != 200:
            raise ValueError("Could not authenticate with credits API.")
        data = response.json()
        print(f"Credits are valid until {datetime.fromtimestamp(data['valid_until']).strftime('%Y-%m-%d')}.")
        return True
    except requests.RequestException as e:
        raise ValueError(f"An error occurred while checking credits: {str(e)}")

def example_usage():
    """
    Example usage of the CVE integration. Demonstrates fetching and handling CVEs.
    Prints details about the latest 5 CVEs, optionally validates credits, then fetches specific details for a specific CVE ID.

    Raises:
        requests.RequestException: If there are any network or request issues during calls to the API.
    """
    cves = get_cves()[:5]  # Fetching latest 5
    print("Latest CVEs (approx. first 10 characters of summary):")
    for idx, cve in enumerate(cves):
        details = get_cve_details(cve['CVE_ID']) if cve.get('CVE_ID') else {}
        credit_status = "valid for credits" if validate_credits() else "not validated"
        print(f"CVE {idx+1}/{len(cves)}: Summary - '{details.get('summary','Unknown')}'. Credits Validity - {credit_status}.")

# Example Usage
example_usage()
