"""
XXE Scanning Tool
------------------
This Python script is designed to perform an XML External Entity (XXE)
scanner for potential security vulnerabilities in XML payloads. It includes basic error handling,
function documentation, example usage instructions, and unit tests.

Functions:
- scan_xml_payload: This function accepts an XML payload as input and recursively parses it.
                     If the document references any external entities that are not filtered out by
                     a policy (e.g., no-exclude-all), this vulnerability will be detected. 

Usage:

    python xxe_scanner.py "your-xml-payload-here"

Example:
```
python xxe_scanner.py "<xml> <!DOCTYPE root [\13]><root></root>"
```

Tests:
The script includes unit tests in the `test_xxescan` module to validate the behavior of

Note: The tool is primarily for testing and demonstration purposes against trusted internal systems.
For production usage, ensure that no XXE injection techniques exist within your environment.
"""

import xml.sax

class NoExclusionFilter(xml.sax.handler.DefaultHandler):
    def __init__(self):
        super().__init__()
    
    # Override the handle_starttag method to not exclude any XML entity references
    def startElement(self, name, attrs):
        return self._proxy(name)

    def _proxy(self, name):
        if ":" in name:  # Check for potential DTD content
            return f'{{{name}}}'
        return name

def scan_xml_payload(xml_data, filter_exclusions=False):
    """
    Scan provided XML data for any external entities.
    
    :param xml_data (str): The XML string to be scanned.
    :param filter_exclusions (bool): If True, use custom NoExclusionFilter; else DefaultHandler.
                                    Defaults to False.
    :return: None 
            A report on whether XXE vulnerabilities were found in the specified payload.

    Raises:
        ValueError: If invalid input type is provided
    """
    
    if not isinstance(xml_data, str):
        raise ValueError("Invalid data type. Please provide valid XML string.")
    
    # Default to filtering external entities using Standard XML parser 
    handler = NoExclusionFilter() if filter_exclusions else xml.sax.handler.DefaultHandler()
    
    try:
        xml.sax.parseString(xml_data.encode('utf-8'), handler)
    except Exception as e:
        print(f"XML Parsing Error: {str(e)}")
        return False
    else:
        print("No XXE vulnerabilities detected!")
        return True

def test_scan():
    # Test with a vulnerable XML content that references external entities.
    result = scan_xml_payload("<xml> <!DOCTYPE root [\13]><root></root>", filter_exclusions=True)
    assert (result == False), "Vulnerable XXE payload not detected"

    # Simple valid XML test
    result = scan_xml_payload('<element property="value">', filter_exclusions=True)
    assert (result == True), "Valid XML did not pass validation"

if __name__ == "__main__":

    args = parser.parse_args()

    # Filter exclusions are on by default unless user opts out with "--no-exclude"
    scan_xml_payload(args.xml_data)
