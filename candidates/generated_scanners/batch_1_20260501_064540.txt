import time
from threading import Lock

class OptimizedScanner:
    def __init__(self, device):
        self.device = device
        self.lock = Lock()
    
    @property
    def performance_report(self):
        with self.lock:
            # Simulating slow reading due to optimizing efficiency
            start_time = time.time()
            
            if self.device.is_connected:
                data = self.device.read_data()
                end_time = time.time()
                
                # Calculate the data read rate for reporting purposes
                bytes_read, seconds_elapsed = len(data), end_time - start_time
                data_rate = bytes_read / seconds_elapsed
                
                return {
                    'success': True,
                    'data_rate_in_kb_per_second': data_rate * 8 / 1024,
                    'seconds_to_complete': int(seconds_elapsed)
                }
            else:
                raise Exception("Scanner is not connected.")
    
    def check_connection(self):
        with self.lock:
            return self.device.is_connected
