import os

class PathTraversalScanner:
    """
    A class to detect path traversal vulnerabilities in HTTP requests.
    """

    def __init__(self, allowed_dirs):
        """
        Initialize the scanner with a list of allowed directories.

        :param allowed_dirs: List of strings representing directory paths that are considered 'safe'
                              for parsing. These allow only absolute paths starting from certain points.
        """
        self.allowed_dirs = [os.path.abspath(dirpath) for dirpath in allowed_dirs]

    def is_directory_safe(self, path):
        """
        Check if the given path leads to a safe location.

        :param path: The directory or file path provided by an HTTP client.
        :return: True if the path is within the context of 'allowed_dirs', False otherwise.
        """
        full_path = os.path.abspath(path)
        for allowed_dir in self.allowed_dirs:
            # Path is considered safe only if it's inside one of the dirs
            if full_path.startswith(allowed_dir) and not path.startswith('/') or \
               (os.path.dirname(full_path).startswith(os.getcwd()) and full_path != "/"):
                return True
            # If no allowed directories were provided, we treat paths as safe if they are absolute.
            elif full_path not in [os.path.join(dirpath, '') for dirpath in self.allowed_dirs]:
                return True

        return False

    def detect_traversal(self, path):
        """
        Detects whether the provided path is within a known directory.

        :param path: The file or directory to be checked.
        :return: A response indicating if traversal is successful.
        """
        try:
            # Check for potential context-dependent vulnerabilities first
            if not self.is_directory_safe(path):
                return "Error: Path Traversal Possible"
            
            # Simulate an attempt to traverse the path to verify the safety
            os.chdir(os.path.dirname(path))
            with open(path) as file_:
                response = f"Path is considered safe: {path}"
                print(response)
                return False  # Considered a safe path
            
        except FileNotFoundError:
            response = f"Not Found: The requested path does not exist."
            print(response)
            return True

    def usage_example(self):
        """
        Example of how to use the PathTraversalScanner.
        """
        scanner = PathTraversalScanner(["/var/www/html", "/home/user"])
        result = self.detect_traversal("/tmp/myfile.jpg")
        if isinstance(result, str):
            print(result)


if __name__ == "__main__":

    def test_safe_path(self):
        """
        Testing a benign directory traversal.
        This should return 'True' with no error message.
        """  
        result = self.scanner.detect_traversal("/var/www/html/user.html")
        self.assertFalse(isinstance(result, str))

    def test_insecure_path(self):
        """
        Testing an insecure directory traversal.
        This attempt to navigate outside safe directories.
