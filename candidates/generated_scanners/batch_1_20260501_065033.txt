import os
from slack_sdk import WebClient
from slack_sdk.errors import SlackApiError

# Configuration for Slack
SLACK_API_TOKEN = "your-slack-api-token"
slack_bot_token = "your-bot-token"

class SlackIntegration:
    def __init__(self):
        self.client = WebClient(token=SLACK_API_TOKEN)

    async def send_message_to_channel(self, channel_id: str, message: str) -> None:
        """
        Send a message to a specified channel.

        :param channel_id: ID of the channel where the message will be sent.
        :param message: Message content to be sent.
        """
        try:
            response = await self.client.chat_postMessage(
                channel=channel_id,
                text=message
            )
            if not response["message"]:
                raise SlackApiError(response=response)
            print(f"Message posted successfully to {channel_id}")
        except SlackApiError as e:
            print(f"A slack error occurred: {e}")

# Example Usage
def main():
    integration = SlackIntegration()
    
    try:
        await integration.send_message_to_channel("C0123456789", "Hello, this is a test message from the bot!")
    except Exception as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    main()
