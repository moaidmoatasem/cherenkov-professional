```python
# .github/workflows/new_repo.yml

name: Create New Repository via GitHub Actions

on:
  schedule:
    - cron: '0 2 * * *"TZ="UTC"' # Run this workflow at midnight UTC every day

jobs:
  create-new-repo-job:
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - name: Checkout code
        uses: actions/checkout@v3
        
      - name: Install GitHub Actions library dependencies
        run: |
          pip install --upgrade github-api
        
        
      - name: Create a repository
        id: create-repo
        uses: actions/github-script@v4
        with:
          script: |
            const { Octokit } = require('@octokit/rest')
            new Octokit({ auth: ORG_TOKEN }).request({
              method: 'POST',
              url: '/user/repos',
              headers: {
                'Accept': 'application/vnd.github+json'
              },
              data: {
                name: github.repository
              }
            }, (err, res) => {
              if (err) {
                echo "Failed to create repository. Error: ${{ err.message }}"
                return 1
              } else {
                echo "Repository ${github.repository} created successfully"
                return 0
              }
            })
        
      - name: Checkout code for future work
        uses: actions/checkout@v3
            
    secrets:
      GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
      ORG_TOKEN: ${{ secrets.ORGANIZATION_PERSONAL_ACCESS_TOKEN }}

```
