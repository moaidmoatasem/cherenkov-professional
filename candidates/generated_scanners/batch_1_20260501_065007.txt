import requests
from flask import Flask, request

app = Flask(__name__)

def create_webhook(url):
    """
    Creates and returns a webhook for monitoring changes in the specified URL.
    
    Parameters:
        url (str): The endpoint URL where the notifications should be sent to.
        
    Returns:
        dict: A dictionary with success or error details.

    Raises:
        Exception: If there is an internal error, e.g. could not connect to provided url.
    """
    def notify_on_change():
        try:
            response = requests.post(url, json={"somekey": "value"}, timeout=10)
            response.raise_for_status()
            return {"status": "success", "details": f"Notification sent successfully with response code: {response.status_code}"}
        except Exception as e:
            raise Exception(f"An error occurred while trying to send notification: {str(e)}")
    
    return notify_on_change

@app.route('/webhook', methods=['POST'])
def webhook():
    url = request.json.get('url')
    if not url or not isinstance(url, str) or not url.startswith("http"):
        return {"status": "failure", "details": "Invalid URL provided"}, 400
    
    notifier = create_webhook(f"{request.scheme}://{request.remote_addr}{url}")
    
    try:
        response_data = notifier()
        
        # Example of processing a successful notification
        response_status, details = (response_data['status'], response_data['details'])
        return {"confirmation": f"Notification sent successfully: {details}"}, 200
        
    except Exception as e:
        error_response = {"status": "failure", "details": str(e)}
        return error_response, 500

if __name_softs_test_env__ == "__main__":
    app.run(debug=True)
