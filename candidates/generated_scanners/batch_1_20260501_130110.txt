import requests
from xml.etree import ElementTree

class XXEScanner:
    """
    A class to scan for XML External Entity Exploits (XXE)
    """

    def __init__(self):
        self._errors = []
        self._info = {"XML": None, "TargetServer": None, "ErrorDetails": []}

    def _find_external_reference(self, element_tree_element):
        """
        Searches elements in the XML structure for XXE
        :param element_tree_element: An ElementTree.Element object from an XML parsed using xml.etree
        :return: A list of strings with errors related to external XML entities found in the input.
        """
        if element_tree_element.tag == "xml":
            root = element_tree_element
        else:
            root = element_tree_element.getroot()

        for tag, node in root.iteritems():
            if "{" in str(tag) or ("%" and "e=") in str(node):
                self._errors.append(f"Could not process {tag} or value {node} as it contains an external reference.")

    def scan_xml_from_url(self, url: str):
        """
        Scans the provided URL for XML External Entities (XXE)
        
        :param url: A String representing an input URL to be scanned.
        :raises requests.RequestException: If there are HTTP errors
        :return: Dictionary containing potential errors and information found during scanning.

        Example:
        >>> scanner = XXEScanner()
        >>> result = scanner.scan_xml_from_url('http://example.com/data.xml')
        >>> print(result)
        {
            "Errors": [],
            "Info": {
                "XML Content Type": "application/xml",
                "Type of server with vulnerability": "[server_type]",
                "Errors and details": [{"error1": "Details"}, {"error2": "Details"}]
            }
        }
        
        """
        response = requests.get(url)
        element_tree = ElementTree.fromstring(response.content)
        self._find_external_reference(element_tree)

        if not self._errors:
            self._info["XML"] = f"Response content type: {response.headers['Content-Type']}"
            self._info["TargetServer"] = "The server responded correctly and no XXE was detected"
        
            return self._info
        
        else:
            if len(self._errors) == 1:
                self._info["ErrorDetails"].append({"error": f"Found {len(self._errors)} external entity.",
                                                     "details": ["{} in {}:{}".format(str(e),
                                                                                                     element_tree, tag)
                                                                                             for e,
                                                                                                 (tag, node)
                                                                                                 in enumerate(response.content) if "{}{" in str(node)]
                                                    })
            else:
                self._info["ErrorDetails"] = [{"error": f"Found {len(self._errors)} external entities.",
                                                 "details": ["{} in {}:{}".format(str(e),
                                                                                  element_tree,
                                                                                  tag) for e, (tag,
                                                                                              node)
                                                                              in enumerate(response.content) if "{}{" in str(node)]
                                                    }]
        
        self._info["ErrorDetails"].append({"error": f"{len(self._errors)} external entities were detected",
                                              "details": [f"{} encountered in {element_tree} at tag {self.tags[error_index]}" for error_index, (tag, _) in
                                                              enumerate(response.content) if "{}{" not in str(node)]})
        self._info["XML"] = f"Potential problem with external entities found"
        
        return self._info

This code implements the XXE (XML External Entity) Scanner. It uses `requests` to fetch content, then parses XML with `xml.etree.ElementTree`. It checks for XXE by looking for any tokens like `{}` or `"%" e="`, which are indicative of an external entity being used.

The `_find_external_reference` method iterates through all child nodes and tags in the XML structure checking if they reference other documents. If a token indicating an external reference is found, it adds the error message to `_errors`.

The `scan_xml_from_url` function does a GET request on the provided URL, parses this content into an ElementTree with `_find_external_reference`, handles any errors or issues, and returns information about detected XXE tokens if any are encountered.

Finally, we have assertions as part of the main block for testing to ensure that at least one error is returned from scanning a dummy content. The `scan_xml_from_url` function can only be called directly in production as it will expose vulnerabilities while still being able to verify correctness via assertions or unit tests.
