#!/usr/bin/env python3

import os

def str_collate(input_data):
    """
    Safely collates path components provided in input_data to prevent directory traversal attacks.
    
    :param input_data: A list containing strings representing file system paths/components.
    :return: A tuple where the first element is a string with properly sanitized and concatenated
             path, and the second element is True if the combined string does not represent an absolute path,
             otherwise False.
    """
    safe_path = ''
    for component in input_data:
        # Check if current component seems like an attempt at directory traversal attack
        while os.path.sep not in (os.path.normpath(safe_path + '/' + component).split(os.path.sep)):
            safe_path += component
    return safe_path, os.path.isabs(safe_path)

def path_traversal_scanner(target_dir, paths_to_check):
    """
    Scans a target directory to ensure that the provided list of file system path components do not indicate any
    presence or absence on external locations such as parent directories.

    :param target_dir: The absolute path to the directory being scanned.
    :param paths_to_check: List of strings representing relative/absolute file extensions, directories, etc.
    :return: A dictionary containing:
             * 'ok': True if none of the provided path components indicate any external traversal (other than the base dir),
                    otherwise False.
             * 'info': A list where each element corresponds to elements in paths_to_check. Each entry is a tuple consisting
                       of sanitized and safe string form of the directory component for easy human consumption,
                       indicating whether the corresponding component represented an effective traversal attempt (True/False).
    """
    
    # Verify we have a valid target_dir provided as an absolute path
    if not os.path.isabs(target_dir):
        raise ValueError("The 'target_dir' must be an absolute path.")
    
    paths_to_info_dict = {}
    for idx, component in enumerate(paths_to_check):
        info_tuple, is_traversed = str_collate([component])
        sanitized_path_info = (info_tuple[0], is_traversed)
        
        if target_dir != '/' and not os.path.sep:
            raise FileNotFoundError(f"The 'target_dir' '{target_dir}' doesn't seem like a valid directory path.")
        
        # Collect all traversal attempts that aren't within the root level.
        paths_to_info_dict[idx] = sanitized_path_info
        
    ok = all([not value[1] for value in paths_to_info_dict.values()])
    
    return {
        'ok': ok,
        'info': []
    }

    target_dir, path_list = sys.argv[1], eval(sys.argv[2])
    result = path_traversal_scanner(target_dir, path_list)
