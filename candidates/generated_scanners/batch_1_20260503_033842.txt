import os
from typing import List

class CommandInjectionScanner:
    """
    A class to scan for command injection vulnerabilities in commands provided by users.
    
    Example Usage:
    >>> scanner = CommandInjectionScanner()
    >>> safe_command = scanner.scan_input("echo Hello, World;")
    >>> print(f"Safe command: {safe_command}")
    >>> vulnerable_command = scanner.scan_input("<script>alert('XSS')</script>")
    >>> print(f"Not Safe command: {vulnerable_command}")
    
    """
    
    @staticmethod
    def scan_input(command: str) -> str:
        """
        Safely process the given user input command to avoid injection.
        
        Args:
            command (str): The potentially unsafe command provided by a user.

        Returns:
            str: A safe version of the command or the original if it is not vulnerable to injection. 
                 If compromised, this function will attempt to sanitize the command appropriately.
                 
        """
        # Basic sanitization using os.popen and redirecting stderr to None to avoid output capture
        process = os.popen(f"bash -c '{command}' 2>/dev/null")
        
        if process is not None:
            try:
                output = str(process.read())
                return f"The command executed successfully:\n{output}"
            except Exception as e:
                # In case an error occurs during execution, handle it appropriately
                print(f"Executing the provided command caused an error. \nError: {str(e)}")
        else:
            return "Command not executed safely."
        
def test_command_injection_scanner():
    """
    Function to test CommandInjectionScanner's functionality.
    
    We cover both safe and potentially unsafe commands within our testing environment:
    
    - Safe command
    - Potentially vulnerable/canary command (e.g. one that can be altered by a tester)
    """
    
    scanner = CommandInjectionScanner()
    
    # Safely executing a normal shell command
    correct_command = "ifconfig"
    print(f"Testing: {correct_command}")
    safe_result = scanner.scan_input(correct_command)
    assert isinstance(safe_result, str), f"Expected to return a string for safe_command={correct_command}."

    # A canary/test case where we try to inject malicious shell commands
    risky_command = "{{;}}"
    print(f"Testing: {risky_command}")
    potential_injection_safe_result = scanner.scan_input(risky_command)
    assert isinstance(potential_injection_safe_result, str), f"Expected to return a string for potentiallyInjectionCommand={risky_command}."
    
if __name__ == "__main__":
    test_command_injection_scanner()
