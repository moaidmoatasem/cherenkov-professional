import re
from typing import List

class VulnerabilitySeverity:
    def __init__(self):
        # Initialize default severities
        self.severities = {
            'low': 1,
            'medium': 3,
            'high': 6,
            'critical': 9
        }

    def score_vulnerability(self, description: str) -> int:
        """
        Scoring function for vulnerability severity based on the key phrase.
        
        :param description: The textual description of the vulnerability.
        :return: An integer severity score.
        """

        # Common keywords associated with different severities
        keyword_map = {
            'potential': 1,
            'caution': 2,
            'warning': 3,
            'critical issue': 6,
            'high risk': 9,
            'urgent attention': 10
        }

        for key, value in self.severities.items():
            if key in description.lower() or any(keyword in f"{desc} {key}".lower() 
                                                 and len(f'{desc} {key}') > 2 
                                                 for desc in re.split(r'\s+', description) 
                                                 for keyword in [x.strip() for x in key.split('|')]):
                return value
        # Default to low severity if no known keywords are found
        return self.severities['low']

    def check_and_score(self, report: str) -> List[dict]:
        """
        Checks vulnerabilities descriptions within a given report and scores them.
        
        :param report: The input text containing vulnerability descriptions.
        :return: A list of dictionaries with vulnerability details including the severity score.
        """

        # Extract unique vulnerability descriptions
        distinct_vulnerabilities = {desc for desc in re.findall(r'([A-Z-a-z ]*):', report)}

        return [{'vulnerability': desc, 'score': self.score_vulnerability(desc)} 
                 for desc in distinct_vulnerabilities]

def example_usage():
    """
    Example usage of the VulnerabilitySeverity class.
    """
    
    vuln_severity = VulnerabilitySeverity()
    report_text = ("Potential vulnerabilities identified. Potential issue detected related to"
                   " authentication and authorization mechanism, potential improvement needed."
                   " Critical vulnerability found in database access layer - urgent attention required.",
                   "A high risk of data breaches through insecure storage practices.")
                   
    print(vuln_severity.check_and_score(report_text))

if __name__ == "__main__":
    example_usage()
