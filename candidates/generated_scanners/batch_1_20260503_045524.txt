```python
import slack_sdk

def add_slack_integration(web_app_url):
    """
    Function to integrate a web application with Slack for notifications.
    
    :param web_app_url: URL of the web application to be integrated into Slack channels.
    :return: A dictionary containing details about the integration, including access token and webhook URI.
    """
    slack_token = 'YOUR_SLACK_BOT_TOKEN_HERE'  # Replace with actual bot token from Slack

    client = slack_sdk.WebClient(token=slack_token)

    try:
        result = client.chat_postMessage(
            channel='#general',
            text=f"Hello from {web_app_url} application. 🚀",
            link_only=False
        )

    except Exception as e:
        error_message = f"{type(e).__name__}: {e}"
        return {'error': error_message}

    webhook_result = result.webhook_response_url

    integrated_details = {
        'access_token': result['response_metadata']['paging']['next_cursor'],
        'webhook_uri': webhook_result
    }

    print(f"Integration complete: {integrated_details}")
    return integrated_details


def main():
    """
    Main function to run the integration process with Slack.
    """
    web_app_url = input("Enter the URL of the web application you want to integrate into Slack channels: ")
    
    result = add_slack_integration(web_app_url)
    
    if 'error' in result:
        print(f"Error occurred during integration: {result['error']}")
    else:
        print("Integration successful. Webhook details have been added.")

if __name as "__main__":
    main()
```

2. The webhook URI and access token in this example are placeholders (`'YOUR_SLACK_BOT_TOKEN_HERE'`). These need to be replaced with an actual bot token obtained through the Slack API.
3. The `add_slack_integration` function takes a URL of your web application as input argument. It then posts a message to a specific channel indicating that the app is operational and using this URI for receiving future notifications.
