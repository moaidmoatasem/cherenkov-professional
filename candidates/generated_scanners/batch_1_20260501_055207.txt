    :return: A list of dictionaries, each representing a specific CVE
    """

    BASE_URL = "https://api.example.cve/v1/cve"
    if not severity or severity.upper() in ['ALL']:
        response = requests.get(BASE_URL)
        return response.json()
    else:
        params = {'severity': severity}
