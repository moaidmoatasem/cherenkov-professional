from flask import Flask, jsonify, request
from flask_sqlalchemy import SQLAlchemy
from werkzeug.exceptions import Unauthorized, NotFound

app = Flask(__name__)
# Configurations can be stored in environment variables for security
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///scan.db'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
db = SQLAlchemy(app)

class Scanner(db.Model):
    """Represents a scan request made to the system."""
    id = db.Column(db.Integer, primary_key=True)
    endpoint = db.Column(db.String(100), nullable=False, unique=True)
    timestamp = db.Column(db.DateTime, default=db.func.now())
    
    def __repr__(self):
        return '<Scanner {}>'.format(self.endpoint)

def unauthorized_error():
    """Custom 401 error handler for Unauthorized exception."""
    return jsonify({'error': 'Unauthorized access'}), Unauthorized.code

def not_found_error():
    """Custom 404 error handler for NotFound exception."""
    return jsonify({'error': 'Endpoint does not exist.'}), NotFound.code

@app.route('/scans', methods=['POST'])
def post_scanner():
    """
    Endpoint to create a new scanner object in the database.
    
    Method: POST
    Parameters:
        - endpoint (str): The specific end-point being scanned.
        
    Response: {
        "id": int,
        "endpoint": str,
        "timestamp": datetime.datetime
    }
    """
    data = request.get_json()
    if not data or 'endpoint' not in data:
        raise ValueError("Incorrect Data Format.")
    
    endpoint = data['endpoint']
    new_scan_request = Scanner(endpoint=endpoint)
    
    try:
        db.session.add(new_scan_request)
        db.session.commit()
        response_object = {
            'id': new_scan_request.id,
            'endpoint': new_scan_request.endpoint,
            'timestamp': str(new_scan_request.timestamp)
        }
        return jsonify(response_object), 201
    except Exception as e:
        app.logger.error("Could not add scanner: %s", e)
        db.session.rollback()
        raise
    
@app.route('/scans/<int:id>', methods=['GET', 'PUT', 'DELETE'])
def handle_scan_request(id):
    """
    Handle GET, PUT, DELETE requests for a specific scanner.
    
    Method: GET/PUT/DELETE
    Parameters:
        - id (int): The identifier of the scanner object in the database.
        
    Error Responses: {
        "error": str
    }
    Response: {
        "id": int,
        "endpoint": str,
        "timestamp": datetime.datetime
    } for GET requests only.
    
    """
    scan_query = Scanner.query.get_or_404(id)
    
    if request.method == 'GET':
        return jsonify({
            'id': scan_query.id,
            'endpoint': scan_query.endpoint,
            'timestamp': str(scan_query.timestamp)
        }), 200
    
    elif request.method == 'PUT':
        
        new_data = request.get_json()
        if not new_data or 'endpoint' in new_data:
            raise ValueError("Incorrect Data Format.")
        
        old_endpoint = scan_query.endpoint
        scan_query.endpoint = new_data['endpoint']
        try:    
            db.session.commit()  
            response_object = {
                "id": scan_query.id,
                "endpoint": scan_query.endpoint,
                "timestamp": str(scan_query.timestamp),
                "message": f"{old_endpoint} was updated to {scan_query.endpoint} successfully."
            }
            return jsonify(response_object), 201   
        except Exception as e:
            app.logger.error("Could not update scanner: %s", e)
            db.session.rollback()
            raise    
    
            
    elif request.method == 'DELETE':
        try:  
            # Handle DELETE specifically to avoid race conditions. 
            db.session.delete(scan_query)
            db.session.commit()   
            response_object = {
                "id": scan_query.id,
                "message": f"Scanner {scan_query.endpoint} successfuly deleted."
            }
            
            return jsonify(response_object), 201  
        except Exception as e:    
            app.logger.error("Could not delete scanner: %s", e)
            db.session.rollback()
            raise

if __name__ == '__main__':
    app.register_error_handler(NotFound, not_found_error)
    app.register_error_handler(Unauthorized, unauthorized_error)
    db.create_all()  # Initialize database
    try:
        app.run(debug=True)    
    except Exception as e:  
        print(f"An error occurred during initialization of the API server. Error: {str(e)}")
