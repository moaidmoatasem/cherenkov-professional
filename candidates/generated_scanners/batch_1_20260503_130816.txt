import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report
from sklearn.linear_model import LogisticRegression

class VulnerabilityPredictionSystem:
    def __init__(self):
        self.model = None
    
    def load_data(self, data_path: str) -> None:
        """
        Load the dataset from a local file.
        
        :param data_path: Path to the data containing features and their labels
        :return: None
        """
        try:
            with open(data_path, 'r') as file:
                lines = file.readlines()
                self.features = [line.split()[0] for line in lines[1:]]  # Assuming feature is at index 0
                self.labels = [int(line.strip().split()[1]) > 0 for line in lines[1:]]
        except FileNotFoundError:
            print("The specified data file could not be found.")
    
    def prepare_data(self) -> None:
        """
        Prepare the dataset by transforming into a suitable format for machine learning.
        
        :return: void
        """
        self.x = np.array(self.features).reshape(-1, 1)
        self.y = np.where(np.apply_alongaxis(lambda x: x[0] > 0.5, -1, self.labels), 1, 0)

    def train_model(self) -> None:
        """
        Train the ML model using cross-validation to handle overfitting.
        
        :return: The trained Logistic Regression model
        """
        if not all([self.x.shape[0] == y.size for x, y in zip((self.x, self.y))]):
            raise ValueError("Data shape mismatch found.")
        
        self.model = LogisticRegression()
        self.model.fit(self.x, self.y)

    def predict_vulnerability(self):
        """
        Predict the potential vulnerabilities based on trained model.
        This is a placeholder for predicting new data points.
        
        :return: None
        """
        # Example of prediction: assuming new features are given in this function
        new_features = ['cve-2018-0046', 's3-access']  # Introduce an example new feature for testing
        pred_probabilities = self.model.predict_proba(self.x)
        predictions = np.where(pred_probabilities[:, 1] > 0.5, True, False)
        
        print(f"Predicted Vulnerability {new_features}: {predictions[0]}")
    
    def model_tester(self):
        """
        Test the trained model's accuracy.
        
        :return: void
        """
        from sklearn.model_selection import KFold
        folds = list(KFold(n_splits=5).split(np.arange(len(self.x))))
        for train_idx, test_idx in folds:
            x_train, x_test = self.x[train_idx], self.x[test_idx]
            y_train, y_true = self.y[train_idx].flatten(), self.y[test_idx].flatten()
            
            preds = np.where(x_train @ self.model.coef_.T + self.model.intercept_ > 0.5, True, False)
            print(f"K-Fold {np.mean(y_true == preds)} percent accuracy with Logistic Regression")
        
        print("Final Model Accuracy: ", self.model.score(self.x, self.y))
