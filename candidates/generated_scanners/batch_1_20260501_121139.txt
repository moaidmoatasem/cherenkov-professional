import subprocess

class CommandInjectionScanner:
    """
    A class to check if given commands are vulnerable to Command Injection.
    
    Methods:
        scan_command(command): Checks if the provided command is potentially vulnerable.
    """

    @staticmethod
    def is_vulnerable_to_injection(script: str, input_value: str) -> bool:
        # Simulate command execution with input sanitization as a placeholder. 
        # In a real-world application, further validation and safe evaluation logic would be needed.
        return subprocess.run(['echo', script.format(input_value)], capture_output=True).returncode != 0

    def scan_command(self, command: str) -> None:
        """
        Scan the provided command for potential Command Injection.

        Args:
            command (str): The command to check for vulnerabilities.

        Raises:
            ValueError: If the command is not a valid string.
        """
        if not isinstance(command, str):
            raise ValueError("Command must be a non-empty string.")
        
        # Checking if there might be injection points in the command
        try:
            input_value = "example_input"  # Placeholder value. In production, consider using safe values or validation logic.
            is_vulnerable = self.is_vulnerable_to_injection(command=command, input_value=input_value)
            if is_vulnerable:
                print(f"The command {command!r} may be vulnerable to Command Injection!")
        except ValueError as e:
            print(str(e))
