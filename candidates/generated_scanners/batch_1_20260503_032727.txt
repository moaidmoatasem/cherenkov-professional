import os

class PathTraversalScanner:
    """
    A class to scan directories for path traversal vulnerabilities.
    
    Methods:
    - detect_path_traversal(path: str) -> bool: Checks if given directory has any path traversal issues.
    """

    def detect_path_traversal(self, path: str) -> bool:
        """
        Detects if a given directory or file path suffers from Path Traversal vulnerabilities.

        Args:
        path (str): The directory or file path to check for vulnerabilities.

        Returns:
            bool: True if any vulnerabilities were detected, False otherwise.
        """

        def is_relative_path(path):
            """Check if the provided path is relative."""
            return os.path.isabs(path) and not os.path.abspath(path).startswith(os.getcwd())

        # Check for relative paths (path-traversal)
        if is_relative_path(path) or os.path.islink(path):
            error_message = f"Path traversal detected: {path} is a relative URL or a symlink."
            print(error_message)
            return True

        # Scan and detect any security issues
        try:
            for root, dirs, files in os.walk(path):
                # Check directory names
                for d in list(dirs[:]):
                    if self.is_path_traversal_vulnerable(d):
                        error_message = f"Path traversal detected: {os.path.join(root, d)} is a vulnerable directory."
                        print(error_message)
                        return True

                # Check file names
                for f in files:
                    if self.is_path_traversal_vulnerable(f):
                        error_message = f"Path traversal detected: File {os.path.join(root, f)} has a vulnerability."
                        print(error_message)
                        return True

        except PermissionError as e:
            error_message = f"The script does not have permission to perform the operation for {path}. Please check your permissions. Exception: {e}"
            print(error_message)
            return True
        except Exception as e:
            error_message = f"Failed to scan {path}: An unexpected error occurred - {e}"
            print(error_message)
            return True

    @staticmethod
    def is_path_traversal_vulnerable(path_str):
        """Static method to validate if the string contains a potential path traversal attack."""
        # Placeholder for actual path traversal validation logic.
        # Example: Check against common sensitive directories or use regular expressions for more sophisticated scanning.
        forbidden_directories = [os.path.join(os.getcwd(), "etc"), os.path.dirname(__file__)]
        return any(directory in path_str for directory in forbidden_directories)
