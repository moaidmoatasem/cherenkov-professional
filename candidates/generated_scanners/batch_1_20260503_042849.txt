import json

class VulnerabilityAssessment:
    """
    A class to assess the severity of vulnerabilities based on CVE (Common Vulnerabilities and Exposures)
    data.
    
    Methods:  
    - __init__(self, cve_data_path): Initializes with a JSON file containing CVE information.
    - assess_severity(self, vulnerability_id): Evaluates the severity of a given vulnerability id.
    """

    def __init__(self, cve_data_path):
        """
        Initializes the VulnerabilityAssessment class by loading the CVE data from a local file.
        
        :param cve_data_path: Path to a JSON file containing CVE information.
        """
        self.cve_data = self.load_cve_data(cve_data_path)
    
    def load_cve_data(self, path):
        """
        Loads CVE data from a specified JSON file. Returns an empty dictionary if the file is not found or invalid.
        
        :param path: Path to the JSON file containing CVE information.
        :return: Dictionary object representing the loaded CVE data.
        """
        try:
            with open(path, 'r') as f:
                return json.load(f)
        except FileNotFoundError:
            print(f"File {path} not found.")
            return {}
        
    def get_cve_data(self, vulnerability_id):
        """
        Retrieves specific CVE information for a given vulnerability ID.
        
        :param vulnerability_id: The id of the CVE to fetch.
        :return: Dictionary containing CVE details or None if not found.
        """
        if self.cve_data:
            return self.cve_data.get(vulnerability_id)
    
    def assess_severity(self, vulnerability_id):
        """
        Evaluates the severity level of a given vulnerability ID based on the loaded CVE data. 
        
        :param vulnerability_id: The id of the vulnerability to determine severity for.
        :return: A dictionary mapping the severity levels ('LOW', 'MEDIUM', 'HIGH', 'CRITICAL') 
                 to their corresponding scores (1, 2, 3, 4 respectively).
        
        Example output:
        {
            "LOW": 1,
            "MEDIUM": 2,
            "HIGH": 3,
            "CRITICAL": 4
        }
        """
    
        cve_info = self.get_cve_data(vulnerability_id)
        if not cve_info:
            print(f"No CVE details found for vulnerability ID: {vulnerability_id}")
            return {"LOW": 1, "MEDIUM": 2, "HIGH": 3, "CRITICAL": 4}
        
        # Simplified example of severity evaluation logic
        score_map = {
            'CVE-2019-1234': {'LOW': False, 'MEDIUM': True, 'HIGH': True, 'CRITICAL': False},
            'CVE-2020-5678': {'LOW': False, 'MEDIUM': False, 'HIGH': True, 'CRITICAL': False}
        }
        
        severity_info = score_map.get(vulnerability_id)
        return {key: 3 - (severity_info[key] * 1) for key in severity_info}

def main():
    """
    A simple example of invoking the VulnerabilityAssessment class and displaying the severity assessment.
    
    :return: None
    Example usage printed to console:
    >>> va = VulnerabilityAssessment('cve_data.json')
    >>> va.assess_severity('CVE-2019-1234')
    {'LOW': 2, 'MEDIUM': 2, 'HIGH': 2, 'CRITICAL': False}
    
    """
    va = VulnerabilityAssessment('cve_data.json')
    result = va.assess_severity('CVE-2019-1234')
    print(json.dumps(result, indent=2))

if __name__ == "__main__":
    main()
