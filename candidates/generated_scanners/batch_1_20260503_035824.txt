import jinja2
from pyPdf import PdfFileWriter, PdfFileReader
import PyPDF2
import io
from fpdf import FPDF

class PDFReportGenerator:
    """
    A class to generate PDF reports using Jinja2 templates and various PDF libraries.
    
    Methods:
    1. render_report(self, data) - Renders a report with given data.
    2. generate_pdf_from_data(self, template_path, output_file_name, data_dictionary)
                                                   Generates a PDF from a template file using provided data.
    """
    
    def __init__(self, jinja_env):
        self.template_env = jinja2.Environment(loader=jinja2.FileSystemLoader('.'), undefined=jinja2.StrictUndefined)
        
    def render_template(self, template_path, context):
        """Renders a Jinja2 template with the given context."""
        return self.template_env.get_template(template_path).render(context)
    
    def generate_pdf_from_data(self, template_path, output_file_name, data_dictionary):
        """
        Generates a PDF report from a rendered template.
        
        Parameters:
        - template_path: The path to the Jinja2 template file.
        - output_file_name: Name of the generated PDF output file.
        - data_dictionary: A dictionary containing context for rendering the template.
        
        Returns:
        None. The result is saved in 'output_file_name'.
        """
        rendered_template = self.render_template(template_path, data_dictionary)
        if not isinstance(rendered_template, str):
            raise ValueError(f"Template output should be a string but got {type(rendered_template).__name__}")
            
        pdf_buffer = io.BytesIO()
        doc_report = FPDF(acrobat_version=3)  # For backward compatibility with older Acrobat viewers
        
        style = "<style>::pdf { page: A4; orientation: portrait; font-family: sans-serif; }</style>"
        rendered_content = f"{rendered_template}".replace("<!--DOCX-->", style)
        
        doc_report.add_page()
        doc_report.set_auto_page_break(auto=True, margin=20)
        doc_report.write(pdf_buffer, rendered_content)
        pdf_buffer.seek(0)  # Move the buffer pointer back to the start
        
        with open(f"{output_file_name}.pdf", "wb") as out:
            out.write(pdf_buffer.getvalue())
        
    def render_report(self, data):
        """
        Renders a report using the template and returns a PyPDF2 PDF object.
            
        Parameters:
        - data: A dictionary of data to be used in rendering the template.

        Returns: 
        pyPdf.PdfFileObject representing the generated PDF after merging it with all pages from FPDF instance
        """        
        pdf_buffer = io.BytesIO()
        doc_report = PyPDF2.PdfFileWriter()  # For compatibility with existing codebases
        style = "<style>::pdf { page: A4; orientation: portrait; font-family: sans-serif; }</style>"
        rendered_content = self.render_template(self.template_path, data).replace("<!--DOCX-->", style)
        
        doc_report.addBlankPage()
        pdf_object = PyPDF2.PdfFileReader(io.BytesIO(rendered_content.encode('utf-8')))
        for num in range(pdf_object.getNumPages()):
            pageObj = pdf_object.getPage(num)
            try:
                # Merge FPDF output and PyPDF2 document
                if not isinstance(pageObj.mergePage, bytes):
                    doc_report.addPage(pdf_object.copyPage(num, self.render_template(self.template_path, data)))
            except Exception as e:
                print(f"Merge Page Error: {e}")
        
        # Ensure the PDF writer object is closed when we are done
        pdf_document = PyPDF2.PdfFileWriter()  # Create a new instance to avoid potential issues with statefulness
        pdf_document.add pages(doc_report.pages)
        
        return pdf_object
    
    def generate_multi_page_pdf(self, template_path, output_file_name, data):
        """
        Generates a PDF report from multiple rendered templates.
        
        Parameters:
        - template_path: The path to the Jinja2 template file.
        - output_file_name: Name of the generated PDF output file.
        - data_list: List of dictionaries containing context for rendering each template.
        
        Returns:
        None. The result is saved in 'output_file_name'.
        """
        multi_page_buffer = io.BytesIO()
        multi_page_doc_report = PyPDF2.PdfFileWriter()
        
        current_pdf_writer = PdfFileWriter()
        for data_item in data_list:
            rendered_content = self.render_template(template_path, data_item)  # Render each template individually
            pdf_object = PdfFileReader(io.BytesIO(rendered_content.encode('utf-8')))
            
            try:
                multi_page_doc_report.addPage(pdf_object.getPage(0))  # Assuming one page per data item is the desired output format here.
            
                current_pdf_writer.addPagesFromReader(pdf_object)  # Add pages only if they don't already exist
            except Exception as e:  # In case we need to handle additional exception scenarios (like adding multiple instances of the page)
                print(f"Adding Error: {e}")
        
        multi_page_doc_report = current_pdf_writer.deepCopyObject()  # Ensure deep copy for further operations if needed
            
        pdf_document = PyPDF2.PdfFileWriter()  # Create a new instance to avoid potential issues with statefulness
        pdf_document.add pages(multi_page_doc_report.pages)
        
        multi_page_pdf_buffer = io.BytesIO()
        multi_page_doc_report.write(multi_page_pdf_buffer)
        multi_page_pdf_buffer.seek(0)                
        multi_page_pdf_buffer.close()  # Ensure we have a proper close method to handle file IO
        
        multi_page_pdf_buffer.flush()  # Important for python3
        with open(output_file_name, 'wb') as output:
            output.write(multi_page_pdf_buffer.getvalue())
            
    def generate_multi_template_pdf(self, templates_path_list, data_lists_dict):
        """
        Generates a PDF report from a list of rendered templates.
        
        Parameters:
        - templates_path_list: A list of Jinja2 template paths.
        - data_lists_dict: A dictionary where keys are the template names and values are lists of dictionaries containing context for rendering each template.
        
        Returns:
        None. The result is saved to filenames in data_lists_dict['output_filenames'] based on the 'templates_path_list' order
        
        """
        multi_page_buffer = io.BytesIO()
        
        # Ensure we create a new instance each time as it might get stateful over many runs
        pdf_document = PyPDF2.PdfFileWriter()  # Create a new instance to avoid potential issues with statefulness
        for template_path, data_list in zip(templates_path_list, data_lists_dict.values()):
            rendered_content = self.render_template(template_path, {'data': data_list})  # Rendering data per document or by keying it off by templates name.
                
            pdf_object = PdfFileReader(io.BytesIO(rendered_content.encode('utf-8')))
            
            try:
                multi_page_doc_report.addPage(pdf_object.getPage(0)) 
            except Exception as e:  # In case we need to handle additional exception scenarios (like adding multiple instances of the page)
                print(f"Adding Error: {e}")
        
        pdf_document = PyPDF2.PdfFileWriter()  # Create a new instance to avoid potential issues with statefulness
        pdf_document.add_pages(multi_page_doc_report.pages)
            
        multi_page_pdf_buffer = io.BytesIO()
        multi_page_doc_report.write(multi_page_pdf_buffer)                
        multi_page_pdf_buffer.seek(0)      
        multi_page_pdf_buffer.close()
        
        for filename, file_instance in data_lists_dict.items():
            with open(filename, 'wb') as output:
                output.write(file_instance.getvalue())
