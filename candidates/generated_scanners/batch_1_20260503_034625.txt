import os
from subprocess import run

class FileUploadVulnerabilityScanner:
    """
    A utility to scan for file upload vulnerabilities in web applications.
    
    Provides methods to identify potential security risks related to improper sanitization of uploaded files. 
    Uses binary files and shell utilities like `file` to analyze the MIME types, extensions,
    and contents of uploaded files. This scanner should be run as a static check post deployment
    but does not replace dynamic scanning or thorough manual review.
    
    Example Usage:
    >>> s = FileUploadVulnerabilityScanner()
    >>> s.scan_directory("/path/to/web/directory")
    """

    @staticmethod
    def scan_file(file_path):
        """
        Inspects the MIME type, content of a single file using shell binary commands.

        Args:
            file_path (str): The path to a file on disk.

        Returns:
            dict: Information about the MIME-type and contents of the scanned file.
        """
        if not os.path.isfile(file_path):
            raise ValueError(f"Provided filepath {file_path} does not point to an actual file.")
        
        # Run file command which can return mime types
        output = run(['file', '--mime-types'] + [file_path], capture_output=True, text=True)
        mime_types = {line.split(':')[0].strip() for line in output.stdout.splitlines()}

        # Get the content of scanned file using 'cat' binary
        with open(file_path, 'rb') as f:
            cat_binary_process = run(['cat', file_path], stdout=subprocess.PIPE)
            content = cat_binary_process.stdout.decode('utf8')
        
        return {
            "mime_types": mime_types,
            "file_content": content
        }

    @staticmethod
    def scan_directory(directory):
        """
        Scans a directory for files, running the MIME analysis on them and returning comprehensive results.

        Args:
            directory (str): The path to a directory where file types and information about content should be checked.
        
        Returns:
            list: A list of dictionaries containing metadata gathered from analyzing each file in the specified directory
                  as well as its parents.

        Note:
            This method recursively searches through the provided directory and subdirectories for files. 
            Each dictionary contains details such as MIME-types, contents, filename with path, and parent directories.
        
        Example Output:
        [
          {
              "file_name": "/webdir/child/path/example.html",
              "mime_types": {"text/html"},
              "content": "... the content of example.html ..."
              "parent_directories": ["common/web"]
          }
        ]
          
        Raises:
            ValueError: If provided directory is not a valid path or if it does not contain any files.
            FileNotFoundError: Raised when scanning subdirectories for files fails due to invalid paths, etc..
        """

        file_infos = []
        found_files = False
        
        # Recursively search the directory and its sub-directories
        for root, _, files in os.walk(directory):
            for filename in files:
                file_path = os.path.join(root, filename)
                
                try:
                    info = FileUploadVulnerabilityScanner.scan_file(file_path)
                    file_infos.append({
                        "file_name": file_path,
                        **info  # Merge the dictionary into one final list.
                    })
                    found_files = True
                except Exception as e:
                    raise ValueError(f"Could not access {file_path}: {e}")
        
        if not found_files:
            raise ValueError(f"The provided directory: {directory} does not have any files to process.")

        return file_infos

# Example usage of the scan_directory method.
if __name__ == "__main__":
    s = FileUploadVulnerabilityScanner()
    results = s.scan_directory("/path/to/web/directory")
    
    for result in results[:10]:  # Display first ten entries for brevity.
        print(result)
