import os
from urllib.parse import urlparse

class FileUploadVulnerabilityScanner:
    """
    Tool for identifying potential security vulnerabilities in file uploads.
    
    This class includes methods to scan static files and determine if they may be 
    vulnerable. It raises exceptions upon issues or returns informative strings.

    Methods:
        - scan_file(path: str) -> str
    """

    def __init__(self):
        pass

    @staticmethod
    def file_type(file_name):
        """
        Determines the MIME type of a given file.
        
        Args:
            file_name (str): The name/path to the file
        
        Returns:
            str: A string indicating the MIME type of the file. If no match is found, 
                 will return "application/octet-stream".
        """
        # Mock function for simplicity in example
        if file_name.endswith('.txt'):
            return 'text/plain'
        elif file_name.endswith('.jpg') or file_name.endswith('.jpeg'):
            return 'image/jpeg'
        else:
            return 'application/octet-stream'

    @staticmethod
    def is_safe_path(url):
        """
        Checks whether a URL path is safe (i.e., should not reveal internal structure).
        
        Args:
            url (str): The full URL of the page
        
        Returns:
            bool: True if path is within expected limits, False otherwise.
        """
        try:
            parsed_url = urlparse(url)
            root_directory = os.path.realpath('/')
            return '/' in parsed_url.path and \
                   root_directory[:len(parsed_url.path)] == parsed_url.path
        except Exception as e:
            raise ValueError(f"Invalid URL: {url}") from e

    @staticmethod
    def safe_content_path(file_paths, upload_dir):
        """
        Validates a set of file paths within the context of an upload directory.
        
        Args:
            file_paths (Iterable): The collection(s) of files to scan and check if they
                are uploaded to their expected location.

            upload_dir (str|Iterable[str]): A single or multiple directories to consider when 
                    determining if these files were submitted to a known safe path within the website.
        
        Returns:
            list(str): The file paths that have been properly validated according to expectations. If no validation is needed, it returns the original set of `file_paths`.
        """
        expected_dirs = os.path.abspath(upload_dir)
        validated_files = []
        
        for path in file_paths:
            
            path_is_validated: bool
            
            if isinstance(expected_dirs, (list, tuple)):
                # Iterating over multiple directories
                for d in expected_dirs:
                    valid_url = FileUploadVulnerabilityScanner.is_safe_path(d + '/' + os.path.basename(path))
                    if valid_url:
                        validated_files.append(path)
                        path_is_validated = True
                        break
            else:
                # For single directory usage only
                valid_url = FileUploadVulnerabilityScanner.is_safe_path(expected_dirs + '/' + os.path.basename(path))
                
                if not expected_dirs == '' and not valid_url:
                    raise ValueError(f'Invalid file upload: path "{expected_dirs + "/" + os.path.basename(path)}" appears unsafe. ')
            if not path_is_validated:
                validated_files.append(path)
                    
        return validated_files


    def scan_file(self, file_path):
        """
        Method for scanning the MIME type and existence checks of a given file.
        
        This method will raise an exception in case there's a security concern detected
        (e.g., non-text or unsafe file path).

        Args:
            file_path (str): The full path to the static file being scanned.

        Raises:
            ValueError: If the MIME type does not match 'text/plain', indicating potential injection risk.
        
        Returns:
            str: A string providing details about whether the file is safe and its MIME type.
        """
        try:
            file_extension = os.path.splitext(file_path)[1][1:].lower()
            expected_mime_type = self.file_type(file_path)
            
            if file_extension in ['/']:
                raise ValueError(f"Unexpected use of path literal: / in {file_path}")
                
            if 'text/plain' != expected_mime_type and self.safe_content_path([file_path], '/static/')[0] is None:
                issue_message = "Detected content-type that suggests malicious intent. Potential security vulnerability detected."
                raise ValueError(issue_message)
            
            safe_url = FileUploadVulnerabilityScanner.is_safe_path("http://example.com" + file_path)
            if not safe_url:
                issue_message = "File location path appears to be outside expected directories/path segments."
                raise ValueError(issue_message)

            return f"{file_path} is {expected_mime_type}, and uploaded to a safe directory."
        except Exception as e:
            str_repr = str(e)
            
            if 'text/plain' != file_extension:  
                str_repr += f"\n\tFile path not expected to be of type '{file_extension}'! "
                
            return str_repr


def main():
    """
    Example Usage
    >>> scanner = FileUploadVulnerabilityScanner()
    >>> result = scanner.scan_file("/uploads/image.jpg")
    
    >>> print(result)
    '/uploads/image.jpg' is image/jpeg, and uploaded to a safe directory.
    """
