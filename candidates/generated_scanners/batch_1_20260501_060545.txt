import logging

class AttackChainDetector:
    """
    The AttackChainDetector class is used to detect and analyze attack chains.
    An attack chain consists of multiple steps or actions carried out by attackers
    during the lifecycle of an incident, typically leading to compromise. This class
    facilitates the detection of potential attack chains in security data.

    Attributes:
        log (Logger): Logging object for recording activity.
        threshold (int): Maximum number of attack stages allowed in a chain.
        max_steps_per_stage (int): Maximum steps permitted within each stage of an attacked chain.
    """

    def __init__(self, threshold=3, max_steps_per_stage=2):
        """
        Initializes the AttackChainDetector with the specified maximum number of attack chains
        and the maximum number of attack stages allowed in a single chain.

        Args:
            threshold (int): Maximum number of attack chains.
            max_steps_per_stage (int): Maximum steps per stage within an attack chain.
        """
        self.log = logging.getLogger(__name__)
        self.threshold = threshold
        self.max_steps_per_stage = max_steps_per_stage

    def detect_attack_chain(self, attack_data: dict) -> bool:
        """
        Detects whether the provided attack data represents a potential attack chain.

        Args:
            attack_data (dict): Dictionary containing attack log entries. Each entry should have a 'stage' key indicating
                the stage in the attack chain and up to 'steps' under each stage indicating individual actions taken.

        Returns:
            bool: True if an attack chain is detected, False otherwise. Detection involves ensuring the total count of steps across all stages does not exceed the maximum threshold.
        """
        self.log.debug("Starting analysis on attack data.")
        if not isinstance(attack_data, dict):
            self.log.error("Input must be a dictionary containing attack logs.")
            return False

        for stage in range(self.threshold + 1):
            total_steps = sum(data['steps'] for data in attack_data.get(str(stage), []))
            self.log.debug(f"Stage {stage}: Total steps: {total_steps}")
            if total_steps > self.max_steps_per_stage:
                return True
        else:
