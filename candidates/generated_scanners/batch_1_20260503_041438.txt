#!/usr/bin/env python

def add_vulnerability_severity_scoring(tool):
    """
    This function aims to score vulnerabilities found by a security scanner.
    The scoring system is based on severity levels such as 'Critical', 'High',
    'Moderate', 'Low'. Each level can apply different weightings that impact the overall
    risk assessment.

    :param tool: Security scanner or tool used for vulnerability detection
    :type tool: str

    Returns:
        dict: A dictionary mapping severities to their respective weights.
    
    Raises:
        ValueError: If the provided tool is not recognized.

    Example usage:
    >>> scoring = add_vulnerability_severity_scoring("Nessus")
    >>> print(scoring)
    {
        'Critical': 5,
        'High': 3,
        'Moderate': 2,
        'Low': 1
    }
    """

    if tool not in ('Nessus', 'OpenVAS'):
        raise ValueError(f"Tool {tool} is not supported. Supported tools are Nessus and OpenVAS.")
    
    scoring_dict = {
        "Critical": 5,
        "High": 3,
        "Moderate": 2,
        "Low": 1
    }
    
    return scoring_dict


if __name__ == "__main__":
    # Example usage of the function
    tool_scoring = add_vulnerability_severity_scoring("Nessus")
    print(tool_scoring)
