#!/usr/bin/env python3

"""
PathTraverseScanner.py: A simple Path Traversal scanner that checks for directory traversal attacks in URLs.
This tool is meant to be used by security professionals and should not be relied upon solely to secure systems.
Due to the sensitive nature of this task, ensure proper authorization before invoking any function.

USAGE:
    path_traversal_scanner(path)

DEPENDENCIES:
    - None
"""

import urllib.parse

def path_traversal_scanner(test_url):
    """
    Checks for directory traversal in a given URL. Given an input test_url,
    it checks if traversing allowed paths may result in access to restricted resources.

    :param test_url: The URL string with potential paths to scan for vulnerabilities.
    :type test_url: str

    :return: True if the path is traversable, False otherwise
    :rtype: bool
    """
    # Parse the input URL and get its components
    parsed_url = urllib.parse.urlparse(test_url)

    allowed_paths = ['./', '../']
    scheme_and_netloc = [f'{parsed_url.scheme}://{parsed_url.netloc}']

    # Extract paths from query, fragment, etc. if they are available
    path_segments = list(parsed_url.path)
    query_elements = parsed_url.query.split('&')
    fragment_elements = parsed_url.fragment.split('#')

    for element in (allowed_paths + scheme_and_netloc):
        try:
            return_path_segment = urllib.parse.quote(element.encode('utf-8'))
        except AttributeError:
            # Handle non-bytestring netloc or allowed paths
            return_path_segment = urllib.parse.quote(str(element).encode('utf-8'))

        path_segments.remove(return_path_segment)
        query_elements.remove(urllib.parse.unquote(return_path_segment))
        fragment_elements.remove(urllib.parse.unquote(return_path_segment))

    if path_segments:
        new_candidate_url_components = (parsed_url.scheme + '://' +
                                         parsed_url.netloc
                                         + '?'
                                         + '&'.join(query_elements)
                                         + f'/{urllib.parse.quote("/".join(path_segments), safe="/")}'
                                         )
        test_url = urllib.parse.urlunparse(new_candidate_url_components)

    elif query_elements:
        new_candidate_url_components = (parsed_url.scheme
                                         + '://' 
                                         + parsed_url.netloc 
                                         + '?'
                                         + '&'.join(query_elements)
                                         + ') #'
                                         + urllib.parse.quote(fragment_elements[0], safe='/')
                                         )
        test_url = urllib.parse.urlunparse(new_candidate_url_components)

    return path_segments or (query_elements and fragment_elements)

def main():
    """
    Example usage of the PathTraverseScanner class. Demonstrates how to use
    the PathTraverseScanner function.
    """

    test_urls = [
        "http://testserver/vulnerable/path?",
        # Add more test URLs as needed...
    ]

    for url in test_urls:
        print(f"Testing URL: {url}")
        result = path_traversal_scanner(url)
        if result is False:
            print(f"No vulnerability found with URL {url}")
        else:
            print(f"\nDirectory traversal detected in one of the paths.\n")

if __name__ == "__main__":
    main()
