import os

class PathTraversalScanner:
    """
    A class to perform path traversal vulnerability scanning.
    
    Attributes:
        target_paths (list): List of paths where scans will be performed.
    """

    def __init__(self, target_paths):
        self.target_paths = target_paths or []

    def scan(self):
        """
        Method that performs the actual path traversal scanning.

        Raises:
            ValueError: If target_paths is not a list type
        """
        if not isinstance(self.target_paths, list):
            raise ValueError("target_paths must be provided as a list")

        # Scan each path and output results.
        for path in self.target_paths:
            try:
                os.access(path, os.F_OK)  # Check if file exists
                response = os.listdir(path)
                print(f"Directory {path} is accessible: {response}")
            except (FileNotFoundError, NotADirectoryError):
                print(f"Path {path} does not exist or is not a directory.")
            except PermissionError:
                print(f"No permission to access path {path}.")

    def run(self) -> None:
        """
        Method that runs the scanner and outputs results.
        
        Returns:
            None, prints output for each target_path found in self.target_paths
        """
        try:
            self.scan()
        except Exception as e:
            print(f"An error occurred: {e}")


def main():
    # Example usage of PathTraversalScanner

    def example_usage():
        scanner = PathTraversalScanner(['/path/to/directory1', '/another/path'])
        scanner.run()

# Running the main function with an example
if __name__ == "__main__":
    main()
