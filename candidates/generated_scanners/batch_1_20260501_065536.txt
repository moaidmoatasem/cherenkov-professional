import time

def perform_feature_scanner(url):
    """
    Function to simulate feature/sensor scanning.
    
    Args:
        url (str): The URL to be scanned by the feature/scanner.

    Returns:
        dict: A dictionary containing the result of scanning. May include error details for failed attempts.
        
    Raises:
        Exception: If any unrecoverable error occurs, such as a network error or insufficient permissions.
    """
    MAX_RETRIES = 3
    TIME_BETWEEN_ATTEMPTS = 5

    results = {
        'scanned_urls': [],
        'errors': []
    }
    
    for retry in range(MAX_RETRIES):
        try:
            # Simulate an external service call that might fail or take some time to complete.
            status_code = _get_url_status(url)
            delay = TIME_BETWEEN_ATTEMPTS - (retry * 2) if retry < MAX_RETRIES-1 else 0
            time.sleep(delay)
            
            if status_code == 200:
                # Successfully scanned the URL. Accumulate results.
                results['scanned_urls'].append(url)
                break
            elif delay > 0 and retry == MAX_RETRIES - 1:
                raise Exception(f"URL scan for {url} failed after all retries.")
        except Exception as e: 
            error_message = str(e)
            if 'insufficient permissions' in error_message or \
               'network timeout' in error_message:
                results['errors'].append({
                    'retry': retry + 1,
                    'exception details': error_message
                })
            elif 'unrecoverable error' in error_message:
                raise Exception(error_message)
    
    return results

def _get_url_status(url):
    """
    Stub function to simulate a successful and an unsuccessful HTTP GET request.
    
    Args:
        url (str): URL to make the request for.
    
    Returns: 
        int: The HTTP status code of the response to the request. 
             200 means everything is fine, other values can indicate errors.
    """
    import requests
    return requests.get(url).status_code
