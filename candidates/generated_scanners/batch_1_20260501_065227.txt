#!/usr/bin/env python

import os
import logging
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score
import pandas as pd


logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class MLBasedVulnerabilityPredictor:
    def __init__(self, data_path, model='RandomForest'):
        self.data = None
        self.model_name = model

        # Load the dataset from a CSV file
        logger.info(f"Loading data from {data_path}")
        
        if not os.path.exists(data_path):
            raise FileNotFoundError(f"The data at URL '{data_path}' does not exist.")
        
        try:
            self.data = pd.read_csv(data_path)
            logger.info("Data loaded successfully.")
        except Exception as e:
            logger.error(f"Error loading the data: {e}", exc_info=True)
        
    def preprocess_data(self):
        # Example preprocessing, including cleaning and feature engineering
        if not self.data is None:
            self.data.fillna(method='ffill', inplace=True)  # Forward fill missing values
            self.data.dropna(inplace=True)  # Drop rows where all columns are NaN
            logger.info("Data preprocessed.")
        else:
            raise ValueError('Data has not been loaded. Call load_data before preprocess.')

    def train_model(self):
        X = self.data.drop(columns=['vulnerability'])
        y = self.data['vulnerability']
        
        # Split data into training and validation sets
        X_train, X_val, y_train, y_val = \
            train_test_split(X, y, test_size=0.15, random_state=42)
        
        if self.model_name == 'RandomForest':
            self.model = RandomForestClassifier(n_estimators=200, random_state=42)
        
        # Train the model
        self.model.fit(X_train, y_train)
        accuracy = accuracy_score(y_true=y_val, y_pred=self.model.predict(X_val))
        logger.info(f"Validation accuracy: {accuracy:.3f}")
    
    def predict(self, data_points):
        if not isinstance(data_points, list) and \
           not isinstance(data_points, tuple):
            raise ValueError('`data_points` must be a list or tuple.')
        
        predictions = self.model.predict(data_points)
        return predictions
        

# Example usage of the MLBasedVulnerabilityPredictor class
def example_usage():
    predictor = MLBasedVulnerabilityPredictor('./path_to_vulnerability_data.csv')
    
    try:
        predictor.preprocess_data()
        predictor.train_model()
        
        # Assuming `example_input` is a valid input for prediction
        examples = [
            [0.2, 1.3],
            [4.5, -3.1]
        ]
        predictions = predictor.predict(examples)
        logger.info(f"Predictions: {predictions}")
    except Exception as e:
        logger.error(f"An unexpected error was encountered: {e}", exc_info=True)

if __name__ == "__main__":
    example_usage()
