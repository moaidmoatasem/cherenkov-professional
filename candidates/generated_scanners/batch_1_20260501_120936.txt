#!/usr/bin/env python3

import xml.etree.cElementTree as ET
from urllib.request import urlopen
from subprocess import Popen, CREATE_NEW_CONSOLE
import json


def validate_xml_content(xml_content):
    """Validate XML content for XXE vulnerabilities."""
    try:
        # Parse XML using ElementTree to check if it's valid XML
        xml = ET.fromstring(xml_content)
    except Exception as e:
        return None  # Return None in case of validation failure

    elements_to_check = [
        'http://www.w3.org/2001/XMLSchema-instance:noNamespaceSchemaLocation',
        'http://example.com/schema'
    ]
    
    for elem_tag in elements_to_check:
        if xml.find(elem_tag) is not None:
            return True  # If found, there might be XXE
        else:
            continue
    return False


def search_for_xxes(url):
    headers = {'User-Agent': 'Python/XMLScan'}
    response = urlopen(f'http://{url}', None, headers=headers)
    xml_content = response.read()

    if validate_xml_content(xml_content):
        print(f'Potential XXE vulnerability detected at {url}')
    else:
        print(f'No XXE vulnerabilities found at {url}')

def main():
    while True:
        target_url = input('Enter the URL to scan for XXEs (or "exit" to quit): ')
        if target_url.lower() == 'exit':
            break
        if not target_url.startswith(('http://', 'https://')):
            target_url = f'http://{target_url}'
        search_for_xxes(target_url)
    

if __name__ == '__main__':
    try:
        main()
    except KeyboardInterrupt:
        print("\nKeyboard Interrupt Detected. Exiting...")
