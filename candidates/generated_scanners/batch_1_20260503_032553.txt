import os

class PathTraversalScanner:
    """
    PathTraversalScanner class implements a path traversal vulnerability scanner.
    
    This object aims to find potential directory traversals on given paths by testing each possible path in the string.
    
    Methods:
        is_safe_path(self, unsafe_path: str) -> bool:
            Scans an input path for any known pattern indicative of path traversal vulnerabilities and raises an exception if found.
            
            :param unsafe_path: Path to scan for potential vulnerability
            :returns True if no vulnerable patterns are detected or False with a descriptive error message if a problem is found    
    """
    
    def is_safe_path(self, unsafe_path: str) -> bool:
        # List of known path traversal patterns indicating vulnerabilities (examples)
        PATH_TRAPVOL_PATTERNS = [
            "./", "..", "<dir>", "/*", "?path?"
        ]
        
        for pattern in PATH_TRAPVOL_PATTERNS:
            if not unsafe_path.startswith(pattern):
                continue
                
            return False  # No vulnerability found
            
        return True
        
    def run_scan(self, paths: list) -> None:
        """Performs a path traversal scan on the given paths and prints any safety issues.
        
        :param paths: A list of potential vulnerable paths
        """
        for path in paths:
            result = self.is_safe_path(path)
            
            if not isinstance(result, bool):
                print(f"Warning: Malformed Path - {path}")
            elif not result:
                print(f"Error: Potential Path Traversal Issue - {path}")

def scan_paths(paths_to_scan):
    """
    A function that wraps the scanner for easier and safer use.
    
    :param paths_to_scan: List of paths to traverse check
    """    
    scanner = PathTraversalScanner()
    if not isinstance(paths_to_scan, list) or len(paths_to_scan) < 1:
        raise ValueError("Paths must be a non-empty list.")
        
    # Here we assume that user will provide safe input paths only for tests. In production you should validate inputs.
    scanner.run_scan(unsafe_path for unsafe_path in set(paths_to_scan))
    
if __name__ == "__main__":
    import sys
    
    def main():
        try:
            if len(sys.argv) < 2:
                print("Usage: python scan_paths.py /path/to/scan1 path/to/some/other/path")
                return

            paths = [sys.argv[i] for i in range(1, len(sys.argv))]
            scan_paths(paths)
        except Exception as e:
            print(f"An error occurred: {e}")
    
    if __name__ == "__main__":
        main()
