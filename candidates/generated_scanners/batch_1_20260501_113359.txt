import xml.etree.ElementTree as ET

class XxeScanner:
    """
    A class to scan for XML External Entities (XXE) vulnerabilities in an input string.
    """

    def __init__(self):
        self.entity_names = set()
        self.logger = init_logger()

    def scan(self, xml_string):
        """ 
        Scans the provided XML string for possible XXE attacks by extracting entity references and tracking them.
        
        Args:
            xml_string (str): The input XML string to be scanned.
            
        Returns:
            list: A list of discovered potential XXE attack entities.
        """
        try:
            root = ET.fromstring(xml_string)
        except ET.ParseError as e:
            print(f"XML Parsing Error: {e}")
            return []

        self._extract_entity_names(root)
        
        result = [entity for entity in self.entity_names if '!' not in entity]
        return result
    
    def _extract_entity_names(self, element):
        """
        Recursively finds all XML entities (including those referenced by the XML or external sources).
        
        Args:
            element (Element): The current xml.etree.Element to recurse through.
        """
        for child in element:
            self._handle_child_element(child)
            self._extract_entity_names(child)
    
    def _handle_child_element(self, element):
        """ 
        Handles processing individual elements within the XML string.
        
        Args: 
            element (Element): The current xml.etree.Element to handle.
        """
        tag_name = element.tag
        if ('@' in tag_name) or ('[' in tag_name):
            self._find_entity_names(tag_name)
    
    def _find_entity_names(self, attr_or_attr_value):
        """ 
        Extracts XML entity names from attribute values and entity references within an element.
        
        Args:
            attr_or_attr_value (str): The XML attribute value or entity reference to be checked.
        """
        if '|' in attr_or_attr_value:
            _, rest = attr_or_attr_value.split('|', 1)
            if len(rest) > 0 and '!"' not in rest:
                self.entity_names.add(attr_or_attr_value)

    def __str__(self):
        return f"XXE Scanner with {len(self.entity_names)} potential XXE attack entities."
