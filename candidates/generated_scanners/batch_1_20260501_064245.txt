import os
import logging
from urllib.parse import unquote

# Configure logging to show only error messages and above (e.g., fatal)
logging.basicConfig(level=logging.ERROR)

class PathTraversalScanner:
    """
    This class is designed to scan a given directory tree for potential path traversal vulnerabilities.
    
    :param base_dir: The root directory to start the scans from. Default is the directory containing scanner.py file.
    """
    
    def __init__(self, base_dir=None):
        if base_dir is None:
            self.base_dir = os.path.dirname(os.path.abspath(__file__))
        else:
            self.base_dir = os.path.expanduser(base_dir)
        self.scan_queue = [self.base_dir]
        self.visited_dirs = set()
        self.errors = []
    
    def scan(self):
        """
        Starts the path traversal scan.
        
        The method starts by setting up initial directory queue and then starts iterating over it,
        performing recursive scans on each new directory until no more directories are found.
        It handles errors gracefully, logging any issues encountered during the process.
        """
        while self.scan_queue:
            current_dir = self.scan_queue.pop()
            
            try:
                if os.path.isdir(current_dir) and current_dir not in self.visited_dirs:
                    # Verify that only allowed directory traversal is checked
                    if current_dir == unquote(unquote(os.path.dirname(current_dir))): 
                        logging.error(f"Potential Path Traversal Vulnerability: {current_dir}")
                        self.errors.append((current_dir, os.listdir(current_dir)))
                    
                    self.visited_dirs.add(current_dir)
                    try:
                        for dirpath, _, filenames in os.walk(current_dir):
                            self.scan_queue.extend(os.path.join(dirpath, f) 
                                                   for f in filenames if os.path.isdir(os.path.join(dirpath, f)))
                    except PermissionError as e:
                        logging.error(f"Permission denied at directory: {dirpath}. Skipping.")
            except Exception as e:
                logging.error(f"Error while scanning directory {current_dir}: {e}")
    
    def summarize(self):
        """
        Prints out errors encountered during the scan.
        
        :return: An error summary dictionary with each scanned directory and its problematic entries
                 (If any).
        """
        return {error[0]: error[1] for error in self.errors}

def main():
    """
    The entry point of this script where you initialize a PathTraversalScanner instance 
    and perform the scan.
    
    Usage:
    ./path_traversal_scan.py <directory>
    
    Where <directory> is any directory, e.g., ~/documents
    If no argument provided (defaults to .), it scans your project path.
    """
    from argparse import ArgumentParser
    
    parser = ArgumentParser(description="Path Traversal Scanner")
    parser.add_argument('path', nargs='?', help='The directory to start scanning.')
    
    args = parser.parse_args()
    
    scanner = PathTraversalScanner(args.path)
    scanner.scan()
    errors_summary = scanner.summarize()
    print(f"Potential Path Traversal Vulnerabilities Found: {len(errors_summary)}")
    for dirpath, entries in sorted(errors_summary):
        print(f'In directory "{dirpath}":')
            "\n".join(entries))

# This part will be executed when you run this script directly.
if __name__ == "__main__":
    main()
