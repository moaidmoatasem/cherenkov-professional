import os


class CommandInjectionScanner:
    def __init__(self):
        self.whitelisted_commands = {
            'ls', 'whoami', 'pwd'
        }  # Whitelisted commands to avoid injection of malicious commands
    
    def scan_command(self, command: str) -> None:
        if not isinstance(command, str):
            raise TypeError("Input must be a string")
        
        for token in os.path.expandvars(os.path.expanduser(command)):
            if self.is_potentially_harmful_token(token):
                sanitized_command = self.sanitize_command(command)
                print(f"Potential command injection detected! Command might have been tampered: {command}")
                print("Sanitized Output:", sanitized_command)
    
    def is_potentially_harmful_token(self, token: str) -> bool:
        # This function could be expanded to include more sophisticated checks
        return token == "&&"  # Example sensitive string

    def sanitize_command(self, command: str) -> str:
        """Sanitize the input command by replacing or escaping potentially harmful tokens"""
        commands = command.replace('&&', '').split(';')  # Split commands with semicolon as a delimiter
        sanitized_commands = [f"{cmd} &&" for cmd in commands]  # Re-add '&&' at end of each split command
        
        return "; ".join(sanitized_commands)


def main():
    scan_instance = CommandInjectionScanner()
    
    # Example usage:
    benign_command = "ls -l && whoami;"
    malicious_command = "whoami; rm *"
    
    print("\nExample 1: Scanning a Benign, Whitelisted Command")
    scan_instance.scan_command(benign_command)
    
    print("\nExample 2: SCannig a Malicious Command with Injection Attempt")
    scan_instance.scan_command(malicious_command)

if __name__ == "__main__":
    main()

# Example tests
def test_scan_command():
    scanner = CommandInjectionScanner()
    benign_bash_cmd = "echo 'Hello, World' && date;"
    malicious_cmd_with_exploit = "rm -rf *; echo 'Malicious Command'"
    
    assert not scanner.is_potentially_harmful_token(os.path.expandvars(os.path.expanduser(benign_bash_cmd)).split(';'))
    
    # Malicious string contains a command injection exploit ('&&')
    assert scanner.is_potentially_harmful_token([token for token in os.path.expandvars(os.path.expanduser(malicious_cmd_with_exploit))])
    
    print("All tests passed successfully!")

if __name__ == "__main__":
    test_scan_command()
