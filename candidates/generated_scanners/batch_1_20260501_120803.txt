#!/usr/bin/env python3
"""
This script implements a path traversal scanner to check if there are any exploitable paths in URLs,
files, or directory structures. It uses various checks including path splitting, URL encoding, and file existence checks.
"""

import os
from urllib.parse import urlparse

class PathTraversalScanner:
    def __init__(self):
        self._safe_paths = set()

    def _add_path(self, base_url, url_fragment):
        """ Safely add a path based on whether it was passed through URL encoding. """
        try:
            safe_path = os.path.normpath(urlparse(base_url)._replace(path=url_fragment).geturl().strip('/'))
            self._safe_paths.add(safe_path)
        except Exception as e:
            print(f"Exception in adding path: {e}")

    def scan(self, base_url):
        """ Scans the given URL for any exploitable paths. """
        parsed_base = urlparse(base_url)

        # Check full URL
        safe_path_full = os.path.join(parsed_base.netloc, parsed_base.path) if parsed_base.path.startswith('/') else parsed_base.geturl()
        self._add_path(base_url=base_url, url_fragment=safe_path_full)

        # Extract query components from base URL if present
        safe_query = {f'{k}={v}' for k, v in parsed_base.query.split('&')}

        # Check full path without the query string
        safe_path = os.path.join(parsed_base.netloc, parsed_base.path) + '?' + '&'.join(safe_query)
        self._add_path(base_url=base_url, url_fragment=safe_path)

        # Split URL into pieces based on forward slashes. This helps in identifying potential paths.
        for i in range(1, len(parsed_base.path.split('/'))):
            safe_path = os.path.join(*parsed_base.path.split('/')[:i], *safe_query)
            self._add_path(base_url=base_url, url_fragment=safe_path)

        # Check if the base URL itself matches any exploitable paths
        for i in range(1, len(parsed_base.netloc.split(':'))):
            safe_path = os.path.join(*parsed_base.netloc.split(':')[2*i-2:i*2], *safe_query)
            self._add_path(base_url=base_url, url_fragment=safe_path)

    def report(self):
        """ Returns a set of all paths identified as potentially exploitable. """
        return self._safe_paths

def main():
    # Example usage with a URL to scan
    base_url = "http://example.com/secure/path"
    scanner = PathTraversalScanner()
    scanner.scan(base_url)
    print("Exploitable Paths Found:", scanner.report())

if __name__ == "__main__":
    main()

import pytest

def test_path_traversal_scan():
    """ Verify the path traversal scanner identifies all potential exploitable paths. """
    base_urls = [
        "http://example.com/user1",
        "ftp://user:pass@localhost/home/user1/",
        "ssh://user:pass@host/path/with/no/dir/"
    ]
    # Expected safe_path_full matches with the full URL
    expected_safe_paths = {
        "http://example.com/user1",
        "ftp://user:pass@localhost/home/user1/",  # Matches when encoded and base path is "/"
        "ssh://user:pass@host/path/with/no/dir/"
    }

    for base_url in base_urls:
        scanned = PathTraversalScanner()
        scanned.scan(base_url)
        assert scanned.report() == expected_safe_paths, f"Got unexpected safe paths from {base_url}!"
