        else:
            raise ValueError(f"Unknown CVE identifier: {cve_id}")
    except Exception as e:
        raise ValueError(f"Error calculating severity score for '{cve_id}': {e}")


class VulnerabilitySeverity:
    def __init__(self, cves):
        self.cves = cves

    def get_severities(self):
        """
        Generate a list of dictionaries containing the CVE ID and calculated vulnerability severity scores.

        Args:
            cves (list): List of CVE identifiers.

        Returns:
            list: A list of dictionaries with 'cve_id' and 'score' keys.
        
        """
        vulnerabilities = []
        for cve in self.cves:
            try:
                score = calculate_severity_score(cve)
                severity_dict = {
                    "cve_id": cve,
                    "score": f"{score:.1f}"   # Limit CVSS score to one decimal place
                }
                vulnerabilities.append(severity_dict)
            except ValueError as e:
                print(f"Could not process CVE: {cve}. Reason: {e}")
        return vulnerabilities
