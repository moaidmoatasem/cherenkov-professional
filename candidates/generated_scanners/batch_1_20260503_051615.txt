```python
# .github/workflows/feature_scan.yml

name: Feature Scan

on:
  push:
    branches:
      - main
      - develop
  pull_request:
    branches: [main]

jobs:
  main:
    runs-on: ubuntu-latest
    name: Run feature scanner
    timeout: 240m
    steps:
      - uses: actions/checkout@v3
        with:
          token: ${{ secrets.GITHUB_TOKEN }}
      - name: Setup Python
        uses: actions/setup-python@v4
        with:
          python-version: "3.8"
      - run: python --version
      - name: Install dependencies
        run: |
          pip install numpy pandas pytest
      - name: Run feature scanner
        env:
          PATH_FEATURES: "/path/to/features/scripts/"
        run: |
          coverage run --omit "./venv/*" --include "*.py" "$(echo ${PATH_FEATURES})*"
      - name: Upload to Codecov
        uses: codecov/codecov-action@v1.5.0
```
