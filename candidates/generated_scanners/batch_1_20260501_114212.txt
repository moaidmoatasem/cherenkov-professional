#!/usr/bin/env python

"""
Module description:
This module provides functions to create dynamic HTML dashboard reports.
It includes features such as error handling and comprehensive documentation for clarity.

Functions:
- generate_dashboard_report(): Generates an HTML-based dashboard report.
- process_report_data(data): Processes the input data before creating the report, supports basic validation.

Example Usage:
>>> from html_dashboard_reports import generate_dashboard_report
>>> data = {"metrics": [{"name": "Memory Usage", "value": 60}, {"name": "CPU Usage", "value": 55}]}
>>> generate_dashboard_report(data)

"""

import os

class DashboardGenerator:
    def __init__(self, report_dir):
        self.report_dir = report_dir
        if not os.path.exists(self.report_dir):
            os.makedirs(self.report_dir)
            
    def process_report_data(self, data):
        """
        Processes the given data for generating an HTML dashboard report.
        
        Args:
        - data: A dictionary containing metrics to be included in the report.

        Returns:
        - None. It modifies a file named 'index.html' in the current directory.
        
        Notes:
        This is a placeholder function for simplicity. In actual application, more complex logic might be required.
        """
        index_filepath = os.path.join(self.report_dir, "index.html")
        with open(index_filepath, "w", encoding="utf-8") as f:
            f.write("<!DOCTYPE html>\n<html lang='en'>\n<head><meta charset='UTF-8'></head>\n<body>")
            
            for metric in data.get("metrics", []):
                value = getattr(metric, 'value', None)
                if isinstance(value, int) and 0 <= value <= 100:
                    f.write(f"<p>{metric['name']}: {value}%</p>")
                else:
                    raise ValueError("Metrics need to be integers in the range of 0-100.")
                
            f.write("</body>\n</html>")
    
    def generate_dashboard_report(self, data):
        """
        Generates an HTML dashboard report by processing 'data' and writing the result to a file.
        
        Args:
        - data: A dictionary containing metric data.

        Returns:
        None. This function writes HTML content to the current working directory identified by self.report_dir.
        
        Example Usage:
            generate_dashboard_report({"metrics": [{"name": "Memory Usage", "value": 60}, {"name": "CPU Usage", "value": 55}]}
        """
        try:
            self.process_report_data(data)
            
        except Exception as e:
            print(f"An error occurred during generation: {e}")
        

def generate_dashboard_report(data):
    generator = DashboardGenerator(".")
    generator.generate_dashboard_report(data)

# Example usage, could be triggered by other scripts or users to provide data
if __name__ == "__main__":
    data = {
        "metrics": [
            {"name": "Memory Usage", "value": 60},
            {"name": "CPU Usage", "value": 55}
        ]
    }
    
    generate_dashboard_report(data)
