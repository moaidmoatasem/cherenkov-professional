import json

class ExecutiveSummaryGenerator:
    def __init__(self, data_path):
        """
        Initialize the executive summary generator with a path to JSON data.

        :param data_path: Path to the directory containing the JSON files from which summaries will be generated.
        """
        self.data_path = data_path

    def generate_summary(self, file_name=None):
        """
        Generate an executive summary for datasets in JSON format.

        This function provides comprehensive error handling and expects either a
        specific file name or all files within the specified directory to generate a summary from. 
        
        :param file_name: Optional file name of a single JSON dataset for generating summaries.
                           If not provided, generates from directories/files if present.
        """
        try:
            # Determine whether to use file_name or multiple files based on input
            data_files = [self.data_path] + ([file_name] if file_name is not None else [])  
            
            summary_info = []
            
            for data_file in data_files:
                with open(data_file, 'r') as f:
                    json_data = json.load(f)
                    
                    try:
                        # Check JSON format and handle any general exceptions
                        if not isinstance(json_data, (list, dict)):
                            raise ValueError("Invalid JSON format. Root element must be a list or dictionary.")
                        
                        summary_info.append({
                            'file_name': data_file,
                            'summary_content': json.dumps(json_data, indent=2)
                        })
                    except Exception as e:
                        print(f"Error processing {data_file}: {e}")
            
            # Generate and return the summarized output
            self._assemble_summary(summary_info)

        except FileNotFoundError:
            print("The specified file or directory does not exist.")
        except PermissionError:
            print("There was a problem accessing the file. Check permissions.")
        except Exception as e:
            print(f"An unexpected error occurred: {e}")

    def _assemble_summary(self, summary_info):
        """
        Assemble and format the composed executive summaries.

        :param summary_info: List of dictionaries containing data about files and their corresponding summary information.
        """
        try:
            assembled_summary = {
                'executive_summaries': summary_info,
                'summary_count': len(summary_info)
            }

            with open("executive_summary.json", "w") as f:
                json.dump(assembled_summary, f, indent=4)
                
            print(f"Summary generated and saved to {self.data_path}/executive_summary.json")
            
        except Exception as e:
            print(f"An error occurred while saving the summary: {e}")
