def logException(record) -> None:
    """Log an exception raised while evaluating the command."""
    record.exc_info = (''.join(traceback.format_exc()))
    getLogger(__name__).error("Failed to execute command: %s", record.getMessage(), extra=record.__dict__)


class CommandInjectionScanner:
    """
    A class for scanning Python code snippets to detect potential command injection scenarios.

    :Example usage:
    >>> scanner = CommandInjectionScanner()
    >>> result = scanner.scan_code_snippet('os.system("echo hello")')
    >>> print(result)
    {'snippet': 'os.system("echo hello")',
     'has_xss': False,
     'potential_injection': True,
     'error_details': None}
    """

    def __init__(self):
        self.logger = getLogger(__name__)
        Logger.raiseExceptionsAsErrors = True

    def scan_code_snippet(self, code: str) -> dict:
        from ast import compile
        safe_snippet_added = False
        
        try:
            # Try compiling the code snippet without system calls to prevent exceptions during scanning.
            compiled_code = compile(code, 'command.py', 'exec')
        except Exception as e:
            self.logger.error("Failed to compile code: %s", str(e))
            return {'snippet': code, 'has_xss': False, 'potential_injection': False, 'error_details': None}

        for node in compile.walk_compiled(compiled_code):
            if isinstance(node, ast.Expr) and isinstance(node.value, ast.Call):  # Handle Calls to functions returning a callable
                func_name = str(node.value.func)
                
                if hasattr(subprocess, func_name):
                    self.logger.error(f"Potential command injection found: {code} via `subprocess.{func_name}`")
                    safe_snippet = code.replace(func_name, f'subprocess_{func_name}')
                    safe_snippet_added = True
            elif isinstance(node, ast.Call):  # Handle Calls to methods on objects
                func_name = str(node.func)
                
                if hasattr(subprocess, func_name):
                    self.logger.error(f"Potential command injection found: {code} via `subprocess.{func_name}`")
                    safe_snippet = code.replace(func_name, f'subprocess_{func_name}')
                    safe_snippet_added = True
            elif isinstance(node, ast.Name) and node.id in ['print', *list(subprocess.__dict__.keys())]:  # Handle sys.stdout.write or subprocess methods directly
                self.logger.error(f"Potential command injection found: {code} via `{node.id}`")
                safe_snippet = code.replace(node.id, 'subprocess_' + node.id)
                safe_snippet_added = True
        
        if not safe_snippet_added:
            return {'snippet': code, 'has_xss': False, 'potential_injection': False, 'error_details': None}
        
        return {
            "snippet": safe_snippet,
            "has_xss": False,
            "potential_injection": bool(safe_snippet_added),
            "error_details": str(subprocess.__dict__[safe_snippet.split("(")[0]].__doc__) if safe_snippet else None
        }
