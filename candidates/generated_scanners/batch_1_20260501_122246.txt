import requests
from urllib.parse import urljoin

class CVEScanner:
    """
    A class to fetch and validate CVE information.
    """
    
    def __init__(self):
        self.base_url = "https://services.nvd.nist.gov/rest/json/cves/"
        
    def get_cve(self, cve_id: str) -> dict:
        """
        Fetches the details of a specific CVE.

        :param cve_id (str): The CVE ID.
        :return: A dictionary representing the CVE details or None if not found.
        """
        try:
            response = requests.get(urljoin(self.base_url, str(cve_id)), timeout=10)
            response.raise_for_status()
        except requests.RequestException as e:
            print(f"Request error: {e}")
            return None
        
        return response.json()['nvdcve-2.0']['vulnerabilities'][cve_id]
        
    def find_cve(self, vulnerability_name: str) -> list[str]:
        """
        Searches CVEs matching a specific name and returns them.

        :param vulnerability_name (str): The search query.
        :return: A list of CVE IDs found or an empty list if not found.
        """
        response = self.get_cve(vulnerability_id=None)
        if not response:
            return []
        
        vulnerabilities = response['CVE_Items']
        matches = [item['cve']['CVE_data_meta']['ID'] for item in vulnerabilities
                   if vulnerability_name.lower() in item['cve']['summary'].lower()]
        
        return matches

# Example of usage
