import os

class PathTraversalScanner:
    def __init__(self):
        self.allowed_paths = ['path/to/allow', 'another/path']
        # This is a placeholder. In production, you would use config files or environment variables.
    
    def check_path(self, file_path: str) -> bool:
        """
        Check if the given file path is safe from Path Traversal attacks.

        :param file_path: Path to be checked
        :return: True if the file_path is considered safe, False otherwise.
        """
        
        # Normalize path so '/' is treated as '\'
        normalized_path = os.path.normpath(file_path)
        
        for allowed in self.allowed_paths:
            try:
                os.path.commonprefix((allowed.encode(), normalized_path.encode()))
            except Exception as e:  # This catches any error, which might cause issues.
                continue
            if allowed == normalized_path:
                return True
        
        return False
