#!/usr/bin/env python

"""
XXEScanner - A tool to detect insecure use cases of XML External Entities (XXE).
"""

import xml.etree.cElementTree as ET


def xxe_scan(xml_string):
    """
    Scans a given XML string for potential XXE vulnerabilities.
    
    :param xml_string: A string containing XML data.
    :return: A tuple with the original XML and any detected issues. 
             Returns None if no issues are found.
    """
    xml_data = ET.fromstring(xml_string)
    
    warnings = []
    
    # This is a simplified heuristic to detect XXE, should be expanded
    for elem in xml_data.iter():
        elem_type = type(elem).__name__
        is_entity_name_element = any([
            f"{elem.tag}.NAME".startswith("xml.") and elem.text is not None,
            f"{elem.tag}.CONTENT".startswith("xml.")
        ])
        
        if "ENTITY" in elem_type or is_entity_name_element:
            warnings.append(f"Potential XXE issue found: {repr(elem)}")
    
    # If no issues are found, it's safe to return none
    return xml_data, warnings

def scan_xml_file(xml_file_path):
    """
    Loads and scans a XML file for XXE vulnerabilities.
    
    :param xml_file_path: The path to the XML file to be scanned.
    :return: A tuple with the original XML data from the file and any detected issues. 
             Returns None if no issues are found in the entire document.
    """
    try:
        with open(xml_file_path, "r", encoding="utf-8") as file:
            xml_string = file.read()
            return xxe_scan(xml_string)
    except Exception as e:
        # General error handling for I/O operations
        print(f"Error reading XML from file: {e}")
        return None


if __name__ == "__main__":
    import sys
    
    if len(sys.argv) != 2:
        print("Usage: xxe_scanner.py <xml_file_path>")
        sys.exit(1)
    
    xml_path = sys.argv[1]
    original_xml, detected_issues = scan_xml_file(xml_path)
    
    if detected_issues:
        print(f"Detected Issues in XML File {xml_path}:\n")
        
        for issue in detected_issues:
            print(issue)
    else:
        print("No XXE issues found.")
