import subprocess

class ExploitGenerator:
    """
    Class to generate exploits automatically using a predefined set of scripts.
    
    The class is responsible for generating exploits from templates based on input parameters.
    """

    def __init__(self, template_path):
        self.template_path = template_path
        
    @staticmethod
    def validate_inputs(data_in: dict) -> None:
        """ Validate input data. """
        if not isinstance(data_in, dict) or len(data_in) < 1:
            raise ValueError("Input must be a non-empty dictionary of parameters.")
    
    def generate_exploit(self, template_path=None):
        """
        Generate an exploit based on provided template.
        
        :param template_path: Path to the template script if different from default.
        :return: None. Executes the generated Python script and prints output for confirmation.
        :raises ValueError: If input data is invalid or missing essential parameters.
        """
        self.validate_inputs(data_in=self.__class__.__dict__)
        
        # Construct full path to the template, overriding with user-provided if necessary
        final_template_path = template_path or self.template_path
        
        try:
            exploit_script = subprocess.run(["python", final_template_path], capture_output=True)
            
            print(f"\n\nGenerated exploit:\n{exploit_script.stdout}\n")
        
        except Exception as e:
            raise ValueError("Error generating the exploit. Please check your template and input parameters.")
  
def main():
    """
    Entry point to handle user interactions, such as setting up inputs or launching the generation.
    
    This function is more of a sample usage scenario for the ExploitGenerator class.
    """
    # Example validation data
    data_in = {"param1": "value1", "endpoint": "/path/to/vulnerable/page"}
    
    generator = ExploitGenerator(template_path="exploit_template.py")
    
    print("\nGenerating exploit using predefined parameters...")
    generator.generate_exploit(final_template_path=generator.template_path)
    
if __name__ == "__main__":
    main()
