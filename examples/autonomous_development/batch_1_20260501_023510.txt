def send_message_to_slack(url: str, token: str, channel: str, message: str) -> None:
    """
    Sends a message to a specified Slack channel using URL authentication.
    
    :param url: API endpoint of the Slack server with URL Authentication enabled
    :param token: Token associated with the provided URL for this specific integration
    :param channel: The channel ID where you want to send the message (e.g., #general or DM)
    :param message: The body of the message to be sent as a string
    """
    
    url_response = requests.get(url, params={'token': token})
    if not url_response.ok:
        raise ValueError(f"Could not authenticate with Slack using URL. Response was {url_response.text}")
    
    # Assuming you have obtained an access token through some means (likely OAuth)
    headers = {'Authorization': f'Bearer {token}'}
    response = requests.post(url=url, headers=headers, json={'channel': channel, 'text': message})
    
    if not response.ok:
        error_message = response.json().get('error', 'Unknown failure')
        raise RuntimeError(f"Failed to send message to Slack: {error_message}")
