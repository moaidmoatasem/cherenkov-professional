name: GitHub-Action-Scanner

on:
  pull_request:
    branches:
      - main
  push:
    branches:
      - main

jobs:
  scan-feature:
    runs-on: ubuntu-latest
    permissions:
      actions: read write
      contents: read
      pull-requests: write

    name: Scan PR Features
    steps:
    - uses: actions/checkout@v2
    - name: Set up Python
      uses: actions/setup-python@v2
      with:
        python-version: '3.8'
    - name: Install dependencies
      run: pip install requests pytest
    - name: Check PR Features
        id: feature-check
      env:
        API_TOKEN_NAME: GH_TOKEN
      run: |
          import os
          import json
          import requests

          API_URL = "https://api.example.com/features"

          def check_api_access():
              try:
                  response = requests.get(API_URL, headers={"Authorization": f"Token {os.environ['GH_TOKEN']}"})
                  if response.status_code != 200:
                      raise Exception(f"Failed to access feature: HTTP Status Code - {response.status_code}")
              except Exception as e:
                  print(e)
                  return False

          def fetch_feature_data():
              try:
                  response = requests.get(API_URL, headers={"Authorization": f"Token {os.environ['GH_TOKEN']}"})
                  data = json.loads(response.text) if response.content else {}
                  features = [feature for feature in data["features"] if feature["status"] == "ACTIVE"]
                  return features
              except Exception as e:
                  print(e)
                  return []

          def check_features():
              non_local_features = fetch_feature_data()
              active_features = []
              missing_features = []

              for feature in non_local_features:
                  try:
                      response = requests.get(feature["url"], headers={"Authorization": f"Token {os.environ['GH_TOKEN']}"})
                      if response.status_code == 200 and json.loads(response.text)['status'] == "ACTIVE":
                          active_features.append(feature)
                      else:
                          missing_features.append(feature)
                  except Exception as e:
                      print(e)
              return len(active_features), len(missing_features)

          def main():
              success = check_api_access() and check_features()
              if not success:
                  raise Exception("Feature scan failed")

          result, failure_count = check_features()

          # Determine action based on the outcome
          if failure_count == 0:
            print(f"All features are up-to-date. Found {len(active_features)} active features out of {result} total checks.")
            os.Exit(0)
          else:
            print(f"Feature scan completed, but some features are missing or outdated. Found {success['active']} active features out of {result} total checks.")
            os.Exit(-1)

      env: ${{ steps.feature-check.outputs.GH_TOKEN }}
      shell: bash

    - name: Output feature status
      env:
        GH_FEATURES: $(jq --null-input '[ "feature1", "feature2" ] [] ' -
                                  [ "non-active-feature1", "non-active-feature2" ] [])
          # Example of fetching additional features
      run: |
        echo $GH_FEATURES

# Testing section (For production-readiness, this part is optional and can be removed per preference)

jobs:

    - name: Set up Python

    - name: Test scanning functionality

        pytest
