import requests
from typing import NoReturn

class WebhookNotifier:
    """
    A class to handle sending webhooks for notifications.
    
    Methods:
        send_webhook(url: str, payload: dict) -> None:
            Sends a webhook notification using the provided `url` and `payload`.
            
    Raises:
        requests.exceptions.RequestException: If the request fails.
    
    Usage example:
        notifier = WebhookNotifier()
        response_code = notifier.send_webhook("https://example.com/notification", {"event": "user_login"})
        print(f"Webhook sent with status code: {response_code}")
    """
    def send_webhook(self, url: str, payload: dict) -> None:
        try:
            # Adjust the headers and auth if needed (e.g., API keys)
            response = requests.post(url, json=payload)
            response.raise_for_status()
        except requests.exceptions.RequestException as e:
            raise requests.exceptions.RequestException(f"Webhook failed to send. Error: {str(e)}")
        
if __name__ == "__main__":
    notifier = WebhookNotifier()
    url = "https://example.com/notification"
    payload = {"event": "user_login"}
    
    # Example usage
    response_code = notifier.send_webhook(url, payload)
    print(f"Webhook sent with status code: {response_code}")
