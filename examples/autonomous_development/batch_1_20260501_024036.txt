class RetryManager(ABC):
    """
    Abstract base class for custom retry implementations.
    Subclasses should define their own retry strategy based on specific needs.
    """

    @abstractmethod
    def should_retry(self, error=None) -> bool:
        """Decide whether to retry a transactional operation given an error."""
        raise NotImplementedError

        self.backoff_factor = backoff_factor

        Args:
