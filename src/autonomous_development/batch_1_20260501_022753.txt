#!/usr/bin/env python

import xml.etree.ElementTree as ET


def xxetool(xml_data):
    """
    This function checks for XXE (XML External Entity) vulnerabilities in an XML document.
    
    Args:
        xml_data (str): The XML data to be checked.

    Returns:
        bool: True if XXE is detected, False otherwise.
        
    Raises:
        ValueError: If the provided data contains improperly formatted XML. 
    """
    try:
        # Parse the XML using ElementTree
        root = ET.fromstring(xml_data)
        entity_namespaces = {}
        
        for node in root.iter():
            for attr_name in ["{{{}}}".format(n) for n in node.nsmap]:
                if getattr(node, attr_name, None):
                    entity_namespaces[attr_name] = node.get(attr_name)
        
        # Create an environment to evaluate XML entities
        xml_env = _get_xml_env(entity_namespaces)
        
        result = _xmlschema_check(root, xml_env)
    
        return bool(result)
    except Exception as e:
        raise ValueError("Malformed XML data provided.")

def _get_xml_env(entities):
    """Helper function to create a clean XML env for evaluation."""
    import xml.dom.minidom as minidom
    from xml.etree.ElementTree import Element
    
    tree = minidom.parse('non_existent_file.xml')  # This file should not exist to simulate safe environment
    root2 = tree.documentElement if tree else "invalid"
    
    env = {
        'ENTITY': entities,
        '_current_element': _get_fake_element(root2)
    }
    return env

def xxetool_xmlschema_check(node, entity_ns):
    """Check the XML document for XXE vulnerabilities using xmlschema."""
    from xml.etree.ElementTree import Element
    
    if type(node) is not Element:
        raise ValueError(f"Node {node} of type {type(node)} is not a valid node.")
    
    nsmap = {k: v.get("{http://www.w3.org/XML/1998/namespace}namespaceURI", '')
             for k, v in entity_ns.items()}
    
    result = _xmlschema_check(node, nsmap)
    return bool(result)

def _get_fake_element(root):
    """Helper function to create a clean fake element."""
    from xml.etree.ElementTree import Element
    try:
        return Element("", root.tag, root.attrib) if hasattr(root, 'attrib') and root.attrib else Element("dummy")
    except AttributeError as e:
        raise ValueError(f"Root {repr(root)} is not an XML element and does not allow extraction of attributes.")
