from urllib.parse import urlparse


class CVEDatabaseIntegration:
    def __init__(self):
        self.cve_api_url = 'https://api.cvedb.com/search'

    def fetch_cve_data(self):
        """
        Fetches data from the CVE database API.

        Returns:
            A dictionary containing the API response. If successful, it will contain at least a 'results' key with further details.
            It can also include error information if unsuccessful (e.g., due to invalid API access).

        Raises:
            ValueError: In case of issues in processing the response data or when errors are encountered.

        """
        parser = urlparse(self.cve_api_url)
        cvedb_token = 'your_cvedb_api_token'  # Replace with actual application token
        params = {
            'q': '',
            'output': 'json',
            'token': cvedb_token,
        }

        try:
            response = requests.get(parser.geturl(), params=params)

            if not response.ok:
                raise ValueError(f"API Request failed. Got status code {response.status_code}")

            return response.json()
        except Exception as e:
            raise ValueError(f"An error occurred while fetching CVE data: {str(e)}")

    def integrate_cves(self, cve_data):
        """
        Integrates fetched CVE details into the system.

        Args:
            cve_data (dict): The JSON-like structure returned by the fetch operation.

        Returns:
            A list of integrated CVEs. Each CVE is a dictionary with 'cve_id', 'description', and other key-value pairs as specified within the API schema.

        """
        if not isinstance(cve_data, dict) or cve_data.get('results') is None:
            raise ValueError("Invalid data structure received from the vulnerable API")

        return [data for data in cve_data['results']]
