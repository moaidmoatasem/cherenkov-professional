import os

from datetime import datetime


class ExecutiveSummaryGenerator:
    """
    A class to generate an executive summary based on provided data.
    
    Methods:
        _validate_data(data): Validates incoming data for missing or unsupported fields.
        _process_and_format(data): Prepares the data in a structured format for writing into files.
        generate_summary(output_dir, data) -> None: Generates and formats data into formatted text and JSON summary files.
    
    """

    def __init__(self):
        self._output_dir = os.path.join(os.getcwd(), "executive_summaries")
        
    def _validate_data(self, data):
        """
        Validates the incoming data for mandatory fields like 'title', 'start_date', and 'end_date'.
        
        Args:
            data (dict): Data provided to generate summaries.
            
        Returns:
            None
        
        """
        if not all(key in data for key in ('report_title', 'start_date', 'end_date')):
            raise ValueError("Data missing mandatory fields: report title, start date and end date.")
    
    def _process_and_format(self, data):
        """
        Processes the given data to generate executive summary formatted text.

        Args:
            data (dict): Data dictionary containing necessary details for producing the summary.
        
        Returns:
            str: Formatted description of the provided summary data.

        """
        # Example processing logic
        current_time = datetime.now().strftime("%Y-%m-%d %H:%M")
        processed_data = {
            'title': f'[ExecSummary] {data["report_title"]}',
            'date_generated': current_time,
            'data': [
                {'text': f'Summary of report "{data["report_title"]}"'},
                {'text': f'Generated on: {current_time}'}
            ]
        }
        
        return str(processed_data)
    
    def generate_summary(self, output_dir):
        """
        Generates an executive summary and formats it into text files.

        Args:
            output_dir (str): Directory path to save the generated executive summaries.
        
        Raises:
            ValueError: If input data is missing required fields.
            
        Returns:
            None
            
        """
        os.makedirs(output_dir, exist_ok=True)
        
        # Example sample_data
        sample_data = {
            'report_title': 'Weekly Security Report',
            'start_date': datetime.strftime(datetime.now() - timedelta(days=7), "%b %d"),
            'end_date': datetime.strftime(datetime.now(), "%b %d")
        }
        
        self._validate_data(sample_data)
        
        summary_text = self._process_and_format(sample_data)
        
        text_file_path = os.path.join(output_dir, f"{sample_data['report_title']}_executive_summary.txt")
        with open(text_file_path, 'w') as file:
            file.write(f"Generated: {datetime.now().strftime('%Y-%m-%d %H:%M')}\n\n{text_file_path}")
        
        json_output_path = os.path.join(output_dir, f"{sample_data['report_title']}_exec_summary.json")
        with open(json_output_path, 'w') as file:
            json_file = {"Summary": summary_text}
            file.write(str(json_file))
        
    def __call__(self):
        """
        Entry point method to generate summaries. Automatically starts data validation and generation.
            
        Returns:
            None
        
        """
        self.generate_summary(self._output_dir)
        

def create_executive_summary_sample_data():
    """
    Creates sample data for testing purposes based on given constraints.

    Args: 
        None

    Returns:
        dict: Sample dictionary containing mandatory fields for generating executive summary.

    """
    return {

        'start_date': (datetime.now() - timedelta(days=2)).strftime("%Y-%m-%d"),
        'end_date': datetime.now().strftime("%Y-%m-%d")
    }
    

def main():
    output_dir = os.path.join(os.getcwd(), "output")
    generator = ExecutiveSummaryGenerator()
    sample_data= create_executive_summary_sample_data()
    summary_generator = generator(sample_data)
 
if __name__ == "__main__":
    main()
