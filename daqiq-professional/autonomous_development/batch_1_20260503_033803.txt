import xml.etree.ElementTree as ET

class XxeScanner:
    """
    A class to scan for XML External Entities (XXE).

    This scanner is designed to test if XML documents contain potentially harmful XXEs.
    
    Methods:
        - search_xxe(xml_content): Scans the content and returns any found external entities.
    """

    def __init__(self):
        self.element_tree = None

    def parse_xml(self, xml_string):
        """
        Parses an XML string into a tree structure to be used for scanning.

        Parameters:
            - xml_string (str): An XML string to parse.

        Returns:
             etree._ElementTree: A parsed Element Tree object.
        """
        try:
            self.element_tree = ET.fromstring(xml_string)
        except ET.ParseError as e:
            raise ValueError(f"XML parsing error: {e}")
        
        return self.element_tree

    def search_xxe(self, xml_content):
        """
        Scans the provided XML content for External Entity invocations.

        Parameters:
            - xml_content (str): The XML string to scan.

        Returns:
            list: A list of Element tags that reference external entities.
        """
        if not self.element_tree:
            self.parse_xml(xml_content)

        found_xxe_tags = []

        # Iterate through all elements in the root
        def _scan(element):
            for child in element:
                _scan(child)
                
                if '://' in element.tag:  # This is a possible indicator for XXE
                    found_xxe_tags.append(element)

        _scan(self.element_tree.getroot())
        
        return found_xxe_tags

def example_usage():
    """
    Example usage of the XxeScanner class.
    """

    xml_content = """<?xml version="1.0"?>
<document>
    <user id="123">
        <name>John Doe</name>
        <!-- Non-XXE content -->
    </user>
    
    <bad_user ref="malicious.xml">I'm external!</bad_user> <!-- This is potentially an XXE -->
</document>"""

    scanner = XxeScanner()
    potential_xxes = scanner.search_xxe(xml_content)
    
    if potential_xxes:
        print("Potential XML External Entities found:")
        for tag in potential_xxes:
            xml_tag = ET.tostring(tag).decode().strip()
            print(f"Tag: {xml_tag}")
    else:
        print("No potential XML External Entities found.")

if __name__ == "__main__":
    example_usage()
