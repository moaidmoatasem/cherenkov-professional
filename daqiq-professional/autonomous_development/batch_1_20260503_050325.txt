```python
# This script implements a lightweight scanner as part of a CI/CD pipeline. 
# It is designed to check for sensitive data in code files.
#
# Features:
# - Checks .py and .sh file types for passwords (hard-coded or in comments)
# - Logs details about each issue found
# - Implements error handling for file reading failures, empty files, etc.

import os

def scan_files(directory: str) -> None:
    """
    Scans a directory tree for Python (.py) and Shell (.sh) scripts, checks for 
    hard-coded passwords and other sensitive data.
    
    Args:
        directory (str): The path to the directory to be scanned.
        
    Returns:
        None
    """
    
    # Ensure we have an existing directory
    if not os.path.isdir(directory):
        print("Error: Directory does not exist. Please provide a valid path.")
        return
    
    # Iterate over files in the directory and its subdirectories
    for root, dirs, files in os.walk(directory):
        for file_name in files:
            file_path = os.path.join(root, file_name)
            
            if file_name.endswith(('.py', '.sh')):
                scan_file(file_path)

def scan_file(filepath: str) -> None:
    """
    Scans a single file for hard-coded passwords or other sensitive data.
    
    Args:
        filepath (str): Full path to the file.
        
    Returns:
        None
    """
    try:
        with open(filepath, 'r') as file:
            content = file.read()
            
            if has_sensitives(content):
                emit_issue(f"Potential issue found in {filepath}")
                
    except FileNotFoundError:
        print(f"Error: File not found - {filepath}.")
        
def has_sensitives(text: str) -> bool:
    """
    Checks the input text for sensitive data.
    
    Args:
        text (str): The content of the file that is being checked.
            
    Returns:
        bool: True if sensitive data is detected, False otherwise.
    """
    # Placeholder implementation. In a real-world scenario, we would use regexes to identify specific patterns
    return "password" in text.lower()
    
def emit_issue(issue_report):
    """
    Logs an issue found during the scanning process.
    
    Args:
        issue_report (str): A human-readable description of the issue.
    """
    print(f"[!] Potential Security Issue Found: {issue_report}")

# Example usage
if __name__ == "__main__":
    scan_files("/path/to/your/project")
```

  - A placeholder implementation uses simple string inspection for passwords within Python and Shell scripts.
