import os
from pathlib import PurePath

class PathTraversalScanner:
    """
    A class to scan for path traversal vulnerabilities in a directory structure.
    
    Methods:
        scan_directory: Scans a given directory for potentially unsafe paths.
    """

    def __init__(self, root_dir):
        self.root_dir = root_dir
    
    def escape_path(self, path):
        return PurePath(path).as_posix()

    def safe_relative_paths(self, root_path, relative_paths):
        """Return valid relative paths within the given root directory."""
        
        escaped_root = self.escape_path(root_path)
        
        for rel_path in relative_paths:
            abs_path = os.path.join(escaped_root, self.escape_path(rel_path))
            
            if os.path.isabs(abs_path) or not all(os.path.exists(subdir) for subdir in PurePath(abs_path).parts[:-1]):
                continue

            yield abs_path
        

    def scan_directory(self, directory):
        """Scan the given directory for potentially unsafe paths."""
        
        unsafe_paths = []
        
        try:
            for root, dirs, files in os.walk(directory): 
                safe_dirs = self.safe_relative_paths(root, [os.path.relpath(os.path.normpath(dirpath), start=root) for dirpath in dirs])
                relative_files = [self.escape_path(f"{rel_dir}/{f}") for rel_dir, f in zip(safe_dirs[:-1], files)]
                
                all_paths = safe_dirs + relative_files
                unsafe_paths.extend(p for p in all_paths if self.is_potentially_vulnerable(os.path.dirname(self.escape_path(p))))
            return unsafe_paths
        except Exception as e:  # Added generic exception handling
            print(f"Error during directory scanning: {str(e)}")
        
    def is_potentially_vulnerable(self, path):
        """Determine if the given absolute path might indicate a potential vulnerability."""
        
        # Example simple heuristic, more complex checks would be needed for production.
        return (not os.path.abspath(path).endswith(os.sep) and len(path.split(os.sep)) > 1)


def main():
    scanner = PathTraversalScanner('/usr')
    unsafe_paths = scanner.scan_directory('/')
    
    if unsafe_paths:
        print(f"Unsafe paths found:\n{unsafe_paths}")
    else:
        print("No unsafe paths found.")
    

if __name__ == "__main__":
    main()
