import os
from datetime import datetime

class ExecutiveSummaryGenerator:
    def __init__(self):
        self.data = []
        self.errors = []

    def collect_data(self, data_sources):
        for source in data_sources:
            try:
                result = source.fetch_data()
                if not isinstance(result, dict):
                    raise ValueError(f"Unexpected data type: {type(result)} from source: {source.name}")
                self.data.append({"source": source.name, "data": result})
                print(f"Sourced data successfully for '{source.name}'")
            except Exception as e:
                error_message = f"Error sourcing data from {source.name}: {str(e)}"
                print(error_message)
                self.errors.append(error_message)

    def generate_summary(self):
        if not self.data:
            raise ValueError("No available data to generate summary. Ensure `collect_data` has been called first.")
        
        # Sort and prepare the data for finalizing a report
        self.data.sort(key=lambda x: (datetime.strptime(x['data']['timestamp'], "%Y-%m-%d %H:%M:%S"), x['source']))
        
        # Generate the summary text
        template = """This is an executive summary:
- [Source] timestamp: Data was collected at {timestamp}, and here's a summary of what we have found.
- [Details]
\nHere are some key findings for each source:

{found_data}
-------------------

Thanks for using our system to track the data sources!
"""

        final_output = template.format(
            found_data='\n'.join([f"- [bold {data['source']}] ({datetime.strptime(data['data']['timestamp'], "%Y-%m-%d %H:%M:%S")}) - <{" ".join(list(data.keys()))}>:"'\n\t'+'\t'.join(map(str, data['data'].values()))+'</>' for data in self.data]),
            timestamp=[datetime.strptime(x['data']['timestamp'], "%Y-%m-%d %H:%M:%S") for x in self.data]
        )
        
        return final_output

    def generate_report(self):
        summary = self.generate_summary()
        print("Summary report generated:")
        print(summary)
