#!/usr/bin/env python3

import pandas as pd
from datetime import datetime
import os.path
import json


class DashboardReportGenerator:
    def __init__(self, data_path):
        self.data_path = data_path
        self.report_data = None

    def load_report_data(self):
        """ Load report data from JSON file. """
        if not os.path.isfile(self.data_path):
            raise FileNotFoundError(f"File {self.data_path} does not exist")
        
        with open(self.data_path, 'r') as file:
            data = json.load(file)
            
        self.report_data = pd.DataFrame(data)

    def generate_report(self):
        """ Generate an HTML dashboard report from the loaded dataset."""
        if self.report_data is None or self.report_data.empty:
            raise ValueError("Data not available. Please load data first.")

        # Simple example of how to filter by date range
        start_date = datetime(2023, 1, 1)
        end_date = datetime.now()

        filtered_report_data = self.report_data[(self.report_data["date"] >= start_date) & (self.report_data["date"] <= end_date)]

        # Example of creating a simple HTML table
        report_html_table = "<table><tr><th>Timestamp</th><th>Value</th></tr>"
        for _, row in filtered_report_data.iterrows():
            ts, value = row["timestamp"], row["value"]
            formatted_ts = datetime.strftime(ts, "%Y-%m-%d %H:%M:%S")
            report_html_table += f"<tr><td>{formatted_ts}</td><td>{value}</td></tr>"
        report_html_table += "</table>"

        # Example body of the HTML dashboard
        html_dashboard_body = (
            "<html>\n"
            "  <head>\n"
            "    <title>Dashboard Report</title>\n"
            '    <script src="https://cdn.plot.ly/plotly-latest.min.js" type="text/javascript"></script>\n"
            "  </head>\n"
            "  <body>\n"
            f'    <h1>{{report_title}}</h1>'
            f"{report_html_table}"
            "  </body>\n"
            "</html>"
        )
        
        report = {
            'title': 'Report for Period: {} to {}'.format(start_date, end_date),
            'dashboard': html_dashboard_body
        }
        
        return json.dumps(report)

    def save_report(self, file_path):
        """ Save generated HTML dashboard report to a specified file."""
        with open(file_path, 'w') as file:
            report_text = self.generate_report()
            file.write(report_text)
        print(f"Report successfully saved to {file_path}")


def example_usage(data_file='data.json', report_file='output.html'):
    """ Example usage of the DashboardReportGenerator to generate a dashboard report."""
    drg = DashboardReportGenerator(data_file)

    # Assuming data is available and correctly formatted in JSON from your source
    # Here we just load it once; you might have a different method for loading data.
    drg.load_report_data()

    # Output the HTML file path instead of displaying it directly as example usage output is typically not displayed to users.
    print(f"HTML report generated and saved at {report_file}")
