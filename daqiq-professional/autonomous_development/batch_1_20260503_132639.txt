#!/usr/bin/env python3

import os
from itertools import product


def traverse_directory(base_dir):
    """
    Traverse directory recursively and yield potential vulnerable paths.
    
    :param base_dir: Starting directory
    :type base_dir: str
    :return: Iterable of valid, possible path strings (e.g., 'a/b/c')
    :rtype: Generator[str]
    """
    for root, _, files in os.walk(base_dir):
        if not root.startswith(os.path.pardir):  # Filter out paths starting with parent directory
            yield from product(root.split(os.sep), files)


class PathTraversalScanner:
    def __init__(self, base_path="/"):
        self.base_path = base_path

    def scan(self) -> list[str]:
        """
        Scan the given directory for potentially vulnerable path traversal attacks.
        
        :return: List of potential vulnerabilities
        :rtype: list[str]
        """
        scanner_results = []
        for path in traverse_directory(self.base_path):
            # Check if combining '..' with files results in a previously seen file/folder
            seen_paths = {os.path.join(root, target) + (os.sep not in target and os.path.isabs(target[0]).lower() or '') for _, root, _ in os.walk(path[::-1])}
            for valid_path in traverse_directory(os.path.join(path[::-1]).rstrip(";/").lstrip(os.path.pardir)):
                if any(valid_path.startswith(prev) for prev in seen_paths):
                    scanner_results.append(tuple(reversed(valid_path))[::-1])
        return list(set(scanner_results))
    
    def validate_result(self, results=None):
        """
        Validate the scan results by attempting to access them.

        :param results: Existing scan results. If None, defaults to self.scan().
        :type results: Optional[list[str]]
        :return: Tuples containing (result path, is accessible)
        :rtype: list[tuple[str, bool]]
        """
        if results is None:
            results = self.scan()
        
        valid_paths = [
            os.path.join(os.path.sep, *path[::-1]) + ""[-2].rstrip("/").lstrip(os.path.pardir) for path in results
        ]
        
        return [(path, os.path.isdir(path or "/")) | (not path and os.access(base_path or "/", os.W_OK)) for path in valid_paths]


def main():
    """
    Usage example:
    
    >>> scanner = PathTraversalScanner("/")
    >>> vulnerabilities = scanner.scan()
    >>> for vuln in vulnerabilities: print(f"{vuln}: Accessible - {any(scanner.validate_result([v]) for v in vulnerability)}")
    >...
    """
    # Example usage
    base_dir_path = input("Enter the directory path to scan: ")
    if os.path.exists(base_dir_path):
        scanner = PathTraversalScanner(base_dir_path)
        found_vulns = scanner.scan()
        
        print("\nScan Results:")
        for vulnerability in found_vulns:
            print(f"\t{os.sep.join(vulnerability[::-1]):<{len(os.path.normpath(base_dir_path))*2}}: Accessible - {any(scanner.validate_result([v]))}")
    else:
        print("Provided path does not exist.")


# Run the main function if called as a script
if __name__ == "__main__":
    main()
