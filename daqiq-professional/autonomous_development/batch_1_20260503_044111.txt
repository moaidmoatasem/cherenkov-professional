import logging

def generate_alert_summary(data, template):
    """
    Generates an alert summary from the given data and a custom template.
    
    :param data: dict - A dictionary of structured information relevant for the executive summary.
    :param template: str - The path to the file containing the template for the summary, with placeholders denoted by '%SPECIAL%'.
    :return: str - The generated alert summary.
    """
    
    # Setup basic logging
    logging.basicConfig(filename='executive_summary.log', level=logging.INFO, format='%(asctime)s:%(levelname)s:%(message)s')

    log_str = "Generating Executive Summary with the following data:\n"
    for key in data:
        log_str += f"{key}: {data[key]}\n"
    
    log_str += "\nUsing template: \n" + template

    # Open and format the template file
    try:
        with open(template, 'r') as tmpl_file:
            content = tmpl_file.read()
 
            for key in data.keys():
                content = content.replace(f'%{key}%', str(data.get(key)).strip())
        
            return content

    except FileNotFoundError:
        logging.warning("Template not found.")
        raise
    except Exception as err:
        logging.error(f"An error occurred: {str(err)}")
        raise RuntimeError('Unknown Error')

def main():
    data = {
        "incident_timestamp": "2023-10-05 09:47:33",
        "affected_systems": ["Network A", "System B"],
        "vulnerabilities_found": True,
        "risk_assessment_score": 75,
        "recommendations": [
            {"name": "Implement additional encryption", "priority_level": 1},
            {"name": "Upgrade software to latest version", "priority_level": 2}
        ]
    }

    # Path to your custom template file
    with open('template.txt', 'r') as f:
        content = f.read()
    
    print(generate_alert_summary(data, content))

if __name__ == "__main__":
    main()
