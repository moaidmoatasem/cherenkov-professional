```python
import os
from slack_sdk import WebClient  # For Slack Integration
from discord.ext import commands, ipc  # For Discord Integration

# Configuration
BOT_TOKEN = "YOUR_DISCORD_BOT_ID_HERE"
SLACK_ACCESS_TOKEN = "YOUR_SLACK_API_TOKEN_HERE"

class SlackIntegration:
    def __init__(self, client: WebClient):
        self.client = client

    def echo(self, message: str) -> None:
        """Echo the input message."""
        if not message.strip():
            print("Nothing to echo.")
            return
        response = {
            "response_type": "ephemeral",
            "text": message,
            "replace_original": True
        }
        self.client.chat_postEphemeral(
            channel=self.client.token_scopes.split(" ")[0],
            user=self.client.user_id,
            **response,
        )

    def send_message(self, text: str) -> None:
        """Send a custom Slack message."""
        response = {
            "text": text
        }
        self.client.chat_postMessage(
            channel=os.getenv('SLACK_CHANNEL'),
            **response
        )


class DiscordIntegration(commands.Cog):
    def __init__(self, bot):
        self.bot = bot

    @commands.command(name="echo")
    async def echo_command(self, ctx: commands.Context, *, message: str) -> None:
        """Echo the input message from a Discord Command."""
        if not message.strip():
            await ctx.send("Nothing to echo.")
            return
        response = {
            "response_type": "ephemeral",
            "text": message,
            "replace_original": True
        }
        await self.bot.send_ipc_request(self.bot.ipc_server.app_id, ip="slack", command=self.echo_command.id, body=response)

    @commands.command(name="message")
    async def send_message_command(self, ctx: commands.Context) -> None:
        """Send a custom message in Discord channels from a command."""
        response = {
            "text": f"Echoing your message as {message}",
        }
        await self.bot.send_ipc_request(self.bot.ipc_server.app_id, ip="slack", command=self.send_message.id, body=response)

    @discord.ext ipc.empty_constructor
    async def send_message(self, ctx: commands.Context, *, text: str) -> None:
        """This function is called from the client's webhook."""
        response = {
            "text": f"Echoing your message as {text}",
        }
        await self.bot.http.patch(f"/ipc/routes/events/{self.bot.ipc_server.event_id}", body=response)


def setup(bot):
    bot.add_cog(DiscordIntegration(bot))
    slack_integration = SlackIntegration(client=WebClient(token=SLACK_ACCESS_TOKEN))
    bot.add_cog(slack_integration)
```

This code sets up both a `SlackIntegration` and a DiscordCog `commands.Cog`. The `setup` function in the bot module ensures these components are registered properly. Error handling is implemented with simple checks to ensure inputs such as messages are not empty before sending requests.

The bot's IPC methods are defined, creating proper request/response structures using JSON dictionaries to send custom Slack/Ephemeral echo-messages.
