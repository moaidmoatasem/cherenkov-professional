# .github/workflows/security.yaml
name: Security Scanning

on:
  push:
    branches:
      - main
    tags:
      - v*

jobs:
  security-scanner:

    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v2

      - name: Install Dependencies
        run: |
          python3 -m pip install --upgrade pip
          python3 -m pip install 'flake8==4.*'
          python3 -m pip install pytest black flake8-pyformat pylint
          python3 -m pip install yamllint

      - name: Run Flakes
        if: ${{ matrix.flake.version == 'flake' }}
        run: |
          # Use specific version of flake8 for consistency
          FLAKE_VERSION='4.0.1'
          python3 -m flake8 --max-complexity=9 --select=E,W999,F600,F82,F83,C,R,H --config .flake8 "$INPUT_REPO_PATH"

      - name: Run Black
        if: ${{ matrix.black.version == 'black' }}
        run: |
          python3 -m black . --check --diff

      - name: Run Pylint
        if: ${{ matrix.pylint.version == 'pylint' }}
        run: |
          # Set a lower version of PyLint for testing compatibility
          pylint_version='2.5.2'
          PYTHON_VERSION=$(python3 --version | grep -oP '\d*\.\d+\K.*')
          TEST_PYLINT_VERSION=--test-pylintrc-path=$PYTHON_VERSION/.$pylint_version/.pylintrc
          python3 -m pylint . $TEST_PYLINT_VERSION

      - name: Run YAMLLint
        if: ${{ matrix.yamllint.version == 'yaml' }}
        run: |
          # Use specific version of yamllint for testing dependencies
          YAML_LINT='0.15.0'
          python3 -m yamllint --version=$YAML_LINT

      - name: Test Integration
        if: ${{ matrix.test.version == 'test' }}
        env:
          TEST_REPO_PATH="$INPUT_TEST_PATH"
          TEST_REPOSITORY_URL="$INPUT_SOURCE_REPO_URL"
          TEST_REVISION="$INPUT_SOURCE_COMMIT"
          PYTHONUNBUFFERED=true
        run: |
          # Assume `pytest` will be installed via pip, so we don't install it explicitly
          python3 -m pytest "$TEST_REPO_PATH" --cov=.
        env:
          INPUT_TEST_PATH: .github/workflows/README.md
          INPUT_SOURCE_REPO_URL: https://github.com/example/repo.git
          INPUT_SOURCE_COMMIT: "main"

      - name: Validate Input Parameters
        if: ${{ matrix.validate.version == 'validate' }}
        run: |
          echo "$INPUT_PUBLISH_NAME"
          echo "$INPUT_PUBLISH_VERSION" | grep -q '^[\d\.\-]+'
          # Optional checks for other inputs

  security-test-runner:

    runs-on: ubuntu-latest

    needs: security-scanner

    env:
      INPUT_TEST_REPO_PATH: .github/workflows/README.md
      INPUT_SOURCE_REPO_URL: https://github.com/example/repo.git
      INPUT_SOURCE_COMMIT: "main"
      INPUT_PUBLISH_NAME: example-package
      INPUT_PUBLISH_VERSION: 1.2.3-internal

    steps:
      - name: Run Test Jobs in Parallel
        id: run-tests
        uses: actions/parallel-runner@master
        with:
          - command: python -m pytest --cov . "$INPUT_TEST_REPO_PATH"
            env: TEST_REPO_PATH="$INPUT_TEST_REPO_PATH" PYTHONUNBUFFERED=true
          - command: bash <(curl -fsSL https://raw.githubusercontent.com/example/package/$INPUT_PUBLISH_VERSION/scripts/validate.sh)
            env: INPUT_PUBLISH_NAME="$INPUT_PUBLISH_NAME" INPUT_PUBLISH_VERSION="$INPUT_PUBLISH_VERSION"
      - name: Fetch Coverage HTML Report
        if: steps.run-tests.outputs.failures == "0"

        uses: actions/cache@v2
        id: save-coverage-reports
        with:
          path: coverage.zip
          key: ${{ runner.os }}-coverage-${GITHUB_SHA}
          restore-keys: |
            ${{ runner.os }}-coverage-

      - name: Create Artifact from ZIP File
        uses: actions/zip@v2
        if: steps.run-tests.outputs.failures == "0"
        with:
          output: coverage.zip

      - name: Upload Coverage Artifacts to GitHub Actions
        run: |
          git config --global user.name "GitHub Action"
          git config --global user.email "<EMAIL>"
          # Ensure coverage zip is not large, especially for artifacts on actions-ci
          # Note: If the size limit increases or you encounter size issues, consider compressing in parallel with a tool like gzip.
          -s ./$INPUT_TEST_REPO_PATH/coverage.html.zip

      - name: Checkout Coveralls Cache
        uses: esbenmadsen/checkout-coveralls-caches@1.3.0
        if: steps.run-tests.outputs.failures != "0"
        with:
          repo_name: example/repo.git
          coveralls_cache_key: SHA_HASH_OF_COVERAGE_REPORTS

      - name: Upload Coveralls Artifacts to GitHub Actions
        id: upload-coveralls-artifacts-to-github-actions
        uses: coverallsapp/github-action@v3.0
        if: ${{ steps.run-tests.outputs.failures != '0' }}
        env:
          COVERALLS_KEY: MY_COVERALLS_API_ACCESS_TOKEN
      - uses: actions/upload-artifact@v2
        with:
          name: "example/repo.git/coverage-reports"
          path: coveralls-artifacts/*.zip

# Custom script to generate .pylintrc for testing compatible versions of PyLint
