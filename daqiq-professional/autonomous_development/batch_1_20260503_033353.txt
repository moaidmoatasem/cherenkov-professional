#!/usr/bin/env python3

import requests
from urllib.parse import urlparse
import sys


def xxe_scanner(target_url):
    """
    Function to scan for XXE (XML External Entity) vulnerabilities in XML documents.
    
    Parameters:
        target_url (str): The URL of the web service to test
    
    Returns:
        vulnerability_found: True if a potential XXE vulnerability is detected, False otherwise
    """
    # Validating input URL format
    try:
        result = urlparse(target_url)
        if not all([result.scheme, result.netloc]):
            raise ValueError("Invalid URL format.")
    except ValueError as e:
        return {"error": str(e)}

    if '://' in target_url:  # For HTTPS usage later we can replace http[s]:// with https://
        target_url = f"https://{target_url.split('https://')[1].split('/')[2]}"

    payload = "<xml><!--[<&]>-->"
    headers = {
        "User-Agent": "XML-Scanner/1.0",
        "Content-Type": "application/xml"
    }

    try:
        response = requests.post(f"{target_url}/test", data=payload, headers=headers, verify=False)
        
        # Detect if XXE was triggered
        vulnerabilities_text = "".join([char for char in response.text if char.isalpha() or char.isspace()])
        is_vulnerable = "<&>" in payload and "<&>>" not in vulnerabilities_text

        return is_vulnerable
    except requests.RequestException as e:
        # Log the error, request exception could be network issues or other server side stuff.
        return {"error": str(e)}

def main():
    """
    Main function to execute XXE Scanner for specified targets.
    
    Usage example:
    ./xxe_scanner.py https://example.com/service
    """
    if len(sys.argv) != 2:
        print("Usage: xxel_scanner <URL>")
        sys.exit(1)
    target_url = sys.argv[1]
    result = xxe_scanner(target_url)

    # Printing results.
    if isinstance(result, dict):
         print(f"Error: {result['error']}")
    elif result:
        print("XXE Vulnerability detected!")
    else:
        print("No XXE vulnerabilities found.")

if __name__ == "__main__":
    main()

# Example Usage
if __name__ == "__main__":
    # Let's manually test.
    manual_target = "https://example.com/test_service"
    if xxe_scanner(manual_target):
        print(f"Target: {manual_target} is vulnerable to XXE.")
    else:
        if xxe_scanner(manual_target + "/"):
            print(f"Target with deeper path access: {manual_target}/ is still vulnerable to XXE.")
        else:
            print("Target with deep access or not, appears safe against XXE attacks.")
