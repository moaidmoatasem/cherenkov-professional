# create_pdf_report_generator.py

import datetime
from fpdf import FPDF
from typing import List


class ReportGenerator(FPDF):
    """
    A class to generate PDF reports. It extends the functionality of FPDF to include report generation.
    
    Attributes:
        title (str): The title or name of the Report.
        pdf_output_path (str): Path where the PDF file will be saved
        data (List[List[str]]): List of lists with values (each inner list containing strings)
            representing a row, and columns in that item are the header names provided when initializing object
    """

    def __init__(self, title: str):
        super().__init__()
        self.title = title.upper()
        self.pdf_output_path = ""
        self.data = []
        self.headers = []

    def set_pdf_output_path(self, pdf_output_path: str) -> None:
        """
        Method to store the output file name in a variable for future reference.
        
        Args:
            pdf_output_path (str): The path where the generated PDF will be saved
        """
        self.pdf_output_path = pdf_output_path

    def set_data(self, data: List[List[str]], headers: str) -> None:
        """
        Method to load values in a list of lists and headers. 
        If no headers are needed then pass an empty string as argument.
        
        Args:
            data (List[List[str]]): A list that contains the report row. Each inner list will be row on PDF document
            headers (str): Header name for each column, separated by commas in a string "h1,h2,h3,..."
                        If no headers are required, pass an empty value.
        """
        
        if not data:
            raise ValueError("Data can't be empty!")
        
        # Convert headers to list based on provided header input
        if headers != "":
            self.headers = [header.strip() for header in headers.split(',')]
        else:
            self.headers = None

        # Prepare headers and rows
        self.data.extend(data)
    
    def generate(self) -> bool:
        """
        Method will initiate the report generation process.
        
        Raises:
            FileNotFoundError: If 'pdf_output_path' is not set before generating report.
            
        Returns:
            Boolean: Return True if successful, else False
        """
        try:
            if not self.pdf_output_path:
                raise FileNotFoundError("'pdf_output_path' must be set before using this method")

            # Check for invalid data and headers formats   
            if self.headers is None and not self.data:
                return False
            
            # Header name needed in all rows; make sure every row matches
            if self.headers and len(set(len(row) for row in self.data)) > 1:
                return False

            self.add_page()
            self.set_font('Times', 'B', 12)
            self.page_margin = 10
            
            # Write headers only if there are any 
            if self.headers:
                width, _ = self.get_text_dimensions(self.title + "\n\n")
                self.set_x(width * -1)      # Move cursor to the right of text
                self.text(0, -(width//3), "REPORT:" + self.title)
                for i, h in enumerate(self.headers):
                    if not self.is_last(i):  # For all but last column header
                        self.cell(-self.page_break_code())
                    elif not self.has_next_page():  
                        self.set_y(self.get_height() - self.text2path('...').h) 
                    else:
                        self.add_page()
                    self.multi_cell(height=self.get_font_h(),
                                    txt=h)
                if not self.is_last(len(self.headers)): 
                     self.jump_to()

            # Loop through data, write all rows
            for idx, row in enumerate(self.data):
                height = self.font_h() * len(row) + (self.line_height // 2) * (len(row) -1)
                width, _ = self.get_text_dimensions(", ".join(row))
                self.set_x(width * -1) 
                
                x0, y0 = -(width//3), -(height//1.5)
                if not self.is_last(idx):
                    y1 = height + (len(self.data)//2) * height
                else:
                    # Only add footer if there's a 2nd page
                    x1, y1, w1, h1 = (
                        width, 
                        -(height//4), 
                        width//3, 
                        height)
                self.multi_cell(height=height,
                                txt=", ".join(row),
                                border=1)  
                
            # Write the final footer information like date
            self.set_font('Helvetica', '', 8)
            x0, y0 = (-width * .6), -(height//2.75)
            if None == self.headers:
                h = "Generated on %s" % datetime.datetime.now().strftime("%Y-%m-%d")
            else:  
                h = ", ".join(self.headers) + ": Generated on %s" % datetime.datetime.now().strftime("%Y-%m-%d")

            # Footer justification
            if not self.is_last(-1):
                x0, y0, w0, _= (
                    -(width/2), 
                    0, 
                    width * .6, 
                    height)
                
            elif None == self.headers:
                x1 = x0 + (w0 / 5.0) 
                self.text(w1, -h1*2//3, h )
            else:  
                x1, y1 = (-(width/4), -(height * .8))
                self.multi_cell(
                    width=width,
                    _data=[(row for row in [f"{x[0]}: {h}" for x in zip(self.headers, row[:len(height)]) if x]), h],
                    new_y=y1)
            
            # Write final footer content on second page
            self.cell(w1, y1-h1, 
                      str(h), 
                      border=(1, 0, 0, 0),
                      align='R', left=0,
                      max_height=-9.5)

            self.output()
            return True
        
        except Exception as e:
            raise RuntimeError(f'Failed to generate PDF at path {self.pdf_output_path}: {e}')


def main():
    """
    Entry point for usage documentation and example
    """

    # Create a ReportGenerator object, setting the title. 
    report = ReportGenerator("Sample Report")
    
    # Set pdf output directory for further use.
    report.set_pdf_output_path('/path/to/save/report/report.pdf')
    
    # Example data to populate the object: 3D list of tuples
    example_data = [
        ["Client", "Sales Revenue"],
        ["John Doe", "$150,000"],
        ["Jane Smith", "$120,000"]
    ]
    
    # Example header names for each column
    example_header_names = "Client,Sales Revenue"
    
    # Set the data and headers. Generate and check if PDF was generated successfully.
    try:
        report.set_data(example_data, example_header_names)
        status = report.generate()
        
        if not status:  
            print("Failed to generate PDF.")
        else:          
            print(f"Report title: {report.title}. PDF saved at '{report.pdf_output_path}'")
    
    except FileNotFoundError as e:
        print('Could not create pdf due to missing file paths:', str(e))
    
    except Exception as e:
        print(f'An error occurred: {e}')
    

# Execute main method for example usage and testing
if __name__ == '__main__':
    main()
