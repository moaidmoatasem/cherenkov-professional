import os

class PathTraversalScanner:
    """
    A class to scan directories for possible path traversal vulnerabilities.
    
    The `scan` method takes a directory path and scans it for files/folders that lead outside of the specified root directory. It avoids scanning itself, as this could inadvertently expose internal paths.
    
    Example usage:
    >>> scanner = PathTraversalScanner("/path/to/root")
    >>> results = scanner.scan()
    >>> print(results)
    """
    
    def __init__(self, base_directory):
        self.base_directory = os.path.abspath(base_directory)  # Ensure we have a valid absolute root directory
    
    def canonicalize_path(self, path_to_verify):
        """
        Helper method to normalize a given file or folder path to its canonical form within the root.
        
        :param path_to_verify: Path to check
        :return: Normalized and validated root-relative path
        """
        assert os.path.isabs(path_to_verify), "Path should be absolute."
        return os.path.normpath(os.path.abspath(path_to_verify))
    
    def get_directories(self):
        """
        Helper method to recursively list all directories within the specified base_directory.
        
        :return: A set of directory names relative to the root
        """
        directories = {os.path.relpath(d, self.base_directory) for d in os.listdir(self.base_directory)
                       if os.path.isdir(os.path.join(self.base_directory, d)) and not d.startswith('.')}
        return directories
    
    def scan(self):
        """
        Scans paths relative to the base directory for path traversal vulnerabilities.
        
        :return: A set of absolute paths that lead outside their scope
        """
        potential_paths = {os.path.join(self.base_directory, d) for d in self.get_directories()}
        
        # Define a safe or expected set of directories based on whitelisting if necessary
        whitelist = {'safe', 'directory'}.union({self.base_directory.replace('/', '\\')}, {self.base_directory})
        
        unsafe_paths = []
        for path_to_verify in potential_paths:
            try:
                os.lstat(path_to_verify)
                # It's considered safe by default unless proven otherwise.
            except FileNotFoundError:
                continue  # Path is not found, which may indicate an existing file/file system error
            except PermissionError:
                continue  # Directory could be inaccessible due to missing permissions
            except IsADirectoryError:
                continue  # The path points at a directory (which might be safe)
            else:  
                relative_path = self.canonicalize_path(path_to_verify)
                if relative_path not in whitelist:
                    unsafe_paths.append(relative_path)
        
        return set(unsafe_paths)
