import re

class CommandInjectionScanner:
    """
    A class to scan for potential command injection vulnerabilities.
    
    Methods:
    - check_command(command): Checks if the given input is safe from command injection attacks
    
    Raises:
    None. Errors are handled and explained in the response.
    
    Returns:
    bool: True if no command injection found, False otherwise
    """
    def __init__(self):
        # Regular expression to identify potential unsafe characters for commands execution.
        self.UNSAFE_CHARACTERS = re.compile(r'\u\w+|`|%|\$|$')

    def check_command(self, input_value):
        """
        Check if the given input is safe from command injection attacks.

        Args:
            input_value (str): The value to be checked for potential injection attack

        Returns:
            bool: True if no command injection found, False otherwise
        """
        # Use raw string for regex so we can use backslashes in the regular expression without escaping.
        unsafe_characters = self.UNSAFE_CHARACTERS.search(input_value)
        
        return unsafe_characters is None

def scan_input(command):
    """
    Scans a given command and returns whether it contains any potential Command Injection flaws.

    Args:
        command (str): The command to be scanned for security vulnerabilities

    Returns:
        (bool): Whether the input can be considered safe or contains commands which could cause trouble.
    
    Examples:
        >>> scan_input("/usr/bin/ls")
        True
        >>> scan_input("//bin/sh -c 'rm -rf /'")
        False
    """
    scanner = CommandInjectionScanner()
    return scanner.check_command(command)

def example_usage():
    """
    Example usage of the command injection scanner.
    
    This function serves to demonstrate how the `scan_input` method could be used in a real-world scenario.
    """
    # Safe input - no unsafe commands
    safe_command = "/usr/bin/ls"
    print(f"Is '{safe_command}' safe from Command Injection? {scan_input(safe_command)}")

    # Unsafe input - might pose an injection risk
    unsafe_command = "//bin/sh -c 'rm -rf /'"
    print(f"Is '{unsafe_command}' safe from Command Injection? {scan_input(unsafe_command)}")

# Running the example usage function to illustrate the functionality
example_usage()
