from flask import Flask, jsonify, request
import logging
import requests

app = Flask(__name__)
logging.basicConfig(level=logging.INFO)

# Endpoint to handle the scanning process
@app.route('/scan', methods=['POST'])
def scan():
    """
    This endpoint is used for initiating a scanner. 
    It assumes that the body of the incoming POST contains all necessary parameters or headers.
    
    Returns:
        JSON Response with status and details: 
            "status": "SUCCESS",
            "message": "Scan initiated successfully",
            "details": {any relevant details provided by the scanner}
    """
    if not request.is_json:
        return jsonify({"status": "FAIL", "message": "Request body must be in JSON format."}), 400

    # Example of scanning logic, this will depend on the actual implementation
    try:
        data = request.json
        response = requests.post("http://scanner-service/scan", json=data)
        scan_result = response.json() if response.status_code == 200 else {"status": "FAIL", "message": "Failed to initiate scan"}
        return jsonify(scan_result), response.status_code
    except Exception as e:
        logging.error(f"Exception occurred while initiating scan: {str(e)}")
        return jsonify({"status": "FAIL", "message": f"Error occurred during scan initiation: {str(e)}"}), 500

@app.route('/ping', methods=['GET'])
def ping():
    """
    Health check endpoint for monitoring the system's availability.
    
    Returns:
        JSON Response: {"status": "UP"}
    """
    return jsonify({"status": "UP"})

# Main entry point
if __name__ == "__main__":
    app.run(debug=True)
