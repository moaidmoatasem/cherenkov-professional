import requests
from json import dumps

def create_webhook_notification(url: str, payload: dict) -> None:
    """
    Creates and sends a webhook notification to the specified URL with the provided payload.

    Parameters:
        url (str): The target URL where the webhook should be sent.
        payload (dict): A dictionary containing data that should be sent as part of the webhook. 

    Returns:
        None
    """

    # Step 1: Create request headers for the API call
    headers = {
        'Content-Type': 'application/json'
    }

    try:
        # Step 2: Send a POST request to the specified URL with the payload and headers
        response = requests.post(url, data=dumps(payload), headers=headers)

        if response.status_code == 201 or response.status_code == 200:
            print("Webhook notification sent successfully.")
        else:
            # Step 3: Handle errors by logging the status code and error message for debugging
            print(f"Failed to send webhook. Status Code: {response.status_code}, Response: {response.text}")
    except requests.RequestException as e:
        # Step 4: Include general request exception handling in case of network issues or other failures
        print(f"An error occurred during the webhook notification process: {e}")
