import os
from urllib.parse import unquote

def path_traversal_scan(url):
    """
    Scans for directory traversal vulnerabilities in HTML and URL paths.
    
    Args:
        url (str): The input URL or HTML content to scan.
        
    Returns:
        list: A list of potentially vulnerable base URLs, relative paths,
              absolute paths, or their combinations that could lead to directory
              traversal exploits. Paths are provided as strings normalized to
              start from the root directory and are stripped of any trailing slash.
              
    Raises:
        ValueError: If the input URL is malformed or cannot be parsed successfully.
    """
    
    if not isinstance(url, str) or not url or not url.startswith(('http://', 'https://')):
        raise ValueError("Invalid input. Please provide a valid HTTP/HTTPS URL.")
    
    parts = unquote(url).split('/')
    sanitized_path = f"/{'/'.join(parts[2:])}"
    
    vulnerabilities = []
    
    # Check for directory traversal in the normalized path
    if sanitized_path[-1] == '/':
        start_index = 0
        while len(sanitized_path) - len(str(start_index)) > 4:
            if sanitized_path[:len(str(start_index))] != str(start_index):
                break
            start_index += 1
        vulnerabilities.append(f"{url}/{sanitized_path[start_index:-1]}")
    
    # Check for directory traversal in relative paths or base URLs
    parts = unquote(url).split('?')
    if len(parts) > 1:
        href_part = parts[0]
    else:
        href_part = url
    
    path_start_index = href_part.find('//') + 2
    end_position_start = path_start_index
    while part := href_part[end_position_start:].find('/') >= 0:
        if href_part[path_start_index:end_position_start] != '/':
            vulnerabilities.extend(_build_path_traversal(vpart=url, base=href_part[:end_position_start], relative=vpart))
        else:
            vulnerabilities.append(f"{url}/{vpart}")
        
        end_position_start += part + 1
    if path_start_index < (len(href_part) - 1):
        href_part = href_part[path_start_index:]
    
    return vulnerabilities


def _build_path_traversal(url, base, relative):
    """Helper function to generate path traversal examples."""
    return [{
        'url': url,
        'base': base,
        'relative': x
    } for x in [f"{b}/{r}" if not x.startswith('..') else f'../{x}'
                for b in base.split('/') 
                for r in relative.split('/')
               ]
]
