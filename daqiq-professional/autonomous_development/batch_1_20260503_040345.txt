import pandas as pd
from fpdf import FPDF
import os


class PDFReportGenerator:
    """
    A class to generate PDF reports using the FPDF library.
    
    Attributes:
        report_title (str): The title of the PDF report.
        report_author (str): The author's name for the credit in the footer.
        
    Methods:
        _generate_header: Helper method for generating header section of the PDF.
        _generate_body: Helper method for generating body data sections of the PDF.
        generate_pdf(report_data, output_path):
            Method to generate the PDF from the provided report data and save to an output file.
            
    """
    
    def __init__(self, report_title="PDF Report", report_author="Your Name"):
        self.report_title = report_title
        self.report_author = report_author
        
    def _generate_header(self, pdf):
        """
        Generate the header section of the PDF containing title and author information.
        
        Args:
            pdf (FPDF): Instance of FPDF library to add the header information.
            
        Returns:
            None; Adds content directly to the PDF object.
        """
        pdf.add_page()
        pdf.set_font("Arial", size=12)
        pdf.cell(0, 10, f"Report Title: {self.report_title}", ln=True)
        # Author name placeholder
        #pdf.cell(0, 10, f"Author Name: {self.report_author}")
        
    def _generate_body(self, data_frame, pdf):
        """
        Generate the body (data content) section of the PDF.
        
        Args:
            data_frame (pd.DataFrame): DataFrame containing the report's data to be printed.
            pdf (FPDF): Instance of FPDF library to add data content sections to.
            
        Returns:
            None; Adds data content directly to the PDF object for each row in the DataFrame.
        """
        for index, row in data_frame.iterrows():
            pdf.cell(0, 10, f"Row Data: {row}")
        
    def generate_pdf(self, report_data, output_path):
        """
        Generate and save the PDF from the provided report data to an output file path.
        
        Args:
            report_data (List or pd.DataFrame): The data to be included in the PDF, either as a list of dictionaries or DataFrame.
            output_path (str): Path where the generated PDF should be saved.
            
        Returns:
            None; Saves the PDF object to the specified output path.
        """
        if not isinstance(report_data, (list, pd.DataFrame)):
            raise ValueError("Input data must be a list or pandas DataFrame.")
        
        pdf = FPDF()
        self._generate_header(pdf)
        self._generate_body(pd.DataFrame(report_data), pdf)
        
        # Save PDF
        os.makedirs(os.path.dirname(output_path), exist_ok=True)
        with open(output_path, "wb") as f:
            pdf.output(f)
            
        print(f"Report generated and saved to {output_path}")

# Example Usage

if __name__ == "__main__":
    data = [
