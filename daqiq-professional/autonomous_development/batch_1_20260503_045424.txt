import requests

class WebhookNotifier:
    """
    A class to send webhooks for notification purposes.
    
    Methods:
        send_notification: Sends a POST request to a specified URL with provided data.
        
    Attributes:
        receiver_url (str): The webhook endpoint to which the payload will be sent.
        error_log_file (str): File where errors should be logged.

    """
    
    def __init__(self, url, error_log):
        self.receiver_url = url
        self.error_log_file = error_log
    
    def send_notification(self, data: dict) -> None:
        """
        Sends a POST request to the provided URL with the specified dict as payload.
        
        Parameters:
            data (dict): The JSON payload data to be sent. Should include any parameters necessary for notification.
            
        Returns:
            None
        
        Raises:
            requests.RequestException if unable to send request successfully.

        """
        headers = {'Content-Type': 'application/json'}
        try:
            response = requests.post(self.receiver_url, json=data, headers=headers)
            response.raise_for_status()
        except requests.RequestException as e:
            with open(self.error_log_file, "a") as error_log:
                error_log.write(f"Failed to send notification {data} due to: {str(e)}\n")
                print(f"Failed to send notification {data} due to: {str(e)}")
