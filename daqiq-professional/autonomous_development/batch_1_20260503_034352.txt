import subprocess

class CommandInjectionScanner:
    """
    A class to scan and detect potential command injection vulnerabilities in input data.
    
    Methods:
        scan_command(command): Scans the provided command for any suspicious or insecure inputs.
    """
    
    def __init__(self):
        pass
    
    def run_command(self, command: str) -> None:
        """
        Executes a shell command with error handling and captures output to stdout and stderr.
        
        Args:
            command (str): The command to execute on the server.
            
        Raises:
            subprocess.CalledProcessError: If the command execution fails or returns non-zero exit status.
            ValueError: If the provided input is not a string type.
        """
        if isinstance(command, str):
            try:
                result = subprocess.run(
                    [command], 
                    check=True,
                    stdout=subprocess.PIPE,
                    stderr=subprocess.STDOUT
                )
                return result.stdout.decode('utf-8')
            except ValueError as e:
                raise ValueError("Input must be a string type.") from e
            
        raise ValueError("Command to run should be provided as a string.")
    
    def scan_command(self, command: str) -> None:
        """
        Scans the given command for any known command injection patterns. 
        
        Args:
            command (str): The command data that shall be scanned.
            
        Returns:
            bool: True if the command doesn't have any signs of a command injection attempt,
                False otherwise.

        Raises:
            ValueError: If the input `command` is not a string type or None.
        """
        if command is None or isinstance(command, str):
            # This list is assumed to contain known vulnerable commands
            SENSITIVE_COMMANDS = ["echo {{'", "\"", "%", "$*", "*%", "|", "&", ";", "||", "&&", "((", "))"]
            
            for sensitive in SENSITIVE_COMMANDS:
                if sensitive in command:
                    print(f"Potential Command Injection Detected: {sensitive}")
                    return False
            
            # If no sensitive commands are found, the command seems safe
            return True
        
        raise ValueError("Command data to scan should be provided as a string object.")
        
def main() -> None:
    """
    Example usage of the CommandInjectionScanner
    
    Returns:
        None
    """
    scanner = CommandInjectionScanner()
    
    # Safe usage example
    print("Checking a secure command. This will return 'Hello,' not inject anything:")
    output = scanner.run_command("echo Hello, this is safe!")
    print(output.strip())
    
    # Potentially dangerous input example
    risky_input = "echo {{'Bad Password'"
    result = scanner.scan_command(risky_input)
    if (result):
        print(f"The provided data may be vulnerable to Command Injection: {risky_input}")

# Run the main function only for demo purposes, not in production code
if __name__ == "__main__":
    main()
