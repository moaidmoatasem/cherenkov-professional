#!/usr/bin/env python3

"""
Module Purpose:
    Generates an HTML dashboard report from security data and includes error handling.
    Documented with comprehensive docstrings to aid future maintenance and testing.

Usage Example:
    The following example demonstrates how to use the create_html_dashboard function.

    # Assuming security_data is a dictionary containing security scans or tests results
    import generate_dashboard
    generate_dashboard.create_html_dashboard(security_data)
"""

import json
from jinja2 import Environment, FileSystemLoader

class HTMLDashboardGenerator:
    def __init__(self):
        self.env = Environment(loader=FileSystemLoader('templates'))

    def render_template(self, template, data=None):
        """
        Render a Jinja2 template with the provided data.

        :param template: Template path or name in templates directory.
        :param data: Data to pass to the template for rendering.
        :return: rendered HTML content as str
        """
        if data:
            template = self.env.get_template(template)
            return template.render(data)
        else:
            return self.env.get_template(template).render()

    def create_html_dashboard(self, security_data):
        """
        Generates an HTML dashboard using the provided security data.

        :param security_data: A dictionary containing security scan results or related data.
        :return: Full rendered HTML content as str
        """
        # Define Jinja2 template and process any errors in rendering the template before proceeding
        try:
            rendered_content = self.render_template('dashboard.html', {'security_data': json.dumps(security_data, indent=4)})
        except Exception as e:
            raise ValueError(f"Error during template processing: {e}")
        
        return rendered_content

def main():
    # Example data for the security dashboard
    example_security_data = {
        "vulnerabilities": [
            {"id": 1, "name": "SQL Injection", "details": "Potential SQL injection attack found.", "severity": "High"},
            {"id": 2, "name": "XSS Attack", "details": "<strong>Click</strong> or <a href='http://untrusted.com'>Open</a>", "severity": "Medium"}
        ],
        # Add more example data here to simulate a real security scan
    }

    dashboard_generator = HTMLDashboardGenerator()
    html_content = dashboard_generator.create_html_dashboard(example_security_data)
    
    with open('dashboard.html', 'w') as file:
        file.write(html_content)

if __name__ == "__main__":
    main()
