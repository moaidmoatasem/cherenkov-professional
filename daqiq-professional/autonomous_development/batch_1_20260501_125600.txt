import requests

def perform_request(url, max_retries=3, delay_seconds=2):
    """
    Performs an HTTP request and retries on failure.
    
    Parameters:
        url (str): The URL to send the request to.
        max_retries (int): Maximum number of times to retry after a failed attempt. Default is 3.
        delay_seconds (int): Number of seconds to wait before each retry. Default is 2, between retries, starting with first one.

    Returns:
        dict: The JSON response from the server.
        
    Raises:
        Exception: Any exception that occurs during the request or any error that happens while trying to handle retries.
    """
    for attempt in range(max_retries + 1):
        try:
            response = requests.get(url, timeout=60)  
            if response.status_code == 200:
                return response.json()
        except requests.exceptions.RequestException as e:  
            if attempt < max_retries:
                print(f"Attempt {attempt + 1} failed! Retrying ({delay_seconds}s later). Error: {e}")
                # Sleep before retrying
                import time
                time.sleep(delay_seconds)
            else:
                raise Exception(f"All attempts to access URL '{url}' have failed, last error was: {str(e)}")
        except Exception as e:  
            if attempt < max_retries:
                print(f"An unexpected error occurred ({attempt + 1} of {max_retries}): Error - {e}")
                # Sleep before retrying
                time.sleep(delay_seconds)
            else:
                raise
