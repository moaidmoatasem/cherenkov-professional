import requests

class BaseApi:
    def __init__(self):
        self.base_url = "https://api.example.com"
    
    def _make_request(self, method, endpoint, data=None):
        """
        A helper method to make HTTP requests.
        
        Parameters:
            - method (str): The HTTP request method. Ex: 'GET', 'POST'
            - endpoint (str): The API endpoint path
            - data (dict or None): Data dictionary to send in the body of the request
        
        Returns:
            dict: A JSON response from the server.
        
        Raises:
            Exception: If an error occurs while making the request.
        """
        url = f"{self.base_url}/{endpoint}"
        try:
            if method == 'GET':
                return requests.get(url)
            elif method in ['POST', 'PUT']:
                if data is not None:
                    response = requests.request(method, url, json=data)
                else:
                    response = requests.request(method, url)
            else:
                raise Exception(f"Unsupported HTTP method: {method}")
        except Exception as e:
            raise Exception(f"An error occurred while making request to the server. Error: {e}")

class ScannerAccess(BaseApi):

    def __init__(self):
        super().__init__()

    def create_user(self, user_id, username, email):
        """
        Create a new user resource

        Parameters:
            - user_id (str): ID of the user
            - username (str): Username for logging in
            - email (str): Email associated with this account to verify it.

        Returns:
            dict: JSON response from the server

        Raises:
            Exception: On server error
        """
        data = {
            'user_id': user_id,
            'username': username,
            'email': email,
        }
        return self._make_request('POST', 'users', data=data)

    def update_user(self, user_id, **kwargs):
        """
        Update an existing user resource
        
        Parameters:
            - user_id (str): The ID of the user to be updated
            - kwargs: Keyword arguments mapping to JSON properties

        Returns:
            dict: JSON response from the server

        Raises:
            Exception: On server error
        """
        update_data = {k: v for k, v in kwargs.items() if v is not None}
        return self._make_request('PUT', f'users/{user_id}', data=update_data)

    def list_users(self):
        """
        List all or matching users

        Returns:
            dict: JSON response from the server
            
        Raises:
            Exception: On server error
        """
        return self._make_request('GET', 'users')

def main():
    # Example Usage
    api = ScannerAccess()
    
    # Create a new user
    print("Creating a new user:", api.create_user(user_id="user123", username="TestUser1", email="test@example.com"))
    
    # Update an existing user's details
    params = {'user_id': 'user123', 'username': 'NewUsername', 'email': 'new_email@example.com'}
    print("Updating a user:", api.update_user(**params))
    
    # List all users
    for user in api.list_users():
        print(f"User: {user}")
