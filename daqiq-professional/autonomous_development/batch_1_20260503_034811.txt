import os
import mimetypes
from urllib.parse import urljoin

class FileUploadVulnerabilityScanner:
    """
    A class to scan file upload endpoints for vulnerabilities.
    
    This scanner checks if a given URL has unvalidated file input field,
    allowing attackers to upload arbitrary files to the server. It does this 
    by trying to send various file types and checking if it gets through.
    If any type is allowed without validation, it's considered vulnerable.

    Attributes:
        base_url: The base URL where vulnerability checker will perform scans.
    """
    
    def __init__(self, base_url):

            bool: True if type of the content allows uploads, False otherwise.  
        """
