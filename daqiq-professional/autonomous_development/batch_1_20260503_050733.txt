```python
import requests

class FeatureScanner:
    """
    This class implements a scanner feature for the CI/CD.
    
    Methods:
    - scan(): Executes the feature scan.
    """

    def __init__(self, project_id: int):
        self.base_url = 'https://gitlab.com/api/v4/projects/{}/'.format(project_id)
        
    def _make_request(self, endpoint: str) -> requests.Response:
        """
        Makes a GET request to the API at the specified endpoint.

        Args:
            - endpoint (str): The endpoint URL within the GitLab project.
            
        Returns:
            Response object returned from the requests.get call.
        """    
        url = self.base_url + endpoint
        headers = {
            'PRIVATE-TOKEN': 'your_access_token'
        }
        
        response = requests.get(url, headers=headers)
        response.raise_for_status()  # Raise an exception for non-2xx responses
        return response

    def scan(self):
        """
        Executes the feature scan defined by the endpoint.
        
        Returns:
            dict: The parsed JSON response.
        """        
        endpoint = 'pipelines/your_pipeline_id/features'
        pipeline_response = self._make_request(endpoint)
        
        if not pipeline_response.ok:
            raise ValueError('Feature scanner failed to retrieve information.')
            
        return pipeline_response.json()

def usage_example():
    """
    Example usage of the FeatureScanner class.
    
    Returns:
        dict: The result from scanning a feature on GitLab project ID 123456789
    """    
    scanner = FeatureScanner(123456789)
    try:
        features_response = scanner.scan()
        print(features_response)
    except ValueError as ve:
        print("Error occurred:", ve)

if __name__ == "__main__":
    usage_example()
```
