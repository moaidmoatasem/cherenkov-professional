import json
import requests

def send_message_to_slack(slack_webhook_url, message):
    """
    Sends a message to a Slack webhook.
    
    Parameters:
        slack_webhook_url (str): The URL of the Slack WebHook.
        message (str): The text message to be sent to Slack.

    Returns:
        str: A representation of the response JSON or 'Operation Successful.'
        
    Raises:
        requests.exceptions.RequestException: If there is an issue sending the message.  
    """
    
    headers = {'Content-Type': 'application/json'}
    try:
        response = requests.post(
            slack_webhook_url,
            data=json.dumps(message),
            headers=headers
        )
        if response.status_code in [200, 201]:
            return "Message sent successfully."
        else:
            return f"Received HTTP status code {response.status_code}. Please check payload and retry. "
    except requests.exceptions.RequestException as e: # Handling request-related_exceptions
        return f"Request failed: {str(e)}."
