import os
from urllib.parse import urlparse

class FileUploadVulnerabilityScanner:
    """
    This class scans files uploaded via a web application and checks for potential security vulnerabilities like directory traversal attacks and executable file uploads.
    
    Methods:
        - __init__(self, allowed_directories_list: list) -> None: Initializes the scanner with a list of allowed directories from which files may be uploaded.
        - scan_file(self, file_path: str) -> bool: Scans a single file path for security vulnerabilities. Returns True if any issues are found; False otherwise.
    """

    # Define a list of safe paths and directories to prevent directory traversal attacks
    SAFE_PATHS = ['safe_directory']

    def __init__(self, allowed_directories_list: list):
        """
        Configures the scanner with a list of allowed directories.

        Parameters:
            allowed_directories_list (list): A list of absolute file system paths that files may be uploaded from.
                                      Example: ['/path/to/uploaded/files', '/secure/app/uploads']
        """
        allowed_dirs_set = set(allowed_directories_list)
        self.safe_paths.extend(['/' + path[1:].lstrip('/') for dir_path in self.SAFE_PATHS
                                 if '/' == dir_path[-1] and dir_path not in allowed_dirs_set])

    def scan_file(self, file_path: str) -> bool:
        """
        Scans a given file path. Returns True if the file is uploaded from an allowed directory or does not show any signs of security vulnerability;
         otherwise, returns False.

        Parameters:
            file_path (str): The target file's full path to be scanned for security issues.
        
        Returns:
            bool: True if no vulnerabilities are found, False if a vulnerability is detected. 
                  When False, the function reports suspicious activities related to file upload origin and contents.
        """
        # Extract basic info from the URL of the request containing potential malicious content
        url = urlparse(file_path)
        host = os.path.basename(url.netloc)

        sensitive_directories = ['/etc', '/tmp', '/proc', '/dev']

        if any(keyword in file_path for keyword in ['../', '../../', '../../../../../']):  # Directory traversal attempt detected
            return True

        for potential_vulnerability_directory in sensitive_directories:
            if os.path.commonprefix([file_path, os.path.join(potential_vulnerability_directory, '')]) == \
                    potential_vulnerability_directory:
                print(f"Suspicious File Upload: Possible malicious file detection - Path could traverse to critical system directory: {potential_vulnerability_directory}")
                return True

        if any(path in self.safe_paths for path in urlparse(file_path).path):  # Checks if the file originates from an allowed directory
            return False

        filename = os.path.basename(url.path)
        
        executable_extensions = [".exe", ".bat", ".com", ".cmd", ".js"]
        if any(filename.endswith(ext) for ext in executable_extensions):
            print(f"Suspicious File Upload: Executable file detected - {filename} can be executed upon download.")
            return True

        return False
