```python
from datetime import datetime

class CreateExecutiveSummaryGenerator:
    """
    Creates an executive summary for a given set of data.
    
    Attributes:
        output_directory (str): The directory where the generated files will be saved.
    """
    
    def __init__(self, output_directory: str):
        self.output_directory = output_directory
        
    def generate_summary(self, data: list) -> None:
        """
        Generates an executive summary from a provided list of data items.
        
        Parameters:
            data (list): A list where each element is a dictionary with data points.
            
        Raises:
            ValueError: If the input data is not a list or contains dictionaries without 'key' and 'value'.
            IOError: If there are issues writing to the output directory.
        """
        if not isinstance(data, list):
            raise ValueError("Input data must be provided as a list.")
        
        file_path = self._generate_summary_file_path()
        
        try:
            with open(file_path, 'w') as file:
                for item in data:
                    if 'key' not in item or 'value' not in item:
                        continue
                    file.write(f"{item['key']}: {item['value']} \n")
        except IOError as e:
            raise IOError(f"Failed to write summary: {e}")
    
    def _generate_summary_file_path(self) -> str:
        """
        Generates the full path for the summary file.
        
        Returns:
            str: The complete path where the generated file will be stored.
        """
        return f"{self.output_directory}/executive_summary_{datetime.now().strftime('%Y%m%d_%H%M%S')}.txt"
    
    def run(self) -> None:
        """
        Runs a generator that will create an executive summary based on provided data.
        
        Raises:
            ValueError: If the input data is incomplete or of incorrect type while calling generate().
        """
        if not self._validate_data():
            raise ValueError("Data validation failed. Data must contain dictionaries with 'key' and 'value'.")
        
        try:
            self.generate_summary(data=self.data)
        except ValueError as e:
            print(f"[!] Error: {e}")
        except IOError as e:
            print(f"[!] IO error occurred while writing to file: {e}")


# Example usage
def main():
    # Assuming data is a valid list of dictionaries ready for processing by the generator.
    data = [
        {"key": "Sales", "value": "20% growth"},

    ]
```
