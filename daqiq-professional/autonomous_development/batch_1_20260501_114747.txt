import re
from datetime import date

class ExecutiveSummaryGenerator:
    def __init__(self):
        self.errors = []
    
    def generate_summary(self, report, team_members_dict):
        """
        Generates an executive summary from a report and team member information.
        
        Parameters:
            - report (str): The structured content of the security analysis or incident report.
            - team_members_dict (dict): A dictionary where keys are employee names and
                                         values are strings describing their role during the incident.
        
        Returns:
            str: The generated executive summary with placeholders populated accordingly.

        Raises:
            ValueError: If the input report content is an empty string or missing.

        """
        if not report.strip():
            raise ValueError("Report content cannot be an empty string.")
        
        current_date = date.today().strftime("%B %d, %Y")
        team_roles = []
        for employee, role in team_members_dict.items():
            team_roles.append(f"{employee}: {role}")
        
        template = (
            f"EXECUTIVE SUMMARY REPORT\n"
            f"\n"
            f"DATE: {current_date}\n"
            f"\n"
            f"A comprehensive review of security incidents and analysis has been completed.\n"
            f"The report is reviewed from the perspective of several relevant stakeholders. "
            f"The findings are as expected, indicating no critical breaches in controls have occurred. \n"
            f"\n"
            f"Key Incidents:\n\n{'\n'.join(re.sub(r'\s+', ' ', incident) for incident in re.split('\n', report.strip()))}\n\n"
            f"Staff Contribution Details:\n{'\n'.join(team_roles)}"
        )

        return template

    def handle_errors(self):
        if self.errors:
            # Log the errors; replace with actual logging or exception handling logic as needed
            print(f"\033[91mERRORS DETECTED: {self.errors}\033[0m")
