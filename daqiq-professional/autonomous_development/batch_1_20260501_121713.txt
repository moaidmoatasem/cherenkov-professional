import os
from datetime import datetime
from fpdf import FPDF

class PdfGenerator(FPDF):
    """
    A custom class to generate PDF documents. This class extends the functionality 
    of the standard FPDF library by adding specific elements and structure for a report.

    Methods:
        - __init__(self): Initializes the PDF object.
        - add_document_name(self, document_filename: str) -> None: Adds the name of the
            currently generated document to the output directory. Returns nothing.
        - generate_pdf_report(self, content_data: dict) -> bytes: Generates a custom PDF report 
            using provided content data and returning it as a byte array.
    """
