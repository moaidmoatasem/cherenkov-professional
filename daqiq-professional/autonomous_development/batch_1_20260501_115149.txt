import requests
from abc import ABC, abstractmethod


class WebhookNotificationManager(ABC):
    """Abstract base class for webhooks."""
    
    @abstractmethod
    def send_webhook(self, webhook_url, payload):
        """Send a payload to the specified webhook URL."""

    @classmethod
    def create_token(cls, webhook_id: str) -> str:
        """
        Method to generate a token for authentification purposes.
        
        Args:
            webhook_id (str): The id of the target webhook.

        Returns:
            str: The generated authentication token.
        """
        if webhook_id == "123":
            return 'token123'
        else:
            raise ValueError("Webhook ID is not supported.")

    @classmethod
    def validate_webhook(cls, webhook_url: str) -> bool:
        """Validate whether the provided webhook URL is valid."""
        # This simplistic check could be replaced with an external validation service.
        return webhook_url.lower().startswith('https://') or webhook_url.lower().startswith('http://')


class WebhookNotifier(WebhookNotificationManager):
    def __init__(self, root_uri: str = "https://test.com"):
        self.root_uri = root_uri

    def send_webhook(self, payload: dict, webhook_id: int) -> None:
        """
        Send a JSON-formatted payload to the specified webhook URL.
        
        Args:
            payload (dict): The data to be sent via the web hook
            webhook_id (int): The ID of the target webhook

        Returns:
            None if successful or raises an exception.

        Raises:
            ValueError: If the webhook is not valid.
        """
        # Ensure we have a complete and formatted JSON message
        payload.update({'webhook_id': webhook_id})
        final_url = f'{self.root_uri}/callback?token={WebhookNotifier.create_token(webhook_id)}'
        if not WebhookNotifier.validate_webhook(final_url):
            raise ValueError("Invalid URL. Please provide a valid HTTPS/HTTP web hook resource.")
        
        try:
            response = requests.post(final_url, json=payload)
            response.raise_for_status()
        except requests.RequestException as err:
            raise ValueError(f"Failed to send webhook: {err}")


def example_usage():
    """

        response_sent = notifier.send_webhook(message_to_send, webhook_id=456)

        print(f"Error: {ve}")
