```yaml
# .github/workflows/security-scanner.yml
name: Security Scanner

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  scan-code:
    name: Code Scanning Job
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v2
        with:
          fetch-depth: 10

      - name: Setup Python environment
        uses: actions/setup-python@v2
        with:
          python-version: "3.9"

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip wheel
          pip install flake8 mypy pylint pytest coverage pydocstyle

      - name: Run Linting and Static Analysis
        uses: checksuite/gh-action-checksuite@v1.5.0
        with:
          # Add custom configurations for each linter, such as Flake8's `black`
          configs:
            flake8:
              version: latest
              options: --max-line-length=120
           
            mypy:
              options: --no-recursive

            pylint:
              enabled: true
              verbose: true
              options: --reports-nested -E --msg-template="{path}:{line}:{column}: [{symbol}({text})]({url})"
            
            coverage:
              threshold: 95
            
            pydocstyle:
              ignore-case-django-strings: True

      - name: Run Tests
        run: |
          pytest --cov=your_project_name tests/
          coverage report -m

      - uses: peaceiris/actions-rhubarb@v1
        with:
          config-path: .github/workflows/gha.yml
          args: run
```
