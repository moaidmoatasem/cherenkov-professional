import os
from slack_sdk import WebClient
from discord_webhook import DiscordWebhook

class SlackDiscordIntegration:
    def __init__(self):
        self.slack_client = WebClient(token=os.environ['SLACK_TOKEN'])
        self.discord_token_uri = os.environ['DISCORD_WEBHOOK']
    
    def send_slack_message(self, message):
        if not self.slack_client:
            raise Exception("Slack client is not initialized")
        
        try:
            response = self.slack_client.chat_postMessage(
                channel="#general", 
                text=message
            )
        except Exception as e:
            raise Exception(f"Failed to send message to Slack: {e}")
    
    def send_discord_message(self, message):
        if not self.discord_token_uri or not self.slack_client:
            raise Exception("Slack/Discord client and webhook token are not initialized")
        
        try:
            webhook = DiscordWebhook(url=self.discord_token_uri)
            webhook.add_embed(
                title="New Message",
                description=message,
                color=0xFF00FF
            )
            response = webhook.execute()
            return response
        except Exception as e:
            raise Exception(f"Failed to send message to Discord: {e}")

def main():
    """
    Main function to demonstrate usage of the SlackDiscordIntegration class.
    """
    slack_discord = SlackDiscordIntegration()
    
    # Example 1: Send a message to Slack
    print("Sending first example message to Slack:")
    try:
        response = slack_discord.send_slack_message(message="Hello from Python!")
        print(f"Response from Slack API: {response}")
    except Exception as e:
        print(e)
    
    # Example 2: Send a message to Discord
    print("\nSending first example webhook message to Discord:")
    try:
        response = slack_discord.send_discord_message(message="Hello from Python!")
        print(f"Response from Discord: {response}")
    except Exception as e:
        print(e)

if __name__ == "__main__":
    main()
```

---

**Note:** To use this code, you need to set the environment variables:
- `SLACK_TOKEN`: Your Slack token for authentication (`os.environ['SLACK_TOKEN']`).
- `DISCORD_WEBHOOK`: The Discord Webhook URL for your channel.

This example provides a basic implementation with error handling and documentation. Ensure all dependencies are correctly installed.
