#!/usr/bin/env python3

import subprocess
from typing import List

class CommandInjectionScanner:
    """
    A Command Injection Scanner that checks for potential command execution vulnerabilities.

    Attributes:
        commands (List[str]): A list of potentially harmful or unauthorized command strings.
    
    Methods:
        scan_input: Takes an input and attempts to execute it as a shell command, which could lead to Command Injection if not properly sanitized.
        __validate_command_params: Helper function to sanitize the input and validate if it is not part of the given commands.
    """

    def __init__(self):
        self.commands = ["echo", "id", "mkdir /nonexistent"]  # Example dangerous commands

    def scan_input(self, potential_command: str) -> None:
        """
        Attempts to execute a provided input as a shell command, and validates against known dangerous commands.

        Args:
            potential_command (str): The potentially dangerous command string.

        Raises:
            ValueError: If the provided 'potential_command' is in 'commands'.
            subprocess.CalledProcessError: If execution fails for any reason.
        """
        validated_command = self.__validate_command_params(potential_command)
        
        # Using a try-except block to handle potential process failures
        try:
            result = subprocess.run(validated_command, check=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
            print(f"Command executed successfully: {validated_command.decode().strip()}")
        except Exception as e:
            raise ValueError("Invalid command. Command not allowed.") from e

    def __validate_command_params(self, potential_command: str) -> List[str]:
        """
        Validates the provided input and ensures it is not a dangerous command.

        Args:
            potential_command (str): The potentially risky command string to be checked.
        
        Raises:
            ValueError: If the 'potential_command' is in self.commands.
        
        Returns:
            List[str]: A list of sanitized commands if they are valid or raise appropriate error otherwise.
        """
        # Ensure input is not a known vulnerable command
        for cmd in self.commands:
            if potential_command.strip().startswith(cmd):
                raise ValueError(f"Command injection attempt detected: '{potential_command.rstrip()}'")
        
        return [f"{i} {potential_command}" for i in ["bash -c", "/bin/bash -c", "$(/bin/bash -c", "'${%system%'], ["/usr/bin/env bash -c"]]
        

def example_usage():
    """
    Demonstrates how to use the CommandInjectionScanner class.
    """
    
    ci_s = CommandInjectionScanner()
    
    # Safe commands
    print("Safe command examples:")
    for cmd in ["echo Hello", "/usr/bin/true"]:
        try:
            ci_s.scan_input(cmd)
        except Exception as ex:
            print(f"Failed to scan: {cmd} -> Reason: {ex}")

    # Unsafe command examples that would raise a ValueError
    unsafe_commands = ["rm -rf /", "id", "'|'cat;echo 'hacked'"
                        ]
    
    for cmd in unsafe_commands:
        try:
            ci_s.scan_input(cmd)
        except Exception as ex:
            print(f"Scan Failed: {cmd} -> Reason: {ex}")

if __name__ == "__main__":
    example_usage()
