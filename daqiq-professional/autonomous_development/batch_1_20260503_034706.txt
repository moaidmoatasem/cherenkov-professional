```python
import os
import hashlib

def is_risky_file(file_name):
    """
    Heuristic-based function to determine if a file name might be risky.
    
    Args:
        file_name (str): The filename to check
    
    Returns:
        bool: True if the file name might be dangerous, False otherwise.
    
    Notes:
        This heuristic is based on common criteria for risk factors.
    """
    return any(char.isdigit() and len(char) > 1 for char in file_name) or '..' in file_name

def md5_check(file_path):
    """
    Function to get the MD5 hash of a file.
    
    Args:
        file_path (str): The path to the file
    
    Returns:
        str: The hexadecimal MD5 digest of the file content
    """
    try:
        with open(file_path, 'rb') as f:
            data = f.read()
            return hashlib.md5(data).hexdigest()
    except Exception as e:
        # Handle potential exceptions like FileNotFoundError or PermissionError
        print(f"File not found or read error: {e}")
        return None

def file_upload_vulnerability_scan(file_path):
    """
    Scans a given path for potentially risky files based on heuristic checks and file integrity checks.
    
    Args:
        file_path (str): The path to scan
    
    Returns:
        dict: Scan results including detected dangerous files, safe files, and hash inconsistencies
    """
    safe_files = []
    suspicious_files = []
    dangerous_files = []

    if not os.path.isdir(file_path):
        return {
            "safe_files": [],
            "suspicious_files": [],
            "dangerous_files": []
        }

    total_files = len(os.listdir(file_path))
    for file_name in sorted(os.listdir(file_path), reverse=True):
        full_file_path = os.path.abspath(os.path.join(file_path, file_name))

        # Check if user can change the filename
        if os.access(full_file_path, os.W_OK) and is_risky_file(file_name):
            dangerous_files.append(full_file_path)
        
        elif md5_check(full_file_path) != file_name:  # MD5 hash check for unexpected changes
            suspicious_files.append(full_file_path)

        else:
            safe_files.append(full_file_path)
        
    result = {
        "safe_files": safe_files,
        "suspicious_files": suspicious_files, 
        "dangerous_files": dangerous_files
    }
    
    # Counting and adding file counts to the results
    for key in ["safe_files", "suspicious_files", "dangerous_files"]:
        result[key].append(len(result[key]))
    
    return result

def scan_directory():
    """
    Main function to start off scanning from a given directory.
    
    Notes:
        Directory should be set by environment variable or hard coded. Environment variable is preferred for flexibility and security.
    """
    import os
    if 'SCAN_DIR' in os.environ:
        file_path = os.getenv('SCAN_DIR')
    else:
        raise ValueError("Environment variable SCAN_DIR not found.")
    
    return file_upload_vulnerability_scan(file_path)

if __name__ == "__main__":
    scan_results = scan_directory()
    print(scan_results)
```

3. **file_upload_vulnerability_scan** : The main scanner logic where risky files, suspicious files with inconsistent MD5 hashes, and safe files are categorized based on their risk assessment and file modification checks.
