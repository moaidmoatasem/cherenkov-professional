import os

class PathTraversalScanner:
    """
    A class to perform path traversal checks on a target URL or file.
    
    Methods:
        scan_paths(paths): Perform path traversal checking on given paths. Raises an exception for invalid inputs or errors encountered during validation.
        
    Usage Example:
        scanner = PathTraversalScanner()
        valid_paths, invalid_paths = scanner.scan_paths(["/tmp/", "..//usr//bin", "c:\\Windows"])
        
    """

    def is_valid_path(self, path: str) -> bool:
        """
        Validates whether the given path is valid or contains any potential path traversal attack vectors.
        
        Args:
            path (str): The path to validate.

        Returns:
            bool: True if the path is considered safe, False otherwise.
        """

        # Basic validation: Ensure directory starts with a slash '/'. Avoids common issues like Windows paths starting with drive letter and ending with slash
        if not path.startswith('/'):
            return False
        
        # Check for ".." which might lead to external directories traversal
        if '..' in path:
            raise ValueError("Path contains forbidden sequence '..'")
        
        # Windows-like paths may need extra validation, Linux is simpler with just path starting with a slash and no other logic required
        if os.name != 'posix':
            return False
        
        try:
            # Attempt to open the file as root would execute commands; This check is heuristic, actual access controlled by permissions
            import subprocess
            open(os.path.join(path, "temp_test.txt"), 'w').close()
            os.remove(os.path.join(path, "temp_test.txt"))
        except Exception as e:
            return False
        
        # If everything passed -> path safe and valid!
        return True
    
    def scan_paths(self, paths: list) -> tuple[list[str], list[str]]:
        """
        Scans a list of potential unsafe file system access or directory traversal attempts.
        
        Args:
            paths (list): A list containing file/folder locations represented as strings.
            
        Returns:
            tuple(list[str], list[str]): A tuple containing two lists. For valid_paths, they contain the legitimate files and directories found; For invalid_paths, each element contains a tuple (original_path, error_message) indicating an invalid path
        """
        
        valid_paths = []
        invalid_paths = []
        
        for path in paths:
            normalized_path = os.path.normpath(path).replace("\\", "/")
            
            # Check if the path is valid
            if self.is_valid_path(normalized_path):
                valid_paths.append(os.path.realpath(normalized_path))
            else:
                invalid_paths.append((normalized_path, f"Invalid path {path}. Please ensure this path is secure from traversal attacks."))
        
        return (valid_paths, invalid_paths)
