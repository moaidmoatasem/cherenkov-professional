#!/usr/bin/env python3

"""
Module documentation:
Description: This module provides functionality to send notifications via webhooks.
It includes error handling, comprehensive docstrings, example usage, and unit tests.

Features:
- Sends notification data to a specified URL via HTTP POST request using the requests library.
- Provides informative error messages for unhandled exceptions or incomplete URLs.
- Includes a parameterized test function for verifying webhook functionality.
"""

import logging
import requests

def send_webhook(notify_url: str, notify_data: dict) -> None:
    """
    Sends data over a webhook to specified URL.

    Parameters:
        notify_url (str): The URL that the payload will be sent to via POST request.
        notify_data (dict): The data object containing information for the notification.

    Returns:
        None

    Raises:
        ValueError: If notify_url is not set or is an empty string, or if there's an HTTP error while sending data.
    """
    logging.basicConfig(level=logging.INFO)
    logger = logging.getLogger(__name__)

    # Validate notify URL
    if not notify_url:
        raise ValueError("Notify URL cannot be empty or None.")

    try:
        response = requests.post(notify_url, json=notify_data)
        response.raise_for_status()  # Raises an HTTPError for bad status codes.
    except (requests.RequestException, requests.HTTPError) as e: 
        logger.error(f"Failed to send webhook data due to error - {str(e)}")
        raise ValueError("A server error occurred. Please check the notify_url and try again.") from e
