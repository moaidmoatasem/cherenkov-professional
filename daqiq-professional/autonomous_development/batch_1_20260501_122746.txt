from flask import Flask, request, jsonify
from flasgger import Swagger
import logging

app = Flask(__name__)
swagger_config = {
    "headers": [],
    "specs": [
        {
            "endpoint": 'apispec',
            'route': '/apispec.json',
            'title': 'API Spec Example'
        }
    ],
    "static_url_path": "/flasgger/docs",
}
Swagger(app, config=swagger_config)


# Configure logging
logging.basicConfig(filename='scanner.log', level=logging.INFO)
logger = logging.getLogger(__name__)

# Constants and variables can be defined here for clarity

SCANNER_ENDPOINTS = [
    {
        'id': 1,
        'name': 'scanner_name_1',
        'scan_type': 'type_1'
    },
    {
        'id': 2,
        'name': 'scanner_name_3',
        'scan_type': 'type_3'
    }
]

@app.errorhandler(400)
def bad_request_error(error):
    return "Bad Request", 400

@app.route('/api/v1/scanners', methods=['GET'])
def list_scanners():
    """
    Return all scanners available in the system.
    
    Returns:
        A JSON object containing a list of dictionaries each representing a scanner.
    """
    result = {"scanners": SCANNER_ENDPOINTS}
    return jsonify(result)

@app.route('/api/v1/scanners/<scanner_id>', methods=['GET'])
def get_scanner_details(scanner_id):
    """
    Fetch details for a specific scanner by its ID.

    Args:
        scanner_id (int): The identifier of the requested scanner.

    Returns:
        A JSON object containing detailed information about the specified scanner.
    """
    for s in SCANNER_ENDPOINTS:
        if int(s['id']) == int(scanner_id):
            return jsonify({'name': s['name'], 'scan_type': s['scan_type']})
    logger.error(f"Scanner with id {scanner_id} not found.")
    return "Scanner Not Found", 404

@app.route('/api/v1/scanners', methods=['POST'])
def create_scanner():
    """
    Create a new scanner based on the provided details.

    Returns:
        A JSON object containing an ID if successfully created.
    """
    try:
        new_id = SCANNER_ENDPOINTS[-1]['id'] + 1
        sc_details = request.json
        sc_details['id'] = new_id
        SCANNER_ENDPOINTS.append(sc_details)
        return jsonify({'status': 'success', 'id': new_id})
    except Exception as e:
        return f"{str(e)} Could not process scanner creation", 500

if __name__ == '__main__':
    app.run(debug=True)
