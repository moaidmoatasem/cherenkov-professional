class CVEIntegration:
    """
    A class to integrate with the Common Vulnerabilities and Exposures (CVE) database.
    
    Methods:
        get_cve_info(cve_id): Retrieves information about a specific CVE-ID from the CVE database.
        validate_and_fetch(*cves_list): Validates list of CVEs, fetches info if valid, handles invalid IDs gracefully. 
    """

    def __init__(self):
        # Initialize any necessary resources such as connections to external APIs, etc.
        self.api_connection = None

    def get_api_connection(self):
        """
        Initializes the connection to the CVE database API endpoint.

        Returns:
            Request or HTTPConnection: An initialized connection to the CVE database API.
        """
        import requests
        if self.api_connection is None:
            self.api_connection = requests.Session()
        return self.api_connection

    def get_cve_info(self, cve_id):
        """
        Retrieves information about a specific CVE-ID.

        Parameters:
            cve_id (str): The ID of the Vulnerability.

        Returns:
            dict: A dictionary containing information about the given CVE-ID.
                  If CPE parsing fails, returns None for that entry in the result dictionary.
    
        Raises:
            requests.RequestException: In case an HTTP request is unsuccessful. 
                                        Indicates invalid JSON response or other request-related issues.
            ValueError: In case the input ID cannot be parsed as a string.
                          This covers scenarios where the ID is partially missing, such as "123-.x."
                """

        if not isinstance(cve_id, str):
            raise ValueError("CVE_ID must be a valid string.")
        
        url = f"https://api.example.com/v2/cves/{cve_id}"
        api_connection = self.get_api_connection()
        response = api_connection.get(url)
        try:
            response.raise_for_status()  # Raises an HTTPError if the HTTP request returned an unsuccessful status code.
            return response.json()
        except requests.RequestException as e:
            raise Exception(f"Could not retrieve CVE information for {cve_id}: {str(e)}")
        except ValueError as e:
            raise ValueError(f"CVE ID could not be parsed: {cve_id}. {str(e)}")

    def validate_and_fetch(self, cves_list):
        """
        Validates a list of CVEs received and fetches additional information if valid,
        handles invalid IDs gracefully.

        Parameters:
            cves_list (list): List of CVE-ID strings to check and fetch info on.
    
        Returns:
            dict: A dictionary where the keys are CVE_ID strings and values are their info dictionaries or None for invalid ones.

        Raises:
            ValueError: If input list contains non-string items other than CPE prefixes. 
                        For instance, lists containing "cpe:" prefixed entries as they are considered valid.
                """

        import re
        cve_info = {}
        
        if not all(isinstance(cve_id, str) for cve_id in cves_list):
            raise ValueError("All items in the CVE list must be strings.")
            
        regex_cpe_prefix_pattern = re.compile("^cpe:/")
        valid_entry_count = 0
        
        for item in cves_list:
            # Validating entries and fetching additional info
            if not regex_cpe_prefix_pattern.search(item):
                cve_info[item] = self.get_cve_info(item)
                valid_entry_count += 1
            else:
                cve_info[item] = None
            
        return {'valid_entries': valid_entry_count, 'invalid_entries': len(cves_list) - valid_entry_count, 'result': cve_info}
    
    def example_usage(self):
        """
        Example usage of CVEIntegration to demonstrate functionality.
    
        Returns:
            dict: An example return from validate_and_fetch method,
                  including the number of successfully fetched entries and those that failed validation or fetch.
                  
        Raises:
            ValueError: If used incorrectly (e.g., list containing None or non-string items)
                """
        
        examples = ["CPE-123", "CVE-456"]
        
        example_result = self.validate_and_fetch(examples)
        assert isinstance(example_result, dict)
        return example_result
