#!/usr/bin/env python

import os
from typing import List

class FileUploadScanner:
    def __init__(self, allowed_extensions: List[str]):
        """
        Initialize the FileUploadScanner with a list of allowed file extensions.

        :param allowed_extensions: List of valid file types to be allowed for upload.
                                    e.g. ['jpg', 'png', 'jpeg']
        """
        self.allowed_extensions = allowed_extensions

    def is_extension_valid(self, filename: str) -> bool:
        """
        Check if the file extension provided in the given filename is within the allowed extensions.

        :param filename: The name of the file including its extension.
        :return: True if the file type is valid, False otherwise.
        """
        _, file_ext = os.path.splitext(filename)
        return '.' + file_ext.strip().lower() in self.allowed_extensions

    def scan_file(self, upload_directory: str) -> List[str]:
        """
        Scan a directory for files with potential security vulnerabilities related to file extensions.

        :param upload_directory: The path of the directory to be scanned.
        :return: A list of filenames within the upload_directory that have invalid or missing extension.
        """
        invalid_files = []

        try:
            if not os.path.isdir(upload_directory):
                raise FileNotFoundError(f"{upload_directory} is not a directory")

            for root, _, files in os.walk(upload_directory):
                for file in files:
                    full_path = os.path.join(root, file)
                    filename, extension = os.path.splitext(full_path)

                    # Check if the file has an invalid extension
                    if not self.is_extension_valid(extension.strip()):
                        invalid_files.append(full_path)

        except Exception as e:
            print(f"Error during scanning: {e}")

        return invalid_files


def main():
    """
    Example usage of the FileUploadScanner. Demonstrates scanning a directory
    and printing out all filenames that are considered unsafe.
    """
    allowed_extensions = ['.jpg', '.png', '.jpeg']
    scanner = FileUploadScanner(allowed_extensions)
    upload_directory_path = '/path/to/upload/directory'
    
    invalid_file_paths = scanner.scan_file(upload_directory_path)

    if invalid_file_paths:
        print(f"Unsafe files found: {invalid_file_paths}")
    else:
        print("No unsafe files detected.")

if __name__ == "__main__":
    main()
