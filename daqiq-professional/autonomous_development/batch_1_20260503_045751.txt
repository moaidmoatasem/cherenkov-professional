import requests


def create_notification(url, message):
    """
    Creates and sends a webhook notification using the provided URL.

    :param url: str - The target endpoint for the webhook.
    :param message: dict - An object containing data that will be sent via the webhook.
                             Typically structured as {'key1': 'value1', ..., 'keyN': 'valueN'}.
    :returns bool: True if the notification was successfully sent, False otherwise.
    """
    # Define the headers to include for the request. This example assumes a simple authentication header "Authorization".
    headers = {
        "Content-Type": "application/json",
        "Authorization": "Bearer your_secret_token"
    }

    try:
        response = requests.post(url, json=message, headers=headers)
        response.raise_for_status()  # Raises an HTTPError if the HTTP request returned an unsuccessful status code.
    except requests.exceptions.HTTPError as http_err:
        print(f"HTTP error occurred: {http_err}")
        return False
    except Exception as err:
        print(f"An error occurred: {err}")
        return False

    status_code = response.status_code
    if status_code == 204:  # Assuming success is indicated by a 204 No Content status code when using POST.
        return True
    else:
        print("Webhook Notification Failed to Send.")
        return False
