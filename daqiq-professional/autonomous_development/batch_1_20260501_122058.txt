import requests

class VulnerabilityScoring:
    """
    A class to score vulnerabilities based on provided details and external API responses.
    
    Attributes:
        base_url (str): The base URL for API calls related to vulnerability scoring.
    """

    def __init__(self, base_url="https://api.vuln_scoring_service.com/"):
        """
        Initialize the VulnerabilityScoring class with a base url where vulnerability scores are retrieved.

        Args:
            base_url (str): A string representing the base url for API calls. Defaults to 'https://api.vuln_scoring_service.com/'.
        """
        self.base_url = base_url

    def __make_request(self, method: str, path: str, data=None) -> dict:
        """
        Make a request to an external vulnerability scoring service. Includes error handling for authentication and API failure.

        Args:
            method (str): The HTTP method ('GET', 'POST', etc.)
            path (str): The relative path from the base URL where vulnerability scores are retrieved.
            data (dict, optional): A dictionary containing query parameters or body content for a POST request. Defaults to None.
        
        Returns:
            dict: JSON response parsed into a Python object.

        Raises:
            requests.RequestException: If the method is not recognized by 'requests' library, 
                                       if an error occurs and authentication fails, 
                                       or if the API returns an error status code.
        """
        URL = f"{self.base_url}{path}"
        
        try:
            response = requests.request(method=method, url=URL, json=data)
            response.raise_for_status()
        except (requests.RequestException) as e:
            raise e
        
        return response.json()

    def get_score(self, details: dict) -> int:
        """
        Fetch vulnerability scores based on the provided vulnerabilities details.

        Args:
            details (dict): A dictionary containing all necessary data to retrieve a specific vulnerability score.
                
                Example structure for 'details':
                    {
                        "vulnerability_id": 123,
                        "severity_range_min": 4.0, 
                        "severity_range_max": 7.0
                    }
        
        Returns:
            int: A raw integer indicating the vulnerability score from the scoring service.

        Raises:
            requests.RequestException: If an error occurs and authentication fails in retrieving the score or if it returns an error status code.
        """
        try:
            return self.__make_request("GET", path="score", data=details)
        except Exception as e:
            raise requests.RequestException("Failed to retrieve score.", cause=e)


if __name__ == "__main__":
    def example_usage():
        details = {
            "vulnerability_id": 123,
            "severity_range_min": 4.0, 
            "severity_range_max": 7.0
        }

        vulnerability_scoring = VulnerabilityScoring()
        
        try:
            score = vulnerability_scoring.get_score(details)
            print(f"Vulnerability score retrieved: {score}")
        except requests.RequestException as e:
            print(f"Error occurred while retrieving score: {e}")

    example_usage()
