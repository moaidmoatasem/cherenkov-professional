import requests

class WebhookNotifier:
    def __init__(self):
        self.webhook_url = "https://example.com/webhooks/notify"
    
    @staticmethod
    def _handle_errors(response):
        """
        Handle errors from the webhook notification.
        
        Args:
            response (requests.Response): The HTTP response object.
            
        Returns:
            str: A message containing detailed information about the error or None if no error occurred.
        """
        error_message = None
        
        if response.status_code != 204:
            try:
                response.raise_for_status()
                error_message = f"Webhook failed with status code {response.status_code}: {str(response.text)}"
            except requests.RequestException as e:
                error_message = str(e)
        
        return error_message

    def notify(self, payload):
        """
        Send a webhook notification to the specified URL.
        
        Args:
            payload (dict): The data to be sent in the request body as JSON.
            
        Returns:
            bool: True if the response status is 204 (No Content) indicating successful delivery. False otherwise.
        """
        headers = {"Content-Type": "application/json"}
        try:
            response = requests.post(self.webhook_url, json=payload, headers=headers)
            error_message = self._handle_errors(response)
            
            return not bool(error_message) and response.status_code == 204
        except Exception as e:
            return False
