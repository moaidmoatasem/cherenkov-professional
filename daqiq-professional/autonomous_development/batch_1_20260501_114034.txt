                encoders.encode_base64(template_pdf)

            attachment['Content-ID'] = '<attachment>'
            attachment.attach(MIMEText('Vulnerability Report', 'plain'))

    def send_email(self, report_message):
        """Send the generated PDF report via email.

        Args:

        """
        import smtplib

        if not isinstance(report_message, MIMEMultipart):
            raise ValueError("Invalid message type. Expected MIMEMultipart")
