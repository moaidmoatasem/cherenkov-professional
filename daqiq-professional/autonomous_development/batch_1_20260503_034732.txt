#!/usr/bin/env python3

import os
import mimetypes
from logging import Logger

class FileUploadVulnerabilityScanner:
    """A class to find potential unsafe file upload vulnerabilities in web applications.
    
    This class scans directories for files uploaded via forms, looking for patterns that indicate a vulnerability 
    such as .php, .jsp, .py extensions. It also checks against known dangerous file types by checking mime-types and
    filenames.

    :param settings: A dictionary containing the paths to scan (directories).
    """

    def __init__(self, settings):
        self.settings = settings
        self.logger: Logger = settings['logger']

    def scan_for_vulnerabilities(self) -> None:
        """Scan specified directories for vulnerability indicators."""
        self._scan_directories(self.settings['directories_to_scan'])

    def _analyze_file_extension(self, file_path, extension: str):
        """Check if the path has a dangerous file extension."""
        mime_type = mimetypes.guess_type(file_path)[0]
        return f'.{extension}' in file_path or (mime_type and mime_type.startswith(f'text/{extension}'))

    def _scan_directories(self, directories_to_scan: list) -> None:
        for directory in directories_to_scan:
            # Iterate through all files in a directory
            for dirpath, _, filenames in os.walk(directory):
                relative_path = os.path.relpath(dirpath, self.settings['web_application_root'])
                for filename in filenames:
                    file_path = os.path.join(dirpath, filename)
                    if not self._analyze_file_extension(file_path, 'php'):
                        path_tuple = (dirpath, relative_path.replace(os.sep, '/'), filename)
                        # Check extension and mime type
                        print(f"{filename} with extension: {self._analyze_file_extension(file_path)} is considered for further analysis!")
    
    def _log(self, message):
        """Log a message after ensuring the logger level is set to logging.DEBUG."""
        if self.logger.getEffectiveLevel() == 20:
            self.logger.debug(message)
