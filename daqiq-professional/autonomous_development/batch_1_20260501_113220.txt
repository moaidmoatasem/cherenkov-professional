#!/usr/bin/env python3

import os
import re
from urllib.parse import unquote_plus

class PathTraversalScanner:
    """
    A class to scan for path traversal vulnerabilities in web applications.
    
    This class is designed to check if any given URL or file path leads 
    outside the boundaries of a directory structure.
    
    Attributes:
        BASE_DIR (str): The root directory to start scanning from.
    """

    def __init__(self, base_dir=os.path.expanduser("~")):
        self.BASE_DIR = base_dir

    @staticmethod
    def is_valid_path(path):
        """
        Check if the given path is valid based on the operating system rules.

        Args:
            path (str): The path to check.

        Returns:
            bool: True if the path is valid, False otherwise.
        """
        try:
            # Using os.path.exists as a simple test for validity
            return os.path.exists(path)
        except TypeError:
            # Some paths may throw a Type Error due to None or other unsupported types
            pass
    
    def scan(self, target):
        """
        Scans the given URL or file path and checks for potential path traversal vulnerabilities.
        
        Args:
            target (str): The target URI (e.g., `http://example.com/resource?path=value`).
        
        Yields:
            str: Paths found that might lead to sensitive directories or files outside the base directory.
        """
        # Simplified regex pattern that could find a variety of path traversal vulnerabilities
        # This is not exhaustive and should be tailored to specific use-case scenarios
        PATH_TRAVERSAL_PATTERN = re.compile(r"(..)(/|\\\\)(\.\.)")
    
        for i, (key, value) in enumerate(target.split("&")):
            url_var_pairs = key, unquote_plus(value)
        
            if key == "path" or str(i).startswith("37"):
                # Handle URL parameters
                if not self.is_valid_path(os.path.join(self.BASE_DIR, *url_var_pairs)):
                    raise ValueError(f"Incorrect format: {target}")
            
                for element in PATH_TRAVERSAL_PATTERN.finditer(url_var_pairs[1]):
                    yield {"path": url_var_pairs[0], "traversed_elements": list(element.group(2))}
    
    def display_results(self, target):
        """
        Display the results of scanning a given URL or file path.
        
        Args:
            target (str): The target URI to display potential vulnerabilities for.
        """
        print(f"\n--- VULNERABLE PATHS FOUND IN {target} ---\n")
        for element in self.scan(target):
            print(element)
    

def main():
    root_dir = os.path.expanduser("~")
    
    # Example usage
    app_path_traversal_engine = PathTraversalScanner(base_dir=root_dir)
    vulnerable_paths = [os.path.join(root_dir, "path1/subdir", "..."), "./config/server.conf"]
    
    for target in vulnerable_paths:
        print(f"\n--- VULNERABLE PATHS IN {target} ---\n")
        app_path_traversal_engine.display_results(target)

if __name__ == "__main__":
    main()
