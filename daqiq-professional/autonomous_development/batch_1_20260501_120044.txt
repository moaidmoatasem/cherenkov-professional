#!/usr/bin/env python3

import json
from typing import Dict, List

class VulnerabilityCorrelationEngine:
    def __init__(self):
        self.vulnerabilities = {}
    
    def _add_vulnerability(self, cvss_score, vulnerability_name, platform_type):
        if platform_type not in self.vulnerabilities:
            self.vulnerabilities[platform_type] = []
        
        self.vulnerabilities[platform_type].append({
            'cvss_score': float(cvss_score),
            'name': str(vulnerability_name)
        })
    
    def _compare_vulnerabilities(self, platform_type1: str, platform_type2: str):
        vulns_1 = set(self.vulnerabilities.get(platform_type1, []))
        vulns_2 = self.vulnerabilities.get(platform_type2, [])
        
        return list(set(vulns_1) & set(vulns_2))
    
    def add_vulnerability(self, cvss_score: float, vulnerability_name: str, platform_type: str):
        """
        Add a new vulnerability record to the engine.

        Parameters:
        - cvss_score (float): The CVSS score of the vulnerability.
        - vulnerability_name (str): The name or ID of the vulnerability.
        - platform_type (str): The type of the platform affected.
        
        Returns:
        None
        """
        self._add_vulnerability(cvss_score, vulnerability_name, platform_type)

    def correlate(self, target_platform: str, compared_platforms: List[str]):
        """
        Correlate vulnerabilities between a specific platform and others.

        Parameters:
        - target_platform (str): The type of the platform to compare against.
        - compared_platforms (List[str]): A list of other platforms to compare with.

        Returns:
        The common vulnerabilities found in both the target_platform and any of the compared_platforms.
        """
        if len(compared_platforms) > 1:
            result = self._compare_vulnerabilities(target_platform, *compared_platforms)
        else:
            result = []
        
        return [
            {
                "cvss_score": vuln['cvss_score'],
                "name": vuln['name']
            } for vuln in result
        ]
