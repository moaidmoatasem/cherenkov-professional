```python
import requests

def create_webhook(url: str) -> None:
    """
    Creates and configures a webhook based on the provided URL.
    
    :param url: The web server receiving notifications via Webhooks
    :type url: str
    :return: None
    :rtype: None
    
    Example Usage:
    
    >>> create_webhook("https://example.com/webhook")
    """
    # Step 1: Define the webhook request and parameters
    headers = {
        "Content-Type": "application/json"
    }
    
    def send_webhook(data: dict) -> None:
        """Sends a JSON object via POST to the provided URL.
        
        :param data: The payload body for Webhook notifications in JSON format
        :type data: dict
        """
        try:
            # Step 2: Send the webhook notification with error handling
            response = requests.post(url, headers=headers, json=data)
            response.raise_for_status()  # Raises an exception if the response status is not OK (status_code 200)
        
            print(f"Posted to {url} successfully. Status Code: {response.status_code}")
            
        except requests.exceptions.RequestException as e:
            # Step 3: Handle errors, such as connection or network issues
            print(f"Failed to send Webhook data to the server: {e}")

    return send_webhook

# Example usage 
if __name__ == "__main__":
    url = "https://example.com/webhook"
    
    webhook_sender = create_webhook(url)
    
    notification_data = {"eventType": "userLogin", "userId": 123, "timestamp": "2023-10-05T14:27:33Z"}
    webhook_sender(notification_data)

    problematic_payload_data = {'this is not json'}
    webhook_sender(problematic_payload_data)

```
