```yaml
name: Scan PyProject.toml for Security Issues

on:
  pull_request:
    branches:
      - main

jobs:
  scan_pypa_requirements:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout Repo
      uses: actions/checkout@v2
      
    - name: Cache dependencies
      uses: actions/cache@v2
      with:
        key: pyproject-toml-cache-${{ hash(inputs.python-version) }}
        path: requirements.txt
        integrity: sha256
        excludes:
          - '**/.DS_Store'
        versions:
          python-3.8: stable

    - name: Install Python dependencies
      id: install-deps
      run: |
        python --version
        # Install required packages for scanning dependencies like PyDepend and PySafest.
        
        pip install pypa-install-tools pypa-requirements safety coveralls

    - name: Run security scanner with PEP 582 compliance check
      id: scan-security
      uses: actions/checkout@v1.4
      with:
        git-ref: HEAD
        submodules: 'recursive'
        
      env:
        # Set required environment variables.
        PYTHON_VERSION: ${{ steps.install-deps.outputs.python-version }}
        USE_PYPASTORE: 'true'

      runs-on: ubuntu-latest
  
      steps:
      - name: Install PyDepend for scanning requirements.txt
        run: |
          python -m pypa-install --pre requisities
          
      - name: Run security tools scan
        if: success()
        env:
          # Example environment variables to ensure reproducibility.
          USE_PEP582_CONFORMANCE_CHECKER: true  # Check PEP-582 compliance with PyDependency version requirements.
          USE_PEP604_FIXFORMATS_SUGGESTION_ONLY: false  # False by default, do NOT auto-fix using "pip check --format=suggest".
        
        run: |
          # Use safety to scan for any dependencies that are vulnerable and log warnings in the output,
          # or ensure none of them are outdated.
          
          pip install PySafest
          
      - name: Capture output from security tools step.
        id: store-safety-output
        env:
          OUTPUT_FILE_CONTENT: ${{ steps.store-deps.outputs.stdout }}
          VULNERABILITY_OUTPUT: !SECURE!

        run: |
          echo "Safety scan results written to ${GITHUB_ENV}#vulnerabilities.txt"

      - name: Upload test file with safety output.
        uses: actions/upload-artifact@v2
        with:
          key: security-vulns  # Should be unique and informative
          path: vulnerabilities.txt

    - name: Publish PyDepend results to GitHub Actions graph.
      id: publish-results
      
      if: steps.scan-security.outcomes.secrets.security-vulns
      env:
        # Define environment variables for use with GitHub API. 
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
      
      runs-on: ubuntu-latest

      uses: codecov/codecov-action@v1
      if: failure()
    
    - name: Publish PySafest results to GitHub Actions graph.
      id: publish-safest-results
      
      env:
        # Environment variables used later are for use within workflows or with secrets.
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        
      runs-on: ubuntu-latest

      uses: codecov/codecov-action@v1
      if: failure()

    - name: Leave a comment about GitHub Actions graph publishing status.
      uses: peaceotw/action-comment@v3.0.0
      
      env:
        # Use environment variables with secure values in Secrets -> Repository secret page or GitHub API access token
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      if: null  # Ensure it runs only once, at the end of the job

      
    - name: Upload test file with PySafest issues.
      uses: actions/upload-artifact@v2
      with:
        key: safest-results  # Should be unique and informative
        path: safest_output.txt
      
    - name: Post a comment in GitHub Pull Request to indicate success or failure, which includes the upload of artifacts.
      uses: GitHub/github-action-send-comment@4.0.0
      if: steps.publish-safest-results.outcomes.succeeded || (
          steps.publish-safest-results.outcomes.failed
        && steps.publish-safest-results.outputs.artifacts_url
                        )
    
      with:
        secret_token: ${{ secrets.GITHUB_TOKEN }}

    - name: Close Pull Request Comment after actions complete.
      uses: github/action-close-comment@v1.7.2

outputs:
  result-url: ${{ steps.publish-safest-results.outputs.url }}
```
