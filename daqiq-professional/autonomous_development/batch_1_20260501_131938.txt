import requests
from urllib.parse import urljoin

class CVEIntegration:

    def fetch_cve_data(self, query=None):
        """
        Fetches CVE data from the NVD (National Vulnerability Database) API.
        
        Args:
            query (str, optional): Optional GET parameter to narrow search results. Defaults to None.

        Returns:
            dict: JSON response containing filtered CVE data if provided with a valid query, else CVE information for all vulnerabilities.
        """
        base_url = "https://services.nvd.nasa.gov/
        full_query_url = urljoin(base_url, 'searchService.json')
        
        headers = {
            'Accept': 'application/json'
        }
        
        parameters = {}
        
        if query:
            parameters['keyword'] = query
       
        response = requests.get(full_query_url, headers=headers, params=parameters)
        return response.json()
    
    def get_cve_details(self, cve_id):
        """
        Fetches detailed CVE information given a CVE ID.
        
        Args:
            cve_id (str): Unique identifier for the CVE in NVD.

        Returns:
            dict: JSON response containing the detailed CVE data if found; an appropriate message otherwise.
        """
        endpoint_url = urljoin(urljoin(self.fetch_cve_data(query=None)['result']['vulnData'][0]['itemsUrl'], 'details.json'))
        
        headers = {
            'Accept': 'application/json'
        }
        
        response = requests.get(endpoint_url, headers=headers)
        
        if cve_id not in response.json():
            return {'message': f"CVE ID {cve_id} not found."}
        
        return response.json()[str(cve_id)]
        
    def find_cves_by_query(self, query):
        """
        Searches for CVEs with the specified keyword.
        
        Args:
            query (str): Search term to use as the filter parameter.

        Returns:
            dict: JSON response containing all CVE data that match the inputted search criteria if a valid query is provided; an appropriate message otherwise.
        """    
        if not bool(query):
            return {'error': 'Query cannot be empty'}
        
        try:
            responses = self.fetch_cve_data(query=query)
            
            if isinstance(responses['result']['vulnData'], list) and not responses['result']['vulnData']:
                raise ValueError('No CVEs found')

            cve_list = []
            for data in responses['result']['vulnData']:
                summary = {
                    'cve_id': str(data['cveId']),
                    'affected_systems_count': int(data['totalAffectedSystemCount']),
                    'last_modified_date': str(data['lastModifiedDate']) if not None else "",
                }
                cve_list.append(summary)
            return {'CVEs Matched Search Criteria': cve_list}
        except Exception as e:
            return {'error': str(e)}
        
    def example_usage(self):
        """
        Example usage of the CVEIntegration features.
        """   
        print(f"Fetching CVEs related to 'SQL Injection':")
        search_results = self.find_cves_by_query("SQL Injection")
        for cve in search_results['CVEs Matched Search Criteria']:
            print(cve)
        
        if not bool(search_results):
            return {"Error": "No CVEs Found"}

        print("\nFetching detailed information on a single CVE:")
        detail_data = self.get_cve_details('CVE-2020-56895')
        print(detail_data)
    
    def perform_tests(self):
        """
        Perform unit tests to ensure robustness and security of the library.
        
        Note: Implementing comprehensive testing is outside the scope for this task, but it's important in a full-scale release phase.
        """
        # Example test case
        response = self.fetch_cve_data(query="SQL injection")
        assert response['totalVulnerabilitiesCount'] > 0, "No CVEs found with specified keyword"
        assert 'CVE-2023-42678' in [cve["ID"] for cve in response['result']['vulnData']], "Did not find CVE ID CQE-42678" 
        
if __name__ == "__main__":
    CVEIntegration().perform_tests()
