```python
import json

class ExecutiveSummaryGenerator:
    def generate_summary(self, security_report):
        # Validate input data (JSON format)
        try:
            report = json.loads(security_report)
        except ValueError as e:
            raise ValueError("Invalid JSON format in the security report.") from e
        
        summary_lines = []
        
        if not isinstance(report, list):
            raise ValueError("Security report must be a list of security findings.")
        
        for finding in report:
            summary_lines.append(self._build_line(finding))
            
        return "\n".join(summary_lines)
    
    def _build_line(self, finding_data: dict) -> str:
        """
        Build a line for the executive summary based on the given dictionary.

        :param finding_data: A dictionary containing information about a security finding.
         The expected keys are 'type', 'location', and 'description'.
        
        Example data structure of finding_data:
        {
            "type": "infoleak",
            "location": "/userprofile/sensitive_file.txt",
            "description": "A secure credentials file readable by all users."
        }
        
        Returns a formatted summary line.
        """
    
        return f"**{finding_data['type']}**: {finding_data['location'].ljust(50, '_')} - **{finding_data['description']}**"
```

import unittest

        findings = [

            "\n".join([
