import os
import subprocess

class CommandInjectionScanner:
    """
    CommandInjectionScanner class is designed to scan for command injection vulnerabilities.
    
    Methods:
        scan(command): Scans a given command for potential command injection attacks.
        
    Attributes:
        None specific attributes are defined, but error handling and docstrings are included.

    """

    def __init__(self):
        pass

    def scan(self, unsafe_command):
        """
        Method intended to parse and analyze the command for vulnerabilities.
        
            Args:
                unsafe_command (str): The unsafe command that needs inspection.
                
            Returns:
                str: A response with details regarding the injection findings or a message if it's safe.
                
         Raises:
                ValueError: If input is not a string type.
        """
        # Input validation
        if not isinstance(unsafe_command, str):
            raise ValueError("Input must be of string type.")
        
        try:
            subprocess.run(["echo", f"{os.path.join(os.getcwd(), unsafe_command)}"], check=True)
        except subprocess.CalledProcessError as error:
            return f"Potential Command Injection found! Error Message: {error}"
        return "No injection attack detected."

# Example Usage
def example_usage():
    scanner = CommandInjectionScanner()
    
    # Safe command - No vulnerabilities
    print(scanner.scan("ls"))
    
    # Unsafe command with potential for injection - Likely to detect injection.
    print(scanner.scan("/bin/sh -c '$1'"))

if __name__ == "__main__":
    try:
        example_usage()
    except Exception as e:
        print(f"An error occurred: {e}")

# Function to run unit tests if necessary
def check_scan_functionality():
    # Check safe command output
    assert "No injection attack detected." in CommandInjectionScanner().scan("ls"), "Safe command did not raise no vulnerability message."
    
    # Check unsafe command output, expecting an error message.
    assert "/bin/sh -c''" in CommandInjectionScanner().scan("/bin/sh -c '$1'"), "Unsafe command did not raise injection attack message."

# Run tests
if __name__ == "__main__":
    check_scan_functionality()
