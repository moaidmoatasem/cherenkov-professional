"""

import requests

class CVEIntegration:
    def __init__(self, base_url='https://api.cvedb.io/v1'):
        self.base_url = base_url

    def get_vulnerability_info(self, cve_id: str) -> dict:
        """
        Retrieve details about a specific CVE.

        :param cve_id: The CVE ID identifier.
        :return: A dictionary containing the information for the specified CVE or None if not found.
        """
        headers = {
            'Accept': 'application/json'
        }
        
        response = requests.get(f'{self.base_url}/{cve_id}', headers=headers)
        response.raise_for_status()
        return response.json()

    def search_vulnerabilities(self, description: str):
        """
        Search for vulnerabilities based on a keyword description.

        :param description: The keyword to use when searching the CVE.
        :return: A list of dictionaries each containing details about a found vulnerability.
        """

        params = {
            'description': description
        }
        
        response = requests.get(f'{self.base_url}/vulnerabilities', headers=headers, params=params)
        response.raise_for_status()
        return [res for res in response.json()]

def get_vulnerability_info(cve_id: str) -> dict:
    """
    Helper function to get CVE info by cve_id. 

    This is a convenience wrapper around the 'CVEIntegration.get_vulnerability_info' method.
    
    :param cve_id: Specific CVE identifier (string format).
    :return: A dictionary with the retrieved information or None on failure.
    """
    integration = CVEIntegration()
    return integration.get_vulnerability_info(cve_id)

def search_vulnerabilities(description: str) -> list:
    """
    Helper function to search for vulnerabilities by keyword. 

    This is a convenience wrapper around the 'CVEIntegration.search_vulnerabilities' method.
    
    :param description: Keyword used for searching (string format).
    :return: A list of dictionaries with each dictionary containing information about matching CVE's.
    """
    integration = CVEIntegration()
    return integration.search_vulnerabilities(description)

if __name__ == "__main__":
    print("CVE Details:", get_vulnerability_info(cve_id='CVE-2023-4567'))
    results = search_vulnerabilities('SQL Injection')
    for result in results:
        print(result)
