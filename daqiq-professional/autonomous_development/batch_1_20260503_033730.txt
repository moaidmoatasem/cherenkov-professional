#!/usr/bin/env python3

"""
XXE Scanner Tool
A simple and effective library to identify XML external entity injection vulnerabilities.
"""

import sys
from xml.parsers.expat import ExpException
import logging
logging.basicConfig(level=logging.INFO, format="[%(asctime)s] %(message)s")

def scan_xml_for_xxes(xml_string):
    """
    Scans an input XML string for XXE (XML External Entity) vulnerabilities.

    :param xml_string: str - The input XML to be scanned.
    :return: Set of found external entity URIs or None if error occurred.
    """
    try:
        parser = _create_xml_parser()
        parser.declared_entities = set()  # Track declared entities
        parsed_doc = parser.Parse(xml_string)
        declared_entity_uris = {
            f"http://{entity_name}" for entity_name, _ in parser.declared_entities.items()
        }
        return declared_entity_uris

    except ExpException as e:
        logging.error(f"Parsing Error: {e}")
        return None
    except Exception as exp:
        logging.error(f"Unexpected Error Occurred: {exp}")
        return None


def _create_xml_parser():
    """
    Creates and returns an XML Parser object with controlled capabilities to prevent XXE.

    :return: An expat XML parser instance.
    """
    # Note: The actual implementation of restricting the XML parser might vary
    # based on the Python version used, this function is just for demonstration purposes.
    import xml.parsers.expat
    return xml.parsers.expat.ParserCreate()


def main(args):
    if not args:
        logging.error("No input provided.")
        sys.exit(1)

    scan_result = scan_xml_for_xxes(args[0])
    if scan_result:  
        print(f"XML contains known XXE vulnerabilities with entities at these locations:")
        for uri in sorted(scan_result):
            print(uri)
    else:
        logging.error("Failed to parse the XML input.")


if __name as __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser(description="Xxx Scanner")
    parser.add_argument("--input", dest="xml_input", type=str, required=True,
                        help="The XML document to check for XXE vulnerabilities.")
    args = parser.parse_args()
    main(args)


if __name__ == "__main__":
    # Example usage:
