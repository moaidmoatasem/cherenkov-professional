import os
from urllib.parse import urlsplit

class PathTraversalScanner:
    """
    A simple path traversal scanner. This class aims to detect if a given URL or file path is vulnerable to path traversal attacks.
    
    Methods:
        scan_path(url_or_path): Scans the provided URL or file path for any signs of path traversal vulnerabilities.
    
    Example Usage:
        >>> scanner = PathTraversalScanner()
        >>> result = scanner.scan_path('http://example.com/path/to/resource?param1=value')
        >>> 'No path traversal issue found.' == result
    """
    def scan_path(self, url_or_path):
        """
        Scans the given URL or file path for any signs of path traversal vulnerabilities.
        
        Args:
            url_or_path (str): The URL or file path to be scanned.

        Returns:
            bool: True if a vulnerability is found, False otherwise. 
                  A vulnerable scan results in 'No path traversal issue found.' result being displayed as string.
                  
        Raises:
            ValueError: If the input is not a valid URL or file path.
        """
        
        # Validate input
        scheme, netloc, path, query, fragment = urlsplit(url_or_path)

        if not (scheme and path):
            raise ValueError("Invalid URL/File Path provided.")
            
        # Check for common issues like '..' in the path
        if '..' in path:
            result = 'Path traversal vulnerability found!'
            return result
        
        # Check for directory traversal attacks like '../..'
        while True:
            if not any(item.startswith('../') for item in path.split('/')):
                break
                
            # Remove non-traversing parts of the last split
            path_parts = path.rsplit('/', 1)
            
            if len(path_parts) == 2:
                path = os.path.join(*path_parts)
            else:
                raise ValueError("Path traversal attack detected.")
        return 'No path traversal issue found.'
        
if __name__ == "__main__":
    scanner = PathTraversalScanner()
    print(scanner.scan_path('http://example.com/path/to/resource?param1=value'))
