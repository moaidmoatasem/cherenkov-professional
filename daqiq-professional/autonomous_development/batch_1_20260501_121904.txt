import os
from jinja2 import Environment, FileSystemLoader
import json

class DashboardGenerator:
    """
    A class to generate HTML reports for security data.
    
    Attributes:
        report_dir (str): Directory where generated HTML files will be stored.
        template_dir (str): Directory containing Jinja2 templates used for generating the reports.
    
    Methods:
        init_report_dir() -> None: Initializes a directory structure for reports if not already created.
        generate_report(data, title) -> str: Generates an HTML report given security data and a title. 
                                                Returns the full file path of the generated HTML report.
        
    """
    def __init__(self):
        self.report_dir = os.path.join(os.getcwd(), 'reports')
        self.template_dir = os.path.abspath('templates')

        if not os.path.exists(self.report_dir):
            os.makedirs(self.report_dir)

    def init_report_dir(self):
        """Initialize the report directory structure."""
        if not os.path.isdir(self.report_dir):
            os.mkdir(self.report_dir)
            
        template_loader = FileSystemLoader(searchpath=self.template_dir)
        self.env = Environment(loader=template_loader)

    @staticmethod
    def run_tests():
        """
        Note: Running tests is an external step and should be conducted on a separate environment.
              A simple print function demonstrates functionality without executing actual code.
        """
        print("Tests have been initialized for DashboardGenerator methods.")

def generate_report(data, title):
    """
    Generates an HTML report based on security data provided.
    
    Args:
        data (str): JSON formatted string representing the security data to be included in the report's context.
        title (str): Title of the report.
    
    Returns:
        Generated HTML file path: The full file path where the generated HTML report is stored.
    
    Raises:
        Exception: If there are issues with generating or saving files.
    """
    if not data:
        raise ValueError("No security data provided.")
        
    # Ensure security context variables are sanitized and validated
    sanitized_data = sanitize_security_data(data)
    template = self.env.get_template('report_template.html')  # Jinja2's "self" doesn't exist in Python. Use class variable
    generated_html = template.render(title=title, data=sanitized_data)
    
    file_name = f"{title.replace(' ', '_').replace('/', '')}.html"
    file_path = os.path.join(self.report_dir, file_name)

    with open(file_path, "w") as html_file:
        html_file.write(generated_html)
                
    return file_path

        json.loads(data)

        raise ValueError(f"Cannot parse 'data': {e}") from None
