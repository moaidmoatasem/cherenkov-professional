```python
# import necessary packages
import json
from flask import Flask, jsonify, render_template

app = Flask(__name__)

@app.route('/')
def index():
    # Example data for dashboard
    dashboard_data = {
        "page1": {"data": [{"id": 1, "title": "Sales Report", "subheader": "Monthly Sales Overview"}],
                  "error": None},
        "page2": {"data": [{"id": 2, "title": "Inventory Check", "subheader": "Current Inventory Level"}],
                   "error": json.dumps({"errors": []})},
        "page3": {"data": [],
                   "error": None}
    }

    return render_template('index.html', pages=dashboard_data)

@app.errorhandler(404)
def page_not_found(e):
    """Error handler for 404 errors"""
    error_message = {
                      "errors": [{
                            "code": 404,
                            "data": e,
                            "message": "Resource not found"
                          }]
                     }
    
    return jsonify(error_message), 404

@app.errorhandler(500)
def internal_server_error(e):
    """Error handler for 500 errors"""
    error_message = {
                      "errors": [{
                                "code": 500,
                                "data": e,
                                "message": "Internal server error"
                              }]
                     }
    
    return jsonify(error_message), 500

if __name__ == '__main__':
    app.run(port=8081, debug=True)
```
