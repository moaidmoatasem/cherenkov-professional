# create_github_actions/actions/main.py

import json
from typing import Union

def perform_feature(feature: str) -> dict:
    """
    Perform the given feature.

    :param feature: The feature to be executed. Expected format: {'name': '<feature_name>', 'args': [...]}
    :return: A dictionary containing the result of the feature execution.
             Each key is the input parameter name and each value is the corresponding output value.
    """
    RESULT = {
        "success": False
    }
    
    # Example data model for features
    features = [
        {"name": "setup_environment", "args": {}},
        {"name": "compile_code", "args": {}},
        {"name": "test_coverage", "args": []}
    ]
    
    feature_data: dict = json.loads(feature)
    
    for curr_feature in features:
        if curr_feature["name"] == feature_data.get("name"):
            RESULT.update({"success": True, "status": curr_feature["name"], **curr_feature["args"]})
            RESULT.pop("args")
            return RESULT
    
    raise ValueError(f"Feature '{feature_data['name']}' not found")

def error_handling(func: callable) -> Union[dict, str]:
    """
    A helper function to handle errors. This function is decorated on top of the feature functions.
    
    :param func: The feature function.
    :return: Result dict or a string if an exception occurred during execution.
    """
    try:
        result = func()
        return result
    except ValueError as err:
        return {"error": str(err)}
    except Exception as e:
        return {"error": f"An unexpected error occurred: {str(e)}"}
    
def example_usage() -> dict:
    """
    Example usage of the create_github_actions/actions/main.py functionalities.

    :return: A dictionary representing the result or an error message in case of failure.
    """
    feature = json.dumps({"name": "setup_environment", "args": {}})
    return error_handling(lambda: perform_feature(feature))

if __name__ == "__main__":
    example_result = example_usage()
    print(json.dumps(example_result, indent=2))

# .github/workflows/main.yml

name: Create GitHub Action - Main Workflow

on:
  workflow_dispatch:

jobs:
  main_job:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v2
      
      - name: Execute Feature Function
        id: execute-feature
        shell: bash
        run: |
          echo "Feature name to be executed:" "$TRAVIS_TAG"
          python ./create_github_actions/main.py

      - name: Log Status
        if: steps.execute-feature.outputs.status != 'null'
        env:
          STATUS_OUTPUT: ${{ steps.execute-feature.outputs.status }}
        run: echo "Execution Status: ${STATUS_OUTPUT:-not set}"
        
      - name: Checkout Result Directory
        uses: actions/checkout@v2
  
      - name: Log JSON Output
        if: steps.execute-feature.outputs.result != 'null'
        env:
          OUTPUT_JSON: ${{ steps.execute-feature.outputs.result }}
        run: echo "Test Execution Result JSON:"
        args:
          --format="pretty"
