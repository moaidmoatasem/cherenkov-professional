import json
from flask import Flask, render_template

app = Flask(__name__)

def create_dashboard_report(data, error=False):
    """
    Creates an HTML report for dashboard data.
    
    :param data: A list of dictionaries where each dictionary represents a row in the table.
                 Each dictionary contains at least 'user' and 'accessed_pages' keys.
    :param error: Boolean indicating if an error occurred during processing (defaults to False).
    :return: String containing the HTML content of the report, encoded as JSON for browser readability.
    """
    if not isinstance(data, list) or len(data) == 0:
        raise ValueError("Data should be a non-empty list of dictionaries.")
    
    # Example error handling setup
    error_message = None
    
    if error:
        error_message = "An unexpected processing error occurred."
    
    html_content = f"""
    <!DOCTYPE html>
    <html lang="en">
    <head>
        <meta charset="UTF-8">
        <title>Dashboard Report</title>
    </head>
    <body>
        {render_dashboard(data, error_message)}
    </body>
    </html>
    """
    
    return json.dumps(html_content)

def render_dashboard(data, error=None):
    html_string = "<table>"
    if error:
        html_string += "Error: {}\n".format(error)
    
    # Construct HTML rows
    for row in data:
        user = row.get('user', 'Unknown User')
        accessed_pages = ", ".join(row.get('accessed_pages', ['Unknown Page']))
        html_string += "<tr>"
        html_string += f"<td>{user}</td>"
        html_string += f"<td>{accessed_pages}</td>"
        html_string += "</tr>\n"
    
    html_string += "</table>"
    return html_string

@app.route('/dashboard-report')
def dashboard():
    data = [
        {"user": "Alice", "accessed_pages": ["Home Page", "Products"]],
        {"user": "Bob", "accessed_pages": ["Home Page"]},
        {"user": "Charlie", "accessed_pages": []}
    ]
    
    # Call function with non-erroneous data
    try:
        report = create_dashboard_report(data)
        return report, 200
    except ValueError as e:
        return str(e), 400

if __name__ == "__main__":
    app.run(debug=True)
