import subprocess
from typing import List

class CommandInjectionScanner:
    """
    A class to scan and detect command injection in Python. It is designed 
    to prevent common security vulnerabilities by ensuring that untrusted data 
    does not influence the construction of a command.

    Attributes:
        None, as this is a static utility function.
        
    Usage examples:
        >>> scanner = CommandInjectionScanner()
        >>> inputs = ["ls", "--color=auto"]
        >>> for input_ in inputs:
        ...     result = scanner.check_command(input_)
        ...     print(result)
    """

    @staticmethod
    def check_command(command_string: str) -> List[str]:
        """
        Method to safely execute a command from a user-input.
        
        Parameters:
            command_string (str): User-provided string, expected to be a valid and safe shell command.

        Returns:
            List[str]: A list containing the results of the command, or None if execution fails due to an injection attempt.
            
        Raises:
            ValueError: If input is not provided as a string type.
        """
        
    @staticmethod
    def ensure_str_value(value):
        """Converts any value to a str"""
        try:
            return str(value)
        except TypeError as e:
            raise ValueError(f"Input must be of str or convertible to str. Got: {type(value).__name__}") from e

    def execute_command(self, sanitized_command: str) -> str:
        """ Executes the command and returns its output."""
        return subprocess.getoutput(sanitized_command)

    def check_command_safety(self, user_input: str) -> bool:
        try:
            if not user_input:
                raise ValueError("Input cannot be an empty string.")
            
            # Basic validation to ensure at least 1 shell character
            if len(user_input) < 2 or ' ' in sanitized_command[0] or sanitized_command.endswith(' '):
                return False
            
            # Ensure at least one command and no leading/trailing whitespace
            first_char = user_input[0]
            
            for index, char in enumerate(user_input):
                if (char.isspace() and not str.isascii(first_char)):
                    return False
                
                elif (not (str.isalnum(char) or (char == ' ')) and not str.isascii(char) and char != first_char:
                    return False
            
        except Exception as e:
            raise ValueError(f"Unexpected error evaluating if input is a command: {e}")
        
        return True
        
    def check_command(self, user_input: str):
        """
        Method to ensure we have a valid and safe shell command from untrusted input.
        
        Parameters:
            user_input (str): Untrusted shell command string provided by a user. 
        
        Returns:
            List[str]: The output or raises an error if it detects possible command injection.

        Raises:
            ValueError: If the input is not a valid and safe shell command.
        """
        
        sanitized_command = self.ensure_str_value(user_input)
        self.check_command_safety(sanitized_command) 
        try:
            result = self.execute_command(sanitized_command)
            return [f"Command executed successfully. Output follows:\n {result}"]
        except subprocess.CalledProcessError as e:  # This means something went wrong
            return "There was an error executing the command."
        except ValueError as e: 
            return f"[ERROR] Input is not a valid and safe shell command. Error: {str(e)}"
