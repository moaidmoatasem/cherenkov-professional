import requests
import json

class CVEDatabaseIntegration:
    """
    This class contains methods to integrate and interact with the CVE (Common Vulnerabilities and Exposures) database.
    
    It provides functionality for fetching vulnerabilities given a query, checking compatibility of provided data types, 
    and managing various error scenarios during API calls. Example usage is also provided within this docstring.
    """
    
    def __init__(self):
        self.base_url = "https://services.nvd.nist.gov/rest/json/cves/1.0"
        
    def fetch_vulnerabilities(self, query: str) -> list:
        """ Fetch vulnerabilities from the CVE database given a query string.
        
        Args:
            query (str): A string to query for in the CVE database.

        Returns:
            list: A list of dictionaries containing vulnerability details if successful; otherwise, an empty list.
        
        Raises:
            requests.RequestException: If there is a network problem during the API call. 
            TypeError: If the `query` parameter type is not supported (should be str).
            ValueError: If any data in provided dictionary does not match expected types.
        """
        if not isinstance(query, str):
            raise TypeError("The query must be of type 'str'")
        
        # Perform the API call
        try:
            response = requests.get(f"{self.base_url}?keyword={query}")
            return json.loads(response.text)
        except requests.RequestException as e:
            raise requests.RequestException(f"An error occurred while fetching data: {e}") from None

def example_usage():
    """ Example usage of CVEDatabaseIntegration """
    query = input("Enter CVE query to fetch vulnerabilities: ")
    cve_integration = CVEDatabaseIntegration()
    try:
        vulnerabilities = cve_integration.fetch_vulnerabilities(query)
        if not len(vulnerabilities):
            print(f"No vulnerabilities found for '{query}'")
        else:
            print(json.dumps(vulnerabilities, indent=4))
    except (requests.RequestException, ValueError, TypeError) as e:
        print(e)

if __name__ == "__main__":
    example_usage()
