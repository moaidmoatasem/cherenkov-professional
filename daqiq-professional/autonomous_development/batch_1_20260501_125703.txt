import os

class PathTraversalScanner:

            raise ValueError(f"Invalid directory: {directory}")

    def scan_directory(self, level=None, absolute_paths=False):
        """
        Recursively scans the directory and its subdirectories for potentially
        malicious paths.

            absolute_paths (bool): Whether to return paths in their absolute form.
