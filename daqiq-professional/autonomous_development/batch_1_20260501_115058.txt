import requests
from json import dumps
from functools import wraps
from inspect import signature, getargspec
import logging

logger = logging.getLogger(__name__)

def handle_exceptions(func):
    """
    Decorator to catch exceptions and log them for better error handling.
    
    :param func: The function to decorate
    :type func: Callable
    :return: wrapped function with exception handling
    :rtype: Callable
    """
    @wraps(func)
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except Exception as e:
            logger.error(f"An error occurred while processing {func.__name__}: {e}", exc_info=True)
            # Define what you want to do on an exception here
            raise  # Raise the original exception or modify it based on your application's needs.
    return wrapper

def create_webhook_notification(webhook_url: str, event_data: dict) -> None:
    """
    Sends a POST request with the provided event data to the specified webhook URL.

    :param webhook_url: The endpoint for which the notification is being sent
    :type webhook_url: str
    :param event_data: The payload of data to be included in the webhook body
    :type event_data: dict
    :return: None
    :rtype: None
    """
    headers = {
        'Content-Type': 'application/json'
    }
    response = requests.post(webhook_url, data=dumps(event_data), headers=headers)
    if response.status_code != 200 and response.status_code != 201:
        raise Exception(f"Failed to send webhook. Status Code: {response.status_code}, Response Body: {response.text}")

@handle_exceptions
def main():
    """
    Main function that acts as an entry point for the application.

    :return: None
    """
    # Example usage, you should replace these with actual data in a production environment.
    event_data = {
        'event_name': 'purchase',
        'user_id': 12345,
        'timestamp': '2023-10-05T14:30:00'
    }
    _create_webhook_notification(event_data)
    
def _create_webhook_notification(event_data):
    """
    This is a helper function to create and send the webhook notification.

    :param event_data: The payload of data to be included in the webhook body
    :type event_data: dict
    """
    webhook_url = 'https://example.com/webhooks/notify'
    create_webhook_notification(webhook_url, event_data)

if __name__ == "__main__":
    main()
