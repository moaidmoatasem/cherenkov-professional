import os

class PathTraversalScanner:
    """
    A class to scan for potential path traversal vulnerabilities in given input paths.
    
    :Examples of usage:
    scanner = PathTraversalScanner()
    is_vulnerable = scanner.scan_directory('/tmp')
    if is_vulnerable:
        print("Potential Path Traversal Vulnerability detected.")
    """

    def __init__(self, base_path: str):
        """
        Initializes the Scanner with a Base path.

        :param base_path: The directory to start scanning from.
        """
        self.base_path = os.path.abspath(base_path)  # Ensure the path is absolute
        if not os.path.isdir(self.base_path):
            raise ValueError(f"Invalid Path: {self.base_path}, Not a Directory")

    def scan_directory(self, relative_path: str) -> bool:
        """
        Scans for potential Path Traversal vulnerabilities in a given directory and its subdirectories.

        :param relative_path: The relative path within the base path to check.
        :return True if the path is considered vulnerable, False otherwise.
        """
        full_path = os.path.abspath(os.path.join(self.base_path, relative_path))
        
        try:
            # Attempt to expand tilde (~) patterns
            username_part = os.environ.get('HOME', '') + relative_path.strip(os.path.sep)
            expanded_full_path = os.path.expanduser(username_part)
            
            if os.access(expanded_full_path, os.W_OK):
                return True
                
            parts = os.path.split(full_path)
        except Exception as e:
            raise ValueError(f"Path traversal invalid: {full_path}. Error {e} occurred.") from e

        # Check for path traversal in components of the full path without raising an exception
        while parts[0] and not os.access(os.path.join(*parts), os.W_OK):
            parts.pop(1)  # Pop the next separator so we can continue checking relative to our parent
        
        return bool(parts[0] and os.access(os.path.join(*parts), os.W_OK))

def scan_directories(scanner, directories: list[str]) -> None:
    """
    Scans multiple directories for potential path traversal vulnerabilities.

    :param scanner: An instance of PathTraversalScanner.
    :param directories: An iterable containing the directory paths to scan.
    """
    vulnerable_paths = set()
    
    for directory in directories:
        if scanner.scan_directory(directory):
            print(f"Potential Path Traversal Vulnerability detected in directory {directory}.")

    return vulnerable_paths

def main():
    # Example usage of the PathTraversalScanner, assumes we have valid base paths to scan
    scan_directories(PathTraversalScanner('/'), ['/tmp', '/var'])

if __name__ == "__main__":
    main()
