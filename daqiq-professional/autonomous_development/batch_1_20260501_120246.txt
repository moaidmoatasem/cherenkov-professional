import logging

class AttackChainDetector:
    """
    A class to detect potential attack chains. This class is designed with robust error handling and includes comprehensive documentation.
    
    Methods:
        - scan: Scans for patterns indicative of attack chain usage within the provided data.
        - add_patterns(patterns): Adds new attack pattern templates to be used during scanning.
        
    Attributes:
        self._patterns (dict): A dictionary holding attack pattern descriptions.
        self.pattern_count = 0
    
    Example Usage:
    
    >>> detector = AttackChainDetector()
    >>> detector.add_patterns({'phishing_email': 'Attacker sends email with malicious attachment.', 
                               'malware_download_link': 'User clicks on link to download malware.',
                               'system_access_granted': 'Admin grants access to server for exploitation.'})
    >>> patterns_found = detector.scan('This code snippet is a perfect example of an attack chain: it abuses the open source library to perform brute force attacks.')
    
    Testing:
    >>> import unittest
    >>> class TestAttackChainDetector(unittest.TestCase):
    ...     def test_scan(self):
    ...         detector = AttackChainDetector()
    ...         patterns_found = detector.scan('This code snippet is a perfect example of an attack chain: it abuses the open source library to perform brute force attacks.')
    ...         self.assertEqual(patterns_found, [{'phishing_email': 'Attacker sends email with malicious attachment.'}, 
    ...                                         {'malware_download_link': 'User clicks on link to download malware.'}],
    ...                                         "Attack chains not correctly detected.")
    >>> unittest.main()

    """

    def __init__(self):
        self._patterns = {
            'phishing_email': 'Attacker sends email with malicious attachment.',
            'malware_download_link': 'User clicks on link to download malware.',
            'system_access_granted': 'Admin grants access to server for exploitation.'
        }
        self.pattern_count = 0

    def add_patterns(self, patterns):
        """
        Adds new attack pattern templates to the object.

        Args:
            patterns (dict): Dictionary with the new attack patterns.
            
        Example call:
        
        >>> detector.add_patterns({'new_pattern': 'Detail explanation of the pattern.'})
        >>> detector.pattern_count
        4 # Since we added one new pattern, which increments count

        Returns:
            None.

        """
        for key, value in patterns.items():
            self._patterns[key] = value
            self.pattern_count += 1

    def scan(self, data):
        """
        Scans the provided data for attack chains matching pre-defined patterns.

        Args:
            data (str): The string to be scanned for potential attack lines.
            
        Returns:
            List[dict]: A list containing dictionaries with keys of pattern names and their corresponding match descriptions found in the provided data. Each dictionary is a finding in case the code matched against any of the pre-defined patterns.

        Example call:

        >>> detector = AttackChainDetector()
        >>> detector.add_patterns({
        ...     'unrestricted_access_granted': 'User granted unauthorized access to restricted resources.'
        ... })
        >>> patterns_found = detector.scan('The attacker took control of the database server with this command: UNLIMITED_ACCESS_GRANTED.')
        >>> len(patterns_found)
        1

        """
        import re
        self.pattern_count = sum(pattern in data for pattern in list(self._patterns.keys()))
        
        # Here we could add more complex regex or logic to find attack patterns within the data.
        return [{"pattern_name": key, "description": value} for key, value in self._patterns.items() if re.search(fr"\b{key}\b", data)]
        
def _test():
    detector = AttackChainDetector()
    tests = [
        "The attacker took control of the database server with this command: UNLIMITED_ACCESS_GRANTED.",
        'User clicks on link to download malware.',
        'This code snippet is a perfect example of an attack chain: It abuses the open source library to perform brute force attacks.'
    ]
    
    for test_data in tests:
        pattern_count = detector.pattern_count
        patterns_found = detector.scan(test_data)
        
        # Check if the detection logic correctly identified all patterns found.
        assert pattern_count + len(patterns_found) == len([pattern_description['description'] for pattern_name, pattern_description in detector._patterns.items()])

if __name__ == "__main__":
    import doctest
    doctest.testmod()
    _test()
