def assign_severity(severity_dict):
    """
    Assigns a final severity level to the vulnerabilities based on their combined scores.
    
    :param severity_dict: The dictionary of vulnerabilities with associated severity scores.
    :return: A list containing tuples of all detected issues, where each tuple is (Vulnerability Name, Overall Severity Score).
    """

    from collections import Counter

    overall_severity = sum(severity_dict.values())
    sorted_vulns_by_score = sorted(severity_dict.items(), reverse=True)

    issue_list = []
    severity_thresholds = {
        1: "High",
        2: "Medium",
        3: "Low"
    }

    for vulnerability, score in sorted_vulns_by_score:
        issue_list.extend([f"{vulnerability} ({score})" for _ in range(int(score))])
        if overall_severity // score < len(issue_list): 
            severity_label = severity_thresholds.get(overall_severity % score)
            if severity_label:
                break
                
    formatted_issues = [issue.rstrip(" ")+f" ({severity_label})" if not issue.endswith(severity_label) else issue for issue in issue_list]
    return formatted_issues
