```python
import os
import subprocess

class CommandInjectionScanner:
    def __init__(self):
        self.valid_commands = None  # This can be set during scanning if needed

    def scan_command_list(self, commands):
        """Scans a list of commands and identifies any likely command injection vulnerabilities.
        
        Args:
            commands (list): A list of strings representing command lines.

        Returns:
            list: List of potentially vulnerable command line entries with details about identified issues. 
                  The returned list contains dictionaries:
                    {
                        "command": string,  # Original unscanned command
                        "issue_desc": string,  # Description of the issue found (e.g., OS Command Injection)
                        "example_of_attack": str,   # Example attack scenario or sample payload
                    }
        """
        results = []
        for index, cmd in enumerate(commands):
            if any([cmd.startswith(('os.', 'subprocess.').lower())]):
                results.append({
                    "command": f"commands[{index}]",
                    "issue_desc": "Command Injection Vulnerability (Possible OS Command or Subprocess call present).",
                    "example_of_attack": "For example, if the scanner can execute os.system('rm -rf /') by supplying user inputs."
                })
            elif '!' in cmd:
                results.append({
                    "command": f"commands[{index}]",
                    "issue_desc": "Command Injection Vulnerability (Triggered by a bang or exclamation point character).",
                    "example_of_attack": "If the command includes !, an attacker could exploit this to achieve unintended actions."
                })
        return results

    def scan_command(self, cmd):
        """Scans a single input string and identifies command injection issues.

        Args:
            cmd (str): A single command line entry.

        Returns:
            dict: Information about the issue found (if any) or null if no issue.
                  Can be one of the dictionaries from `scan_command_list`.
        """
        # Simplified validation, in a real case environment this would be more complex
        issues = self.scan_command_list([cmd])
        return next((issue for issue in issues), None)


def main():
    """Example usage function to demonstrate how to use the CommandInjectionScanner class."""
    commands = ["whoami", "!echo hi", "os.system('rm -rf /')"]

    scanner = CommandInjectionScanner()
    
    # Check a single command
    result_single = scanner.scan_command(commands[2])
    print(f"Single test:\n{result_single}")
    
    # Check multiple commmands at once
    results_combined = scanner.scan_command_list(commands)
    print(f"Multiple commands:\n{results_combined}\n")

    # Error handling in real-world applications might be more complex but included for simplicity here.
    try:
        results_combined = scanner.scan_command_list(['os.system("whoami")'])
        if any(issue['issue_desc'] == "Command Injection Vulnerability (Possible OS Command or Subprocess call present)."
                for issue in results_combined):
            print("Attempted attack detected, output shown above.")
        else:
            raise ValueError("No command injection issues found!")
    except Exception as e:
        print(f"Exception encountered: {e}")

# Execute the main function if this script is run from the command line
if __name__ == "__main__":
```
