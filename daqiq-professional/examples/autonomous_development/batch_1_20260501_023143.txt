```python
import sys

class VulnerabilityScoring:
    """
    A class to score vulnerabilities based on severity levels.
    
    Methods:
        - score_vulnerability(vuln_severity: str) -> int: Scores the vulnerability level.
        
    """

    # Severity levels and their corresponding scores
    SEVERITY_LEVELS = {
        "High": 10,
        "Medium": 7,
        "Low": 3,
        "Info": 1,
    }

    def score_vulnerability(self, vuln_severity: str) -> int:
        """
        Scores the vulnerability based on its severity level.
        
        Args:
            vuln_severity (str): The severity of the vulnerability ("High", "Medium", "Low", or "Info").

        Returns:
            The score assigned to the given vulnerability severity.

        Raises:
            ValueError: If the input severity is not recognized.
        """
        
        try:
            return self.SEVERITY_LEVELS[vuln_severity.title()]
        except KeyError as e:
            raise ValueError(f"Unknown vulnerability severity level '{vuln_severity}'") from e

def main():
    """Example usage of VulnerabilityScoring instance."""
    scorer = VulnerabilityScoring()
    
    # Testing the score_vulnerability method with valid and invalid inputs 
    assert scorer.score_vulnerability('High') == 10
    assert scorer.score_vulnerability('Medium') == 7
    assert scorer.score_vulnerability('Low') == 3  
    assert scorer.score_vulnerability('Info') == 1
    
    try:
        _ = scorer.score_vulnerability('Unknown')
    except ValueError as ve:
        print(f"Caught expected error: {ve}")
    
    # The following fails and should raise an exception
    # With a message "Caught expected error: Unknown vulnerability severity level 'unknown'"
    try:
        _ = scorer.score_vulnerability('unknown')
    except ValueError as e:
        assert str(e) == "Unknown vulnerability severity level 'unknown'"

if __name__ == "__main__":
    main()
```
