import os

def create_dashboard_report(input_data):
    """
    This function generates an HTML dashboard report based on the input data.
    
    :param input_data: A dictionary containing the data to be included in the report. Expected structure:
                        {
                            "title": "<report_title>", # String, required
                            "sections": [
                                {"label": "<section_label>", # String, required
                                 "content": "<section_content_html>"}] # String (HTML), optional content can be None
                            "charts": [
                                {"title": "<chart_title>", # String, required
                                 "type": "<chart_type>", # 'bar', 'pie', etc.
                                 "data": ["data_for_chart_1", "data_for_chart_2"]} # List of numerical values for each data point in the chart
                            }, # List of dictionaries, optional 
                        }
    :return: A tuple containing:
                0. success: A boolean indicating whether creating the report was successful.
                1. error_message: If an exception occurred during processing, a descriptive string is returned here; otherwise, None.

            Note that `input_data` dictionary can be generated by various sources such as API calls or databases and may vary in structure
    """
    
    try:
        # Validate input data (This step might involve regex checks, type checking against expected values, etc.)
        
        sections_html = []
        charts_html = []

        if input_data.get('sections'):
            for section_data in input_data['sections']:
                content = section_data.get('content')
                if not content: 
                    continue  # Ignore empty sections
                sections_html.append(f"<section {section_data['label']}>{content}</section>")
        
        if input_data.get('charts'):
            for chart_data in input_data['charts']:   
                charts_html.append(f"<chart {chart_data['title']} type='{chart_data['type']}'><values>{chart_data['data']}</values></chart>") 
        
        html_report = f"""
        <!DOCTYPE html>
        <html lang='en'>
            <head>
                <meta charset='UTF-8'>
                <title>{input_data['title']}</title>
            </head>
            <body style='font-family:sans-serif; background:#eee;padding:.5em 0'>
                {''.join(sections_html)}
                {''.join(charts_html)} 
                <footer style='background:rgb(30, 144, 255); text-align:center; padding:1rem;color:white;font-size:.8rem'>{input_data.get('copyright', 'Creative Commons License')}</footer>
            </body>  
        </html>
        """
        
        with open(os.path.join(input_data['output_dir'] or '.', f'{input_data["title"]}.html'), "w") as report_file:
            report_file.write(html_report)
            
    except Exception as e:
        return False, str(e)

# Example Usage
def example_usage():
    try:
        input_data = {
            'title': 'Performance Dashboard',
            'sections': [
                {'label': 'Product Sales', 'content': '<p>The Product Sale Numbers for Q1 2023</p>'},
                 ],
            'charts': [
                {'title': 'Revenue Breakdown by Category', 
                'type': 'pie',
                'data': [45, 20, 30]},
            ]
        }
        
        success, error = create_dashboard_report(input_data)
        if not success:
            print(f"Failed to generate report: {error}")
        else:
            print(f"Report generated successfully: {input_data['title']}.html")
            
    except Exception as e:
        print(str(e))
  
if __name__ == '__main__':
    example_usage()
