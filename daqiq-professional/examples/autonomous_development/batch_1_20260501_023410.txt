from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
import json

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///scanners.db'
db = SQLAlchemy(app)

class Scanner(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(50), nullable=False)
    description = db.Column(db.Text, nullable=False)
    enabled = db.Column(db.Boolean, default=True)

# Initialize the database
with app.app_context():
    db.create_all()

@app.route('/api/scanners', methods=['POST'])
def create_scanner():
    """
    Create a new scanner entry.
    
    Raises:
        HTTPException: If there was an error creating the entry.
        
    Returns:
        dict: Information about the created entry.
    """
    if not request.json or not all(key in request.json for key in ('name', 'description')):
        raise BadRequest('Bad Request')
  
    new_entry = Scanner(name=request.json['name'], description=json.dumps(request.json['description']))
    db.session.add(new_entry)
    try:
        db.session.commit()
    except Exception as e:
        # Database write failed due to conflict with an entry. 
        # For production, we would most likely want to resolve the conflict first.
        raise SQLAlchemyError('Internal Server Error')
    return {
        'name': new_entry.name,
        'description': request.json['description'],
        'id': new_entry.id
    }

@app.route('/api/scanners', methods=['GET'])
def get_all_scanners():
    """
    Get all enabled scanners.

    Returns:
        list: List of active scanner entries.
    """
    return jsonify(Scanner.query.filter_by(enabled=True).all())

@app.route('/api/scanners/<int:id>', methods=['PUT'])
def update_scanner(id):
    """
    Update a specific scanner.

    Raises:
        HTTPException: If the requested ID is not found in the database or if there are any errors updating.
    
    Returns:
        bool: True if the update was successful, False otherwise.
    """
    if request.method == "PUT":
        target_entry = Scanner.query.filter_by(id=id).first()
        if target_entry is None:
            raise HTTPException('The requested ID was not found.')
        
        try:
            db.session.commit()
            return {'id': id, 'name': target_entry.name, 'description': json.loads(target_entry.description), 'status': True}
        except Exception as e:
            # Database write failed for some reason. The only safe thing to do would be to roll back in such a scenario.
            raise SQLAlchemyError('Unable to update')
    return {'message': "This method should only be accessed via PUT request."}

    if target_entry is None:

@app.route('/api/scanners/<int:id>', methods=['DELETE'])
def delete_scanner(id):
    """
    Delete a specific scanner.

    Parameters:
    - id (int): ID of the scanner entry.
    
    Raises:
        HTTPException: If the requested ID is not found in the database or if there are any errors deleting.
    
    Returns:
        bool: True if the deletion was successful, False otherwise.
    """
    target_entry = Scanner.query.filter_by(id=id).first()
    if id != target_entry.id:
        raise HTTPException('The specified id was not found.')
    
    try:
        db.session.delete(target_entry)
        db.session.commit()
        return {'message': 'Scanner successfully deleted.'}
    except Exception as e:
        raise SQLAlchemyError('Unable to delete')
