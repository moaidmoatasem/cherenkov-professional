import subprocess

class CommandInjectionScanner:
    def __init__(self):
        """
        Initializes the CommandInjectionScanner.
        """

    def scan_command(self, user_input):
        """
        Scans and executes a user-provided command if it does not contain any injection vectors.

        Args:
            user_input (str): The command to be checked for injection.

        Returns:
            str: A message indicating whether the command is safe or contains injection vectors.
        """

        def check_for_injection(command):
            # Define common injection patterns
            forbidden_pattern = 'exec'
            if forbidden_pattern in command.lower():
                return True

            return False

        if not check_for_injection(user_input):
            try:
                subprocess.run(user_input.split(), check=True)
                return f"The command is safe: {user_input}"
            except Exception as e:
                # Handle errors gracefully and provide meaningful feedback
                error_msg = f"User-provided input is potentially harmful. Error: {str(e)}"
        else:
            error_msg = "Injection detected in user-provided input."

        return error_msg

# Example usage
if __name__ == "__main__":
    user_input_safe = 'ls -l'
    scan_result = CommandInjectionScanner().scan_command(user_input_safe)
    print(scan_result)

    user_input_harmful = f"rm {subprocess.check_output(['seq', '-w', '1024']).decode('utf-8')}"
    scan_result_harmful = CommandInjectionScanner().scan_command(user_input_harmful)
    print(scan_result_harmful)
