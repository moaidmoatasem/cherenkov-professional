#!/usr/bin/env python3

"""
FileUploadVulnScanner.py - A tool to scan for insecure file upload vulnerabilities within web applications.
It scans directories where files can be uploaded and checks for potential security issues.
"""

import os
from urllib.parse import urljoin, urlparse
from requests.exceptions import RequestException

class FileUploadVulnScanner:
    """
    This class implements a script to scan a given web application directory for file upload vulnerabilities.
    The main functionalities include checking allowed MIME types and sanitizing user input (path).
    """

    def __init__(self, base_url):
        self.base_url = base_url
        self.allowed_mimes = set()
        self.uploads_dir = None

    @staticmethod
    def _urljoin_base_path(base: str, path: str) -> str:
        """
        Helper method to join URL paths relative to the given base.
        Handles both absolute and relative URLs.
        """
        parsed_base = urlparse(base)
        return urljoin(parsed_base.scheme + "://" + parsed_base.netloc + parsed_base.path,
                       os.path.normpath(os.path.abspath(os.path.join(*os.path.split(parsed_base.path)))))

    def _extract_uploads_dir(self):
        """
        Scans the base application for '/uploads' directory, which is commonly used in file upload endpoints.
        This helps determine if scanning will be done at all. It assumes that '/uploads/files' or similar might exist too.
        """
        payload = "/uploads/"
        response = self._send_request(self.__urljoin_base_path("api", payload))
        return self.__extract_payload_directories_from_response(response.text)

    def __extract_payload_directories_from_response(self, content):
        uploads_dir = []
        for line in content.split("\n"):
            if 'uploads' in line:
                uploads_dir.append('/'.join(line.strip('/').split()))
        return [d for d in set(uploads_dir) if os.path.normpath(os.path.abspath(d)).startswith(os.path.join(self.base_url, self.uploads_dir))]

    def _get_allowed_mimes(self):
        """
        Sends out a request to the allowed MIME-types endpoint of the web application.
        Extracts and returns all MIME types as lowercase for easier matching later.
        """
        try:
            response = self._send_request(urljoin(self.base_url, 'api/mime_types'))
            content_type_header = response.headers.get("Content-Type", "")
            mime_headers = list(filter(lambda v: v.startswith('text'), [content_type_header]))
            if mime_headers:
                self.allowed_mimes.update(x.strip() for x in mime_headers[0].replace('<li>', '').split(', '))
        except RequestException as e:
            raise ValueError(f"Error fetching allowed MIME types: {e}")

    def _send_request(self, url):

            response = requests.get(url)
            response.raise_for_status()
            return response
        except RequestException as e:
            raise SystemError(f'Failed to fetch URL: {url}. Reason: {str(e)}')

    def run_scan(self):
        """
        Initiates the scan by checking for uploads directories and obtaining allowed MIME types.
        A typical usage sequence involves a call like: scanner = Scanner(base_url='http://example.com', uploads_dir='/uploads') 
        where 'base_url' is the root path of your application (typically '/') or '/api' endpoint, 
        and 'uploads_dir' points to the upload directory. Then scanning will attempt to find allowed MIME types.
        """
        self.uploads_dir = self._extract_uploads_dir()
        if len(self.uploads_dir) == 1:
            print(f"Potential file uploads found at: {self.__urljoin_base_path('api', self.uploads_dir[0])}")
            self._get_allowed_mimes()
        else:
            print("No uploads directories are found.")
            return
