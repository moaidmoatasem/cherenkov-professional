import os

class HtmlDashboardReports:
    """
    Generates an HTML report based on various inputs.
    
    Attributes:
        reports_directory (str): The directory where generated reports will be saved.
    
    Methods:
        generate_report(name, data):
            Generates an HTML dashboard report with the given name and data.
            
        clean_up(directory=os.path.join(os.getcwd(), 'temp')):
            Performs necessary cleanup actions after reporting.
    """
    
    def __init__(self, reports_directory):
        self.reports_directory = os.path.abspath(reports_directory)
        
    @staticmethod
    def _create_report_file(name, contents, location):
        """ Helper function to create the HTML report file. """
        with open(os.path.join(location, f"{name}.html"), 'w') as fh:
            fh.write(contents)

    def generate_report(self, name: str, data: dict) -> None:
        """
        Generates an HTML dashboard report.

        Args:
            name (str): The title of the report. Expects filename format.
            data (dict): Data to be included in the reports
        
        Raises:
            ValueError: If `data` does not have expected keys.
        
        Example usage:
        >>> HtmlDashboardReports('/path/to/reports').generate_report('PerformanceReport', {'key1': 'Data1'})
        """
        if not all(key in data for key in ('title', 'content')):
            raise ValueError("The data must contain 'title' and 'content'")
        
        try:
            self._create_report_file(name, self.render_html(data), self.reports_directory)
        except Exception as err:
            print(f"An error occurred while generating the report: {str(err)}")
    
    def render_html(self, data):
        """
        Converts a data dictionary into an HTML file.
        
        Args:
            data (dict): The data to be transformed into HTML
        
        Returns:
            str: Contents of the HTML document.
            
        Example usage:
            >>> report_data = {'title': 'Performance Report', 
                               'content': """<p>Test content here...</p>"""
                              }
            >>> HtmlDashboardReports('/reports').render_html(report_data)
        """
        return f"""<!DOCTYPE html>
                    <html lang="en">
                    <head>
                        <meta charset="UTF-8">
                        <title>{data['title']}</title>
                    </head>
                    <body>
                        {data['content']}
                    </body>
                    </html>"""

    def clean_up(self, directory):
        """
        Cleans up reports file from previous report runs in a given path.
        
        Args:
            directory (str): The directory including the reports
        
        Example usage:
            >>> HtmlDashboardReports('/reports_path').clean_up('/reports_path')
        """
        # Logic to delete .html files inside directory
        for root, _, files in os.walk(directory):
            for file_ in files:
                if file_.endswith('.html'):
                    os.remove(os.path.join(root, file_))
                    
def main():
    reports_directory = '/path/to/reports'
    dashboard_reports = HtmlDashboardReports(reports_directory)
    
    # Example data
    user_data = {
        "title": "Performance Report",
        'content': '<p>Test Data</p>'
    }
    
    try:
        dashboard_reports.generate_report('performance', user_data)
    except Exception as err:
        print(f"An error occurred: {err}")
