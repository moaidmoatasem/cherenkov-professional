class FileUploadVulnerabilityScanner:
    """
    A class to scan for potential file upload vulnerabilities in a web application.
    
    Methods:
        test_file_upload: Scans the specified directory for suspicious patterns associated with file uploads.

    Attributes:
        forbidden_exts (list): List of extensions considered potentially harmful if served as files.
        allowed_exts (set): Set of accepted file types by default. Can be extended in a production environment.
        target_dir (str): The directory to scan for potential upload vulnerabilities.
    """
    
    def __init__(self, target_dir: str):
        self.forbidden_exts = {'.inc', '.php', '.jsp', '.pl', '.asp', '.aspx'}
        self.allowed_exts = set()
        self.target_dir = os.path.abspath(target_dir)
        
    def test_file_upload(self) -> tuple[list, list]:
        """
        Scans the specified directory for files from specific extensions and URLs.

        Returns:
            A tuple containing:
            - List of files found with forbidden extensions.
            - List of files found on invalid or unexpected paths.
        """
        suspicious_files = []
        insecure_paths_files = []

        # Iterate over all files in the target directory
        for root, _, files in os.walk(self.target_dir):
            for file_name in files:
                file_path = os.path.join(root, file_name)
                
                if os.path.islink(file_path):  # Skip symlinked directories, just get their contents
                    continue
                
                urlparts = urlparse(os.path.abspath(file_path))
                extension = os.path.splitext(urlparts.path.rsplit(os.sep, 1)[-1])[1]

                if extension in self.forbidden_exts:
