from flask import Flask, jsonify, request
from pymongo import MongoClient

app = Flask(__name__)

# Database connection setup
MONGODB_URI = "your_mongodb_connection_string"
client = MongoClient(MONGODB_URI)
db = client.scanner_db
scanners_collection = db.scanners

@app.route('/scanners', methods=['POST'])
def create_scanner():
    """
    Create a new scanner entry.
    
    Requests should include the following fields:
    - name (String)
    - ip_address_or_domain (String)
    - enabled (Boolean, default=True)
    
    Returns a formatted JSON response: {'success': True | False, 'message': str}
    """
    data = request.get_json()
    new_scanner = scanners_collection.insert_one(data)
    
    if new_scanner.acknowledged:
        return jsonify({"success": True, "message": f"Scanner created with the ID {new_scanner.inserted_id}"})
    else:
        return jsonify({"success": False, "message": "Failed to create scanner entry"}), 500

@app.route('/scanners/<ObjectId:scanner_id>', methods=['PUT'])
def update_scanner(scanner_id):
    """
    Update an existing scanner with a new configuration.
    
    Returns a formatted JSON response: {'success': True | False, 'message': str}
    """
    updated_count = scanners_collection.update_one({'_id': scanner_id}, {"$set": request.get_json()}).modified_count
    
    if updated_count > 0:
        return jsonify({"success": True, "message": f"Scanner with ID {scanner_id} updated successfully"})
    else:
        return jsonify({"success": False, "message": f"No changes made for Scanner with ID {scanner_id}"}, 400)

@app.route('/scanners/enable', methods=['POST'])
def enable_scanner():
    """
    Enable a scanner
    
    Requests should include the following fields to filter by ip or domain:
    - name (String)
    
    Returns a formatted JSON response: {'success': True | False, 'message': str}
    """
    data = request.get_json()
    updated_count = scanners_collection.update_many(
        {'$and': [{'name': data['name']}, {'enabled': False}]},
        {"$set": {"enabled": True}}
    ).modified_count
    
    if updated_count > 0:
        return jsonify({"success": True, "message": f"All enabled scanners were updated"})
    else:
        if scanners_collection.update_one(
            {'enabled': False},
            {"$set": {"name": data['name'], "enabled": True, "timestamp": request.args.get('timestamp')}}
        ).modified_count:
            return jsonify({"success": True, "message": f"Single enabled scanr was updated"})
        else:
            return jsonify({"success": False, "message": f"No changes made for enabled Scanner with name {data['name']}"}, 400)

@app.route('/scanners/enable', methods=['DELETE'])
def disable_scanner():
    """
    Disable a scanner
    
    Requests should include the following fields to filter by ip or domain:
    - name (String)
    
    Returns a formatted JSON response: {'success': True | False, 'message': str}
    """
    data = request.get_json()
    updated_count = scanners_collection.update_many(
        {'$and': [{'name': data['name']}, {'enabled': True}]},
        {"$set": {"enabled": False}}
    ).modified_count
    
    if updated_count > 0:
        return jsonify({"success": True, "message": f"All disabled scanners were updated"})
    else:
        if scanners_collection.update_one(
            {'enabled': True},
            {"$set": {"name": data['name'], "timestamp": request.args.get('timestamp')}}
        ).modified_count:
            return jsonify({"success": True, "message": f"Single disabled scanr was updated"})
        else:
            return jsonify({"success": False, "message": f"No changes made for enabled Scanner with name {data['name']}"}, 400)

@app.route('/scanners', methods=['GET'])
def get_scanner():
    """
    Get a scanner entry.

    Parameters:
    - name (string): Name of the scanner to retrieve
    - ip_address_or_domain (string): IP address or domain of the device to filter by
    - timestamp (int): Only return disabled scanners as of this time

    Returns: Scans, formatted as `{'name': str, 'ip_address_or_domain': str, 'enabled': bool}`
    """
    name = request.args.get('name')
    ip_addr_or_domain = request.args.get('ip_adress_or_domain')
    timestamp = request.args.get('timestamp')

    if not (name or ip_addr_or_domain):
        return jsonify({'error': "Either name or ip_address_or_domain must be provided"}), 400

    filter_data = {'enabled': True}
    
    if name:
        filter_data['name'] = name
    
    if ip_addr_or_domain: 
        filter_data.update(ip_address_or_domain=ip_addr_or_domain)
    
    if timestamp:
        timestamp_filter = {"timestamp": timestamp}
        filter_data = {"$and": [filter_data, timestamp_filter]}
    
    scanner_results = scanners_collection.find(filter_data).sort("name", 1)
    
    return jsonify([{"name": s["name"], "ip_address_or_domain": s["ip_address_or_domain"], "enabled": s["enabled"]} for s in scanner_results])

if __name__ == '__main__':
    app.run(debug=False)
